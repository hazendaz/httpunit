44      void decodeWithCharacterSetAsArg() throws Exception {
45          String expected = "newpage\u30b5\u30f3\u30d7\u30eb"; // "\u30b5\u30f3\u30d7\u30eb" means "SAMPLE" in Japanese
46          // EUC-JP characterSet
47

49          String actual = HttpUnitUtils.decode(encodedString, "EUC-JP");
50          assertEquals(expected, actual, "decoded string");
51      }

58      @Test
59      void parseContentHeader() throws Exception {
60          String[] headers = { "", "text/plain", "text/html; charset=Cp1252", "text/html; charset=ISO-8859-8",
61                  "text/html; charset=EUC-JP", "text/html charset=windows-1251", "text/html; charset=utf-8",
62                  "text/html; charset = utf-8", "text/html; charset=\"ISO-8859-8\"" };
63          String[][] expected = { { "text/plain", null }, { "text/plain", null }, { "text/html", "Cp1252" },
64                  { "text/html", "ISO-8859-8" }, { "text/html", "EUC-JP" }, { "text/html", "windows-1251" },
65                  { "text/html", "utf-8" }, { "text/html", "utf-8" }, { "text/html", "ISO-8859-8" } };
66          for (int i = 0; i < headers.length; i++) {
67              String[] result = HttpUnitUtils.parseContentTypeHeader(headers[i]);
68              assertEquals(2, result.length);
69              assertEquals(expected[i][0], result[0], "header " + i);
70              assertEquals(expected[i][1], result[1], "header " + i);
71          } // for
72      }

140     @Test
141     void hebrewForm() throws Exception {
142         String hebrewName = "\u05d0\u05d1\u05d2\u05d3";
143         defineResource("HebrewForm.html", "<html><head></head>" + "<form method=POST action=\"SayHello\">"
144                 + "<input type=text name=name><input type=submit></form></body></html>");
145         setResourceCharSet("HebrewForm.html", "ISO-8859-8", true);
146         defineResource("SayHello", new PseudoServlet() {
147             @Override
148             public WebResource getPostResponse() throws UnsupportedEncodingException {
149                 String name = getParameter("name")[0];
150                 WebResource result = new WebResource("<html><body><table><tr><td>Hello, "
151                         + new String(name.getBytes(StandardCharsets.ISO_8859_1), "ISO-8859-8")
152                         + "</td></tr></table></body></html>");
153                 result.setCharacterSet("ISO-8859-8");
154                 result.setSendCharacterSet(true);
155                 return result;
156             }
157         });
158 
159         WebConversation wc = new WebConversation();
160         WebResponse formPage = wc.getResponse(getHostPath() + "/HebrewForm.html");
161         WebForm form = formPage.getForms()[0];
162         WebRequest request = form.getRequest();
163         request.setParameter("name", hebrewName);
164 
165         WebResponse answer = wc.getResponse(request);
166         String[][] cells = answer.getTables()[0].asText();
167 
168         assertEquals("Hello, " + hebrewName, cells[0][0], "Message");
169         assertEquals("ISO-8859-8", answer.getCharacterSet(), "Character set");
170     }
