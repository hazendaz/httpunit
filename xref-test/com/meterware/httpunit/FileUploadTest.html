421 class MimeEcho extends PseudoServlet {
422     @Override
423     public WebResource getPostResponse() {
424         StringBuilder sb = new StringBuilder();
425         try {
426             String contentType = getHeader("Content-Type");
427             DataSource ds = new ByteArrayDataSource(contentType, getBody());
428             MimeMultipart mm = new MimeMultipart(ds);
429             int numParts = mm.getCount();
430             for (int i = 0; i < numParts; i++) {
431                 appendPart(sb, (MimeBodyPart) mm.getBodyPart(i));
432                 if (i < numParts - 1) {
433                     sb.append('&');
434                 }
435             }
436         } catch (MessagingException | IOException e) {
437             e.printStackTrace();
438             sb.append("Oops: ").append(e);
439         }
440 
441         return new WebResource(sb.toString(), "text/plain");
442     }
443 
444     private void appendPart(StringBuilder sb, MimeBodyPart mbp) throws IOException, MessagingException {
445         String[] disposition = mbp.getHeader("Content-Disposition");
446         String name = getHeaderAttribute(disposition[0], "name");
447         if (mbp.getFileName() == null) {
448             appendFieldValue(name, sb, mbp);
449         } else {
450             sb.append(mbp.getContentType()).append(':');
451             appendFileSpecs(name, sb, mbp);
452         }
453     }
454 
455     private void appendFieldValue(String parameterName, StringBuilder sb, MimeBodyPart mbp)
456             throws IOException, MessagingException {
457         sb.append(parameterName).append("=").append(URLEncoder.encode(mbp.getContent().toString()));
458     }
459 
460     private void appendFileSpecs(String parameterName, StringBuilder sb, MimeBodyPart mbp)
461             throws IOException, MessagingException {
462         String filename = mbp.getFileName();
463         filename = filename.substring(filename.lastIndexOf(File.separator) + 1);
464         BufferedReader br = new BufferedReader(new StringReader(mbp.getContent().toString()));
465         int numLines = 0;
466         while (br.readLine() != null) {
467             numLines++;
468         }
469 
470         sb.append(parameterName).append(".name=").append(filename).append("&");
471         sb.append(parameterName).append(".lines=").append(numLines);
472     }
473 
474     private String getHeaderAttribute(String headerValue, String attributeName) {
475         StringTokenizer st = new StringTokenizer(headerValue, ";=", /* returnTokens */ true);
476 
477         int state = 0;
478         String name = "";
479         while (st.hasMoreTokens()) {
480             String token = st.nextToken();
481             if (token.equals(";")) {
482                 state = 1; // next token is attribute name
483             } else if (token.equals("=")) {
484                 if (state == 1) {
485                     state = 2; // next token is attribute value
486                 } else {
487                     state = 0; // reset and keep looking
488                 }
489             } else if (state == 1) {
490                 name = token.trim();
491             } else if (state == 2 && name.equalsIgnoreCase(attributeName)) {
492                 return stripQuotes(token.trim());
493             }
494         }
495         return "";
496     }
497 
498     private String stripQuotes(String value) {
499         if (value.startsWith("\"") && value.endsWith("\"")) {
500             return value.substring(1, value.length() - 1);
501         }
502         return value;
503     }
