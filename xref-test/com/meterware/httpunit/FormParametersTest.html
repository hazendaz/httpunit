475         WebForm form = page.getForms()[0];
476         WebRequest request = page.getForms()[0].getRequest();
477

481         assertTrue(form.isReadOnlyParameter("age"), "'age' should be reported as read-only");
482 
483         assertMatchingSet("selected color", new String[] { "red" }, form.getParameterValues("color"));
484         assertEquals("kangaroo", form.getParameterValue("species"), "selected animal");
485         assertEquals("12", form.getParameterValue("age"), "age");

491         validateSetParameterRejected(request, "color", new String[] { "blue" }, "unchecking 'red'");
