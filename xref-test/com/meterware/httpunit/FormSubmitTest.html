239      * test for bug report [2264431] double submit problem version 1.7 would have problem with double submits
240      */
241     @Test
242     void doubleSubmitProblem() throws Exception {
243         boolean[] states = { false, true };
244         String[] expected = { "", "1" };
245         for (int i = 0; i < states.length; i++) {
246             // countMySelf Tipp from http://www.tipstrs.com/tip/1084/Static-variables-in-Javascript
247             defineWebPage("Default",
248                     "<form method=GET action = \"Default.html\" onsubmit=\"javascript:countMyself();\">"
249                             + "<script type='JavaScript'>\n" + "function countMyself() {\n"
250                             + "  // Check to see if the counter has been initialized\n"
251                             + "  if ( typeof countMyself.counter == 'undefined' ) {\n"
252                             + "      // It has not... perform the initilization\n" + "      countMyself.counter = 0;\n"
253                             + "  }\n" + "\n" + "  // Do something stupid to indicate the value\n"
254                             + "  alert(++countMyself.counter);\n" + "}\n" + "</script>"
255                             + "<Input type=submit name='update' onclick='return " + states[i] + ";'></form>"
256                             + "</form>");
257             WebResponse page = _wc.getResponse(getHostPath() + "/Default.html");
258             WebForm form = page.getForms()[0];
259             form.submit();
260             String alert = _wc.popNextAlert();
261             assertEquals(alert, expected[i], "There should be " + expected[i] + " submits for onclick state '"
262                     + states[i] + "' but there are " + alert);
263         }
264     }
