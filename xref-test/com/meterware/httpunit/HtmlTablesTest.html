368     void cellsWithID() throws Exception {
369         defineWebPage("Default",
370                 "<h2>Interesting data</h2>" + "<table id=\"table\" summary=little>"
371                         + "<tr><td>Title</td><td>Data</td></tr>"
372                         + "<tr><td id=\"id1\">value1</td><td id=\"id2\">value2</td><td>Value</td></tr>"
373                         + "<tr><td>&nbsp;</td><td>&nbsp;</td><td>Value</td></tr>" + "</table>");
374 
375         WebResponse page = _wc.getResponse(getHostPath() + "/Default.html");
376         WebTable table = page.getTableWithID("table");
377         assertNotNull(table, "there is a table");
378         TableCell cell = table.getTableCellWithID("id1");
379         assertNotNull(cell, "cell id1");
380         assertEquals("value1", cell.getText(), "Value of cell id1");
381         cell = table.getTableCellWithID("id2");
382         assertNotNull(cell, "cell id2");
383         assertEquals("value2", cell.getText(), "Value of cell id2");
384 
385         // test non existent cell id
386         cell = table.getTableCellWithID("nonExistingID");
387         assertNull(cell, "cell id2");
388 
389         cell = (TableCell) page.getElementWithID("id1");
390         assertEquals("value1", cell.getText(), "value of cell found from page");
391     }
