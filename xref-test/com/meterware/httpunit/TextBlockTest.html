37      /**
38       * Paragraph detection.
39       *
40       * @throws Exception
41       *             the exception
42       */
43      @Test
44      void paragraphDetection() throws Exception {
45          defineWebPage("SimplePage",
46                  "<p>This has no forms or links since we don't care " + "about them</p>"
47                          + "<p class='comment'>But it does have three paragraphs</p>\n"
48                          + "<p>Which is what we want to find</p>");
49          WebConversation wc = new WebConversation();
50          WebResponse response = wc.getResponse(getHostPath() + "/SimplePage.html");
51          assertEquals(3, response.getTextBlocks().length, "Number of paragraphs");
52          assertEquals("This has no forms or links since we don't care about them", response.getTextBlocks()[0].getText(),
53                  "First paragraph");
54          BlockElement comment = response.getFirstMatchingTextBlock(TextBlock.MATCH_CLASS, "comment");
55          assertNotNull(comment, "Did not find a comment paragraph");
56          assertEquals("But it does have three paragraphs", comment.getText(), "Comment paragraph");
57      }

60       * Text conversion.
61       *
62       * @throws Exception
63       *             the exception
64       */
65      @Test
66      void textConversion() throws Exception {
67          defineWebPage("SimplePage", "<p>Here is a line<br>followed by another</p>");
68          WebConversation wc = new WebConversation();
69          WebResponse response = wc.getResponse(getHostPath() + "/SimplePage.html");
70          BufferedReader br = new BufferedReader(new StringReader(response.getTextBlocks()[0].getText()));
71          assertEquals("Here is a line", br.readLine(), "First line");
72          assertEquals("followed by another", br.readLine(), "Second line");
73          br.readLine();
74      }

76      /**
77       * Header detection.
78       *
79       * @throws Exception
80       *             the exception
81       */
82      @Test
83      void headerDetection() throws Exception {
84          defineWebPage("SimplePage",
85                  "<h1>Here is a section</h1>\n" + "with some text" + "<h2>A subsection</h2>" + "<p>Some more text</p>");
86          WebConversation wc = new WebConversation();
87          WebResponse response = wc.getResponse(getHostPath() + "/SimplePage.html");
88          TextBlock header1 = response.getFirstMatchingTextBlock(TextBlock.MATCH_TAG, "H1");
89          assertNotNull(header1, "Did not find the H1 header");
90          assertEquals("Here is a section", header1.getText(), "H1 header");
91          TextBlock header2 = response.getFirstMatchingTextBlock(TextBlock.MATCH_TAG, "h2");
92          assertNotNull(header2, "Did not find the h2 header");
93          assertEquals("A subsection", header2.getText(), "H2 header");
94          assertEquals("with some text", response.getNextTextBlock(header1).getText(), "Text under header 1");
95      }

97      /**
98       * Embedded links.
99       *
100      * @throws Exception
101      *             the exception
102      */
103     @Test
104     void embeddedLinks() throws Exception {
105         defineWebPage("SimplePage", "<h1>Here is a section</h1>\n"
106                 + "<p>with a <a id='httpunit' href='http://httpunit.org'>link to the home page</a></p>");
107         WebConversation wc = new WebConversation();
108         WebResponse response = wc.getResponse(getHostPath() + "/SimplePage.html");
109         BlockElement paragraph = response.getTextBlocks()[1];
110         assertNotNull(paragraph.getLinks(), "Did not retrieve any links");
111         assertNotNull(paragraph.getLinkWithID("httpunit"), "Did not find the httpunit link");
112         assertNull(response.getTextBlocks()[0].getLinkWithID("httpunit"),
113                 "Should not have found the httpunit link in the header");
114         assertNotNull(paragraph.getFirstMatchingLink(WebLink.MATCH_CONTAINED_TEXT, "home page"),
115                 "Did not find the home page link");
116         assertEquals("http://httpunit.org", paragraph.getLinkWithID("httpunit").getRequest().getURL().toExternalForm(),
117                 "embedded link url");
118     }
