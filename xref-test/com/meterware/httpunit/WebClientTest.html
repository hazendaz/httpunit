101     @Test
102     void undefinedResource() throws IOException {
103         boolean originalState = HttpUnitOptions.getExceptionsThrownOnErrorStatus();
104         // try two cases for throwException true on i==0, false on i==1
105         for (int i = 0; i < 2; i++) {
106             boolean throwException = i == 0;
107             HttpUnitOptions.setExceptionsThrownOnErrorStatus(throwException);
108             WebResponse response = null;
109             try {
110                 WebConversation wc = new WebConversation();
111                 WebRequest request = new GetMethodWebRequest(getHostPath() + "/undefined");
112                 response = wc.getResponse(request);
113                 if (throwException) {
114                     fail("there should have been an exception here");
115                 }
116             } catch (HttpNotFoundException hnfe) {
117                 assertTrue(throwException);
118                 response = hnfe.getResponse();
119             } catch (Exception e) {
120                 fail("there should be no exception here");
121             }
122             assertTrue(response != null);
123             assertEquals(HttpURLConnection.HTTP_NOT_FOUND, response.getResponseCode());
124             if (throwException) {
125                 assertEquals("", response.getText(), "with throwException=" + throwException);
126                 assertEquals("unable to find /undefined", response.getResponseMessage(),
127                         "with throwException=" + throwException);
128             } else {
129                 // FIXME what do we expect here and how do we get it!
130                 assertEquals("unable to find /undefined", response.getText(), "with throwException=" + throwException);
131                 assertNull(response.getResponseMessage());
132             }
133         }
134         HttpUnitOptions.setExceptionsThrownOnErrorStatus(originalState);
135     }

344     /**
345      * test for 1799532 ] Patched CookieJar for dealing with empty cookies I had a problem testing a web app that sent
346      * empty cookie values to the client.
347      * <p/>
348      * Within the same http response, a cookie was set by the web app twice, once with some non-empty value and the
349      * second time with an empty value. The intention of the web app obviously was to delete the cookie, wich actually
350      * occurs in IE and Firefox.
351      * <p/>
352      * In HttpUnit though the cookie was stored with the empty value and sent back to the server in the next request.
353      * That confused the web app and caused application errors, because it didn't expect that cookie back with the empty
354      * value.
355      * <p/>
356      * I cannot really judge if this is actually a bug in HttpUnit or just more strict appliance of protocols than real
357      * Browsers do. Of course it is at least an ugly behaviour of the web app, but that wasn't my focus. It also seems
358      * from forums that others might have had the same problem.
359      * <p/>
360      * To over come that problem, I had to change the CookieJar class in HttpUnit
361      * <p/>
362      * to not store cookies with empty values, but remove them completely instead. The modified source file is attached
363      * (based on HttpUnit 1.6.2).
364      * <p/>
365      * Russel, please check if you want to integrate this modification in a future HttpUnit Release. 2007-12-28:
366      * wf@bitplan.com - looked at http://www.ietf.org/rfc/rfc2109.txt there is no statement about empty value handling
367      * -the standard asks for opaque handling ... no mocking about with values by the server ...
368      *
369      * @throws Exception
370      *             the exception
371      */
372     @Test
373     void emptyCookie() throws Exception {
374         defineResource("bounce", new CookieEcho());
375         WebConversation wc = new WebConversation();
376         wc.putCookie("EMPTYVALUE", "non-empty");
377         WebResponse response = wc.getResponse(getHostPath() + "/bounce");
378         assertEquals("EMPTYVALUE=non-empty", response.getText(), "Cookies sent");
379         wc.putCookie("SOMECOOKIE", "some value");
380         wc.putCookie("EMPTYVALUE", null);
381         response = wc.getResponse(getHostPath() + "/bounce");
382         // System.err.println(response.getText());
383         String[] names = wc.getCookieNames();
384         // should we expect a 1 or a two here?
385         // see also testCookies where 8 is currently the correct value and 7 would

389         assertEquals(1, names.length, "number of cookies");
390         // for (int i=0;i<names.length;i++) System.err.println(names[i]);

611                 if (withOpaque) {

615                 return resource;
616             }
617         });
618         WebConversation wc = new WebConversation();

620         WebResponse wr = wc.getResponse(getHostPath() + "/dir/index.html");
621         StringBuilder expectedHeaderStr = new StringBuilder("Digest username=\"Mufasa\",")
622                 .append("       realm=\"testrealm@host.com\",")

628         }
629         HttpHeader expectedHeader = new HttpHeader(expectedHeaderStr.toString());

718      * Assert headers equals.
719      *
720      * @param expectedHeader
721      *            the expected header
722      * @param actualHeader
723      *            the actual header
724      */
725     private void assertHeadersEquals(HttpHeader expectedHeader, HttpHeader actualHeader) {
726         assertEquals(expectedHeader.getLabel(), actualHeader.getLabel(), "Authentication type");
727         if (!expectedHeader.equals(actualHeader)) {
728             Deltas deltas = new Deltas();
729             Set actualKeys = new HashSet(actualHeader.getProperties().keySet());
730             for (Iterator eachKey = expectedHeader.getProperties().keySet().iterator(); eachKey.hasNext();) {
731                 String key = (String) eachKey.next();
732                 if (!actualKeys.contains(key)) {
733                     deltas.addMissingValue(key, expectedHeader.getProperty(key));
734                 } else {
735                     actualKeys.remove(key);
736                     deltas.compareValues(key, expectedHeader.getProperty(key), actualHeader.getProperty(key));
737                 }
738             }
739             for (Iterator eachKey = actualKeys.iterator(); eachKey.hasNext();) {
740                 String key = (String) eachKey.next();
741                 deltas.addExtraValue(key, actualHeader.getProperty(key));
742             }
743             fail("Header not as expected: " + deltas);
744         }
745     }

747     /**
748      * The Class Deltas.
749      */
750     static class Deltas {
751 
752         /** The missing values. */
753         private ArrayList _missingValues = new ArrayList<>();
754 
755         /** The extra values. */
756         private ArrayList _extraValues = new ArrayList<>();
757 
758         @Override
759         public String toString() {
760             StringBuilder sb = new StringBuilder();
761             if (!_missingValues.isEmpty()) {
762                 sb.append("missing: ").append(_missingValues);
763             }
764             if (!_extraValues.isEmpty()) {
765                 sb.append("extra: ").append(_extraValues);
766             }
767             return sb.toString();
768         }
769 
770         /**
771          * Adds the missing value.
772          *
773          * @param key
774          *            the key
775          * @param value
776          *            the value
777          */
778         void addMissingValue(Object key, Object value) {
779             _missingValues.add(key + "=" + value);
780         }
781 
782         /**
783          * Adds the extra value.
784          *
785          * @param key
786          *            the key
787          * @param value
788          *            the value
789          */
790         void addExtraValue(Object key, Object value) {
791             _extraValues.add(key + "=" + value);
792         }
793 
794         /**
795          * Compare values.
796          *
797          * @param key
798          *            the key
799          * @param expected
800          *            the expected
801          * @param actual
802          *            the actual
803          */
804         void compareValues(Object key, Object expected, Object actual) {
805             if (!expected.equals(actual)) {
806                 addMissingValue(key, expected);
807                 addExtraValue(key, actual);
808             }
809         }

1172 
1173         /** The response text. */
1174         private String _responseText;
1175 
1176         /** The suppress length header. */
1177         private boolean _suppressLengthHeader;
1178 
1179         /**
1180          * Instantiates a new compressed pseudo servlet.
1181          *
1182          * @param responseText
1183          *            the response text
1184          */
1185         public CompressedPseudoServlet(String responseText) {
1186             _responseText = responseText;
1187         }

1189         /**
1190          * Instantiates a new compressed pseudo servlet.
1191          *
1192          * @param responseText
1193          *            the response text
1194          * @param suppressLengthHeader
1195          *            the suppress length header
1196          */
1197         public CompressedPseudoServlet(String responseText, boolean suppressLengthHeader) {
1198             this(responseText);
1199             _suppressLengthHeader = suppressLengthHeader;
1200         }
1201 
1202         @Override
1203         public WebResource getGetResponse() throws IOException {
1204             if (!userAcceptsGZIP()) {
1205                 return new WebResource(_responseText.getBytes(StandardCharsets.UTF_8), "text/plain");
1206             }
1207             WebResource result = new WebResource(getCompressedContents(), "text/plain");
1208             if (_suppressLengthHeader) {
1209                 result.suppressAutomaticLengthHeader();
1210             }
1211             result.addHeader("Content-Encoding: gzip");
1212             return result;
1213         }

1215         /**
1216          * User accepts GZIP.
1217          *
1218          * @return true, if successful
1219          */
1220         private boolean userAcceptsGZIP() {
1221             String header = getHeader("Accept-Encoding");
1222             if (header == null) {
1223                 return false;
1224             }
1225             return header.toLowerCase(Locale.ENGLISH).indexOf("gzip") >= 0;
1226         }

1229          * Gets the compressed contents.
1230          *
1231          * @return the compressed contents
1232          *
1233          * @throws IOException
1234          *             Signals that an I/O exception has occurred.
1235          */
1236         private byte[] getCompressedContents() throws IOException {
1237             ByteArrayOutputStream baos = new ByteArrayOutputStream();
1238             GZIPOutputStream gzip = new GZIPOutputStream(baos);
1239             try (OutputStreamWriter out = new OutputStreamWriter(gzip, StandardCharsets.UTF_8)) {
1240                 out.write(_responseText);
1241             }
1242             return baos.toByteArray();
1243         }
