97          for (int i = 0; i < 2; i++) {
98              boolean throwException = i == 0;

100             WebResponse response = null;
101             try {
102                 WebConversation wc = new WebConversation();
103                 WebRequest request = new GetMethodWebRequest(getHostPath() + "/undefined");
104                 response = wc.getResponse(request);
105                 if (throwException) {
106                     fail("there should have been an exception here");
107                 }
108             } catch (HttpNotFoundException hnfe) {
109                 assertTrue(throwException);
110                 response = hnfe.getResponse();
111             } catch (Exception e) {
112                 fail("there should be no exception here");
113             }
114             assertTrue(response != null);
115             assertEquals(HttpURLConnection.HTTP_NOT_FOUND, response.getResponseCode());
116             if (throwException) {
117                 assertEquals("", response.getText(), "with throwException=" + throwException);
118                 assertEquals("unable to find /undefined", response.getResponseMessage(),
119                         "with throwException=" + throwException);
120             } else {
121                 // FIXME what do we expect here and how do we get it!
122                 assertEquals("unable to find /undefined", response.getText(), "with throwException=" + throwException);
123                 assertNull(response.getResponseMessage());
124             }
125         }
126         HttpUnitOptions.setExceptionsThrownOnErrorStatus(originalState);

292     /**
293      * test for 1799532 ] Patched CookieJar for dealing with empty cookies I had a problem testing a web app that sent
294      * empty cookie values to the client.
295      * <p/>
296      * Within the same http response, a cookie was set by the web app twice, once with some non-empty value and the
297      * second time with an empty value. The intention of the web app obviously was to delete the cookie, wich actually
298      * occurs in IE and Firefox.
299      * <p/>
300      * In HttpUnit though the cookie was stored with the empty value and sent back to the server in the next request.
301      * That confused the web app and caused application errors, because it didn't expect that cookie back with the empty
302      * value.
303      * <p/>
304      * I cannot really judge if this is actually a bug in HttpUnit or just more strict appliance of protocols than real
305      * Browsers do. Of course it is at least an ugly behaviour of the web app, but that wasn't my focus. It also seems
306      * from forums that others might have had the same problem.
307      * <p/>
308      * To over come that problem, I had to change the CookieJar class in HttpUnit
309      * <p/>
310      * to not store cookies with empty values, but remove them completely instead. The modified source file is attached
311      * (based on HttpUnit 1.6.2).
312      * <p/>
313      * Russel, please check if you want to integrate this modification in a future HttpUnit Release. 2007-12-28:
314      * wf@bitplan.com - looked at http://www.ietf.org/rfc/rfc2109.txt there is no statement about empty value handling
315      * -the standard asks for opaque handling ... no mocking about with values by the server ...
316      *
317      * @throws Exception
318      */
319     @Test
320     void testEmptyCookie() throws Exception {
321         defineResource("bounce", new CookieEcho());
322         WebConversation wc = new WebConversation();
323         wc.putCookie("EMPTYVALUE", "non-empty");
324         WebResponse response = wc.getResponse(getHostPath() + "/bounce");
325         assertEquals("EMPTYVALUE=non-empty", response.getText(), "Cookies sent");
326         wc.putCookie("SOMECOOKIE", "some value");
327         wc.putCookie("EMPTYVALUE", null);
328         response = wc.getResponse(getHostPath() + "/bounce");
329         // System.err.println(response.getText());
330         String[] names = wc.getCookieNames();
331         // should we expect a 1 or a two here?
332         // see also testCookies where 8 is currently the correct value and 7 would
333         // be if we handle empty strings as cookie deletions
334         // as long as 1799532 is rejected we'll go for a 2 here ...
335         // [ 1371208 ] Patch for Cookie bug #1371204 has a solution to use 1 ...
336         assertEquals(1, names.length, "number of cookies");
337         // for (int i=0;i<names.length;i++) System.err.println(names[i]);
338     }

624     private void assertHeadersEquals(HttpHeader expectedHeader, HttpHeader actualHeader) {
625         assertEquals(expectedHeader.getLabel(), actualHeader.getLabel(), "Authentication type");
626         if (!expectedHeader.equals(actualHeader)) {
627             Deltas deltas = new Deltas();
628             Set actualKeys = new HashSet(actualHeader.getProperties().keySet());
629             for (Iterator eachKey = expectedHeader.getProperties().keySet().iterator(); eachKey.hasNext();) {
630                 String key = (String) eachKey.next();
631                 if (!actualKeys.contains(key)) {
632                     deltas.addMissingValue(key, expectedHeader.getProperty(key));
633                 } else {
634                     actualKeys.remove(key);
635                     deltas.compareValues(key, expectedHeader.getProperty(key), actualHeader.getProperty(key));
636                 }
637             }
638             for (Iterator eachKey = actualKeys.iterator(); eachKey.hasNext();) {
639                 String key = (String) eachKey.next();
640                 deltas.addExtraValue(key, actualHeader.getProperty(key));
641             }
642             fail("Header not as expected: " + deltas);
643         }
644     }

646     static class Deltas {
647         private ArrayList _missingValues = new ArrayList();
648         private ArrayList _extraValues = new ArrayList();
649 
650         @Override
651         public String toString() {
652             StringBuilder sb = new StringBuilder();
653             if (!_missingValues.isEmpty()) {
654                 sb.append("missing: ").append(_missingValues);
655             }
656             if (!_extraValues.isEmpty()) {
657                 sb.append("extra: ").append(_extraValues);
658             }
659             return sb.toString();
660         }
661 
662         void addMissingValue(Object key, Object value) {
663             _missingValues.add(key + "=" + value);
664         }
665 
666         void addExtraValue(Object key, Object value) {
667             _extraValues.add(key + "=" + value);
668         }
669 
670         void compareValues(Object key, Object expected, Object actual) {
671             if (!expected.equals(actual)) {
672                 addMissingValue(key, expected);
673                 addExtraValue(key, actual);
674             }
675         }
676     }

1014     private class CompressedPseudoServlet extends PseudoServlet {
1015 
1016         private String _responseText;
1017         private boolean _suppressLengthHeader;
1018 
1019         public CompressedPseudoServlet(String responseText) {
1020             _responseText = responseText;
1021         }
1022 
1023         public CompressedPseudoServlet(String responseText, boolean suppressLengthHeader) {
1024             this(responseText);
1025             _suppressLengthHeader = suppressLengthHeader;
1026         }
1027 
1028         @Override
1029         public WebResource getGetResponse() throws IOException {
1030             if (!userAcceptsGZIP()) {
1031                 return new WebResource(_responseText.getBytes(StandardCharsets.UTF_8), "text/plain");
1032             }
1033             WebResource result = new WebResource(getCompressedContents(), "text/plain");
1034             if (_suppressLengthHeader) {
1035                 result.suppressAutomaticLengthHeader();
1036             }
1037             result.addHeader("Content-Encoding: gzip");
1038             return result;
1039         }
1040 
1041         private boolean userAcceptsGZIP() {
1042             String header = getHeader("Accept-Encoding");
1043             if (header == null) {
1044                 return false;
1045             }
1046             return header.toLowerCase().indexOf("gzip") >= 0;
1047         }
1048 
1049         private byte[] getCompressedContents() throws IOException {
1050             ByteArrayOutputStream baos = new ByteArrayOutputStream();
1051             GZIPOutputStream gzip = new GZIPOutputStream(baos);
1052             OutputStreamWriter out = new OutputStreamWriter(gzip);
1053             out.write(_responseText);
1054             out.flush();
1055             out.close();
1056             return baos.toByteArray();
1057         }
