285     /**
286      * Gets the name of frame with URL.
287      *
288      * @param wc
289      *            the wc
290      * @param urlString
291      *            the url string
292      *
293      * @return the name of frame with URL
294      */
295     private String getNameOfFrameWithURL(WebConversation wc, String urlString) {
296         String[] names = wc.getFrameNames();
297         for (String name : names) {
298             WebResponse candidate = wc.getFrameContents(name);
299             if (candidate.getURL().toExternalForm().indexOf(urlString) >= 0) {
300                 return name;
301             }
302         }
303         return null;
304     }
305 
306     /**
307      * Gets the frame with URL.
308      *
309      * @param wc
310      *            the wc
311      * @param urlString
312      *            the url string
313      *
314      * @return the frame with URL
315      */
316     private WebResponse getFrameWithURL(WebConversation wc, String urlString) {
317         String name = getNameOfFrameWithURL(wc, urlString);
318         if (name == null) {
319             return null;
320         }
321         return wc.getFrameContents(name);
322     }
