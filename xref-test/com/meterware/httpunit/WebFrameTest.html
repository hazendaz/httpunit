210     private String getNameOfFrameWithURL(WebConversation wc, String urlString) {
211         String[] names = wc.getFrameNames();
212         for (String name : names) {
213             WebResponse candidate = wc.getFrameContents(name);
214             if (candidate.getURL().toExternalForm().indexOf(urlString) >= 0) {
215                 return name;
216             }
217         }
218         return null;
219     }
220 
221     private WebResponse getFrameWithURL(WebConversation wc, String urlString) {
222         String name = getNameOfFrameWithURL(wc, urlString);
223         if (name == null) {
224             return null;
225         }
226         return wc.getFrameContents(name);
227     }
