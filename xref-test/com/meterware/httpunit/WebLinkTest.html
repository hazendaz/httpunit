344     @Test
345     void testParametersOnLinks() throws Exception {
346         defineResource("ParameterLinks.html",
347                 "<html><head><title>Param on Link Page</title></head>\n" + "<body>"
348                         + "<a href=\"/other.html\">no parameter link</A>\n"
349                         + "<a href=\"/other.html?param1=value1\">one parameter link</A>\n"
350                         + "<a href=\"/other.html?param1=value1&param2=value2\">two parameters link</A>\n"
351                         + "<a href=\"/other.html?param1=value1&param1=value3\">two values link</A>\n"
352                         + "<a href=\"/other.html?param1=value1&param2=value2&param1=value3\">two values link</A>\n"
353                         + "</body></html>\n");
354         WebConversation wc = new WebConversation();
355 
356         WebLink[] links = wc.getResponse(getHostPath() + "/ParameterLinks.html").getLinks();
357         assertNotNull(links);
358         assertEquals(5, links.length, "number of links");
359         WebRequest request;
360 
361         // first link should not have any param
362         request = links[0].getRequest();
363         assertNotNull(request);
364         String[] names = request.getRequestParameterNames();
365         assertNotNull(names);
366         assertEquals(0, names.length, "Num parameters found");
367         assertEquals("", request.getParameter("nonexistent"), "Non Existent parameter should be empty");
368 
369         // second link should have one parameter
370         checkLinkParameters(links[1], new String[] { "param1" }, new String[][] { { "value1" } });
371 
372         // third link should have 2 parameters. !! Order of parameters cannot be guaranted.
373         checkLinkParameters(links[2], new String[] { "param1", "param2" },
374                 new String[][] { { "value1" }, { "value2" } });
375 
376         // fourth link should have 1 parameter with 2 values.
377         checkLinkParameters(links[3], new String[] { "param1" }, new String[][] { { "value1", "value3" } });
378 
379         // fifth link should have 2 parameters with one with 2 values.
380         checkLinkParameters(links[4], new String[] { "param1", "param2" },
381                 new String[][] { { "value1", "value3" }, { "value2" } });
382     }
383 
384     private void checkLinkParameters(WebLink link, String[] expectedNames, String[][] expectedValues) {
385         WebRequest request = link.getRequest();
386         assertNotNull(request);
387         assertMatchingSet("Parameter names", expectedNames, request.getRequestParameterNames());
388         for (int i = 0; i < expectedValues.length; i++) {
389             assertMatchingSet(expectedNames[i] + " values", expectedValues[i],
390                     request.getParameterValues(expectedNames[i]));
391         }
392     }
