73      /**
74       * test for Bug report 1908117 by firebird74 http://www.w3.org/Addressing/URL/url-spec.html says Spaces and control
75       * characters in URLs must be escaped for transmission in HTTP so %20 ist used for space/blank trying out <html>
76       * <body> <a href="http://www.bit plan.com">BITlan blank link</a> </body> </html> in different browers gives
77       * different results. Firefox 2.0.0.13 will not show a link at all Internet Explorer 6.0 will show a link that has
78       * http://www.bit%20plan.com as it's target
79       *
80       * @throws Exception
81       *             the exception
82       */
83      @Test
84      void linkToURLWithBlanks() throws Exception {
85          defineWebPage("urlwithblank", "<a href='http://bla.fasel.com/a/b/lorem ipsum.pdf'>link with blank</a>"
86                  + "<a href='http://bla.fasel.com/a/b/lorem%20ipsum.pdf'>link with blank</a>");
87          WebConversation wc = new WebConversation();
88          WebResponse resp = wc.getResponse(getHostPath() + "/urlwithblank.html");
89          WebLink[] webLinks = resp.getLinks();
90          assertEquals(2, webLinks.length, "There should be two link but there are " + webLinks.length);
91          // TODO this is what we expect in fact
92          // String blankLink1="http://bla.fasel.com/a/b/lorem%20ipsum.pdf";
93          String blankLink1 = "http://bla.fasel.com/a/b/loremipsum.pdf";
94          String blankLink2 = "http://bla.fasel.com/a/b/lorem%20ipsum.pdf";
95          WebLink link1 = webLinks[0];
96          assertEquals(link1.getURLString(), blankLink1,
97                  "the blank in the link1 should be converted but we got '" + link1.getURLString() + "'");
98          WebLink link2 = webLinks[1];
99          assertEquals(link2.getURLString(), blankLink2,
100                 "the blank %20 in the link2 should not be converted but we got '" + link2.getURLString() + "'");
101     }

444     void parametersOnLinks() throws Exception {
445         defineResource("ParameterLinks.html",
446                 "<html><head><title>Param on Link Page</title></head>\n" + "<body>"
447                         + "<a href=\"/other.html\">no parameter link</A>\n"
448                         + "<a href=\"/other.html?param1=value1\">one parameter link</A>\n"
449                         + "<a href=\"/other.html?param1=value1&param2=value2\">two parameters link</A>\n"
450                         + "<a href=\"/other.html?param1=value1&param1=value3\">two values link</A>\n"
451                         + "<a href=\"/other.html?param1=value1&param2=value2&param1=value3\">two values link</A>\n"
452                         + "</body></html>\n");
453         WebConversation wc = new WebConversation();
454 
455         WebLink[] links = wc.getResponse(getHostPath() + "/ParameterLinks.html").getLinks();
456         assertNotNull(links);
457         assertEquals(5, links.length, "number of links");
458         WebRequest request;
459 
460         // first link should not have any param
461         request = links[0].getRequest();
462         assertNotNull(request);
463         String[] names = request.getRequestParameterNames();
464         assertNotNull(names);
465         assertEquals(0, names.length, "Num parameters found");
466         assertEquals("", request.getParameter("nonexistent"), "Non Existent parameter should be empty");
467 
468         // second link should have one parameter
469         checkLinkParameters(links[1], new String[] { "param1" }, new String[][] { { "value1" } });
470 
471         // third link should have 2 parameters. !! Order of parameters cannot be guaranted.
472         checkLinkParameters(links[2], new String[] { "param1", "param2" },
473                 new String[][] { { "value1" }, { "value2" } });
474 
475         // fourth link should have 1 parameter with 2 values.
476         checkLinkParameters(links[3], new String[] { "param1" }, new String[][] { { "value1", "value3" } });
477 
478         // fifth link should have 2 parameters with one with 2 values.
479         checkLinkParameters(links[4], new String[] { "param1", "param2" },
480                 new String[][] { { "value1", "value3" }, { "value2" } });
481     }

483     /**
484      * Check link parameters.
485      *
486      * @param link
487      *            the link
488      * @param expectedNames
489      *            the expected names
490      * @param expectedValues
491      *            the expected values
492      */
493     private void checkLinkParameters(WebLink link, String[] expectedNames, String[][] expectedValues) {
494         WebRequest request = link.getRequest();
495         assertNotNull(request);
496         assertMatchingSet("Parameter names", expectedNames, request.getRequestParameterNames());
497         for (int i = 0; i < expectedValues.length; i++) {
498             assertMatchingSet(expectedNames[i] + " values", expectedValues[i],
499                     request.getParameterValues(expectedNames[i]));
500         }
501     }
