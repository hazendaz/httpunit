88          defineResource("TextPage.txt", "Just text", "text/plain");
89          defineResource("SimplePage.html",
90                  "<html><head><title>A Sample Page</title></head><body>Something here</body></html>", "text/html");
91          defineResource("StructuredPage.html",
92                  "<!DOCTYPE html PUBLIC '-//W3C//DTD XHTML Basic 1.0//EN' 'http://www.w3.org/TR/xhtml-basic/xhtml-basic10.dtd'>"
93                          + "<html><head><title>A Structured Page</title></head><body>Something here</body></html>",
94                  "text/xhtml");
95          defineResource("XHTMLPage.html",
96                  "<!DOCTYPE html PUBLIC '-//W3C//DTD XHTML Basic 1.0//EN' 'http://www.w3.org/TR/xhtml-basic/xhtml-basic10.dtd'>"
97                          + "<html><head><title>An XHTML Page</title></head><body>Something here</body></html>",
98                  "application/xhtml+xml");
99          // see http://www.xmlrpc.com/spec
100         defineResource("XMLRPC.html",
101                 "<?xml version=\"1.0\"?><methodCall><methodName>stock.getQuote</methodName><params><param><value>JAVA</value></param></params></methodCall>",
102                 "text/xml");
103         WebConversation wc = new WebConversation();
104         HttpUnitOptions.setCheckHtmlContentType(true);
105         try {
106             wc.getResponse(getHostPath() + "/TextPage.txt").getReceivedPage().getTitle();
107             fail("Should have rejected attempt to get a title from a text page");
108         } catch (NotHTMLException e) {
109         }
110 
111         WebResponse simplePage = wc.getResponse(getHostPath() + "/SimplePage.html");
112         assertEquals("A Sample Page", simplePage.getReceivedPage().getTitle(), "HTML Title");
113 
114         WebResponse structuredPage = wc.getResponse(getHostPath() + "/StructuredPage.html");
115         assertEquals("A Structured Page", structuredPage.getReceivedPage().getTitle(), "XHTML Title");
116         Document root = structuredPage.getDOM();
117         assertTrue(root != null, "document root of structured  Page should be available");
118 
119         WebResponse xhtmlPage = wc.getResponse(getHostPath() + "/XHTMLPage.html");
120         assertEquals("An XHTML Page", xhtmlPage.getReceivedPage().getTitle(), "XHTML Title");
121 
122         WebResponse xmlrpcPage = wc.getResponse(getHostPath() + "/XMLRPC.html");
123         root = xmlrpcPage.getDOM();
124         assertTrue(root != null, "document root of xml RPC page should be available");
125         NodeList elements = root.getElementsByTagName("methodCall");
126         assertEquals(1, elements.getLength(), "there should be one methodCall node");
127 
128     }

599     /**
600      * Create a fragment of HTML defining JavaScript that writes a document into a different window.
601      *
602      * @param document
603      *            the document to be written
604      * @param targetWindow
605      *            the name of the target window to open
606      *
607      * @return a fragment of HTML text
608      */
609     private String scriptToWriteAnotherDocument(String document, String targetWindow) {
610         StringBuilder buff = new StringBuilder();
611         buff.append("<script language=\"JavaScript\">\n");
612         buff.append("target = window.open('', '").append(targetWindow).append("');\n");
613         buff.append("target.document.write('").append(document).append("');\n");
614         buff.append("target.document.close();\n");
615         buff.append("</script>\n");
616         return buff.toString();
617     }
