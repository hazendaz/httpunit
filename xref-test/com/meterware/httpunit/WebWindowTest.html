41      /**
42       * Verifies that clicking on a link that specifies the _blank target creates a new window, populated with the
43       * contents of the referenced page.
44       *
45       * @throws Exception
46       *             on any unexpected problem.
47       */
48      @Test
49      void testNewTarget() throws Exception {
50          defineResource("goHere", "You made it!");
51          defineWebPage("start", "<a href='goHere' id='go' target='_blank'>here</a>");
52  
53          WebClient wc = new WebConversation();
54          assertEquals(1, wc.getOpenWindows().length, "Number of initial windows");
55          WebWindow main = wc.getMainWindow();
56          WebResponse initialPage = main.getResponse(getHostPath() + "/start.html");
57          initialPage.getLinkWithID("go").click();
58          assertEquals(2, wc.getOpenWindows().length, "Number of windows after following link");
59          assertEquals(initialPage, main.getCurrentPage(), "Main page in original window");
60          WebWindow other = wc.getOpenWindows()[1];
61          assertEquals("You made it!", other.getCurrentPage().getText(), "New window contents");
62  
63          main.close();
64          assertTrue(main.isClosed(), "Original main window is not closed");
65          assertFalse(other.isClosed(), "New window has been closed");
66  
67          assertEquals(1, wc.getOpenWindows().length, "Num open windows");
68          assertEquals(other, wc.getMainWindow(), "Main window");
69      }

71      @Test
72      void testUnknownTarget() throws Exception {
73          defineResource("goThere", "You came back!");
74          defineResource("goHere", "You made it!");
75          defineWebPage("start", "<a href='goHere' id='go' target='somewhere'>here</a>"
76                  + "<a href='goThere' id='return' target='somewhere'>there</a>");
77  
78          WebClient wc = new WebConversation();
79          WebResponse initialPage = wc.getResponse(getHostPath() + "/start.html");
80          initialPage.getLinkWithID("go").click();
81          assertEquals(2, wc.getOpenWindows().length, "Number of windows after following link");
82          WebWindow other = wc.getOpenWindows()[1];
83          assertEquals("You made it!", other.getCurrentPage().getText(), "New window contents");
84          initialPage.getLinkWithID("return").click();
85          assertEquals(2, wc.getOpenWindows().length, "Number of windows after following link");
86          assertEquals("You came back!", other.getCurrentPage().getText(), "Updated window contents");
87      }

89      @Test
90      void testTargetInAnotherWindow() throws Exception {
91          defineWebPage("linker", "<a href='start.html' target='_blank'>start</a>");
92          defineResource("Frames.html",
93                  "<html>" + "<frameset cols=\"20%,80%\">" + "    <frame src=\"linker.html\" name=\"here\">"
94                          + "    <frame name=\"somewhere\">" + "</frameset></html>");
95          defineResource("goHere", "You made it!");
96          defineWebPage("start", "<a href='goHere' id='go' target='somewhere'>here</a>");
97  
98          WebClient wc = new WebConversation();
99          WebResponse initialPage = wc.getResponse(getHostPath() + "/Frames.html");
100         initialPage.getSubframeContents("here").getLinkWith("start").click();
101         assertEquals(2, wc.getOpenWindows().length, "# Open windows");
102         WebWindow other = wc.getOpenWindows()[1];
103         other.getCurrentPage().getLinkWithID("go").click();
104         assertEquals("You made it!", initialPage.getSubframeContents("somewhere").getText(), "New frame contents");
105     }

117     @Test
118     void testListeners() throws Exception {
119         defineResource("goHere", "You made it!");
120         defineWebPage("start", "<a href='goHere' id='go' target='_blank'>here</a>");
121 
122         final ArrayList newWindowContents = new ArrayList();
123         final ArrayList closedWindows = new ArrayList();
124         WebClient wc = new WebConversation();
125         wc.addWindowListener(new WebWindowListener() {
126             @Override
127             public void windowOpened(WebClient client, WebWindow window) {
128                 try {
129                     newWindowContents.add(window.getCurrentPage().getText());
130                 } catch (IOException e) {
131                     fail("Error trying to read page");
132                 }
133             }
134 
135             @Override
136             public void windowClosed(WebClient client, WebWindow window) {
137                 closedWindows.add(window);
138             }
139         });
140         WebResponse initialPage = wc.getResponse(getHostPath() + "/start.html");
141         initialPage.getLinkWithID("go").click();
142         assertFalse(newWindowContents.isEmpty(), "No window opened");
143         assertEquals("You made it!", newWindowContents.get(0), "New window contents");
144         assertTrue(closedWindows.isEmpty(), "Window already reported closed");
145 
146         WebWindow main = wc.getMainWindow();
147         WebWindow other = wc.getOpenWindows()[1];
148         other.close();
149         assertEquals(main, wc.getMainWindow(), "Main window");
150         assertFalse(closedWindows.isEmpty(), "No windows reported closed");
151         assertEquals(other, closedWindows.get(0), "Window reported closed");
152     }

154     @Test
155     void testWindowIndependence() throws Exception {
156         defineResource("next", "You made it!", "text/plain");
157         defineWebPage("goHere", "<a href='next' id=proceed>more</a>");
158         defineWebPage("start", "<a href='goHere.html' id='go' target='_blank'>here</a>");
159 
160         WebClient wc = new WebConversation();
161         WebWindow main = wc.getMainWindow();
162         WebResponse initialPage = wc.getResponse(getHostPath() + "/start.html");
163         initialPage.getLinkWithID("go").click();
164         WebWindow other = wc.getOpenWindows()[1];
165         other.getResponse(other.getCurrentPage().getLinkWithID("proceed").getRequest());
166         assertEquals(getHostPath() + "/start.html", main.getCurrentPage().getURL().toExternalForm(), "Main page URL");
167         assertEquals("You made it!", other.getCurrentPage().getText(), "New window contents");
168     }
