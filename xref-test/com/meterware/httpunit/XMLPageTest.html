105     /**
106      * Traversal.
107      *
108      * @throws Exception
109      *             the exception
110      */
111     @Test
112     void traversal() throws Exception {
113         defineResource("SimplePage.xml",
114                 "<?xml version='1.0' ?><zero><main><first><second/></first><main><normal/><simple/></main><after/></main><end/></zero>",
115                 "text/xml");
116         WebConversation wc = new WebConversation();
117         WebRequest request = new GetMethodWebRequest(getHostPath() + "/SimplePage.xml");
118         WebResponse simplePage = wc.getResponse(request);
119         NodeUtils.PreOrderTraversal pot = new NodeUtils.PreOrderTraversal(simplePage.getDOM());
120         final StringBuilder sb = new StringBuilder();
121         pot.perform(new NodeUtils.NodeAction() {
122             @Override
123             public boolean processElement(NodeUtils.PreOrderTraversal traversal, Element element) {
124                 if (element.getNodeName().equalsIgnoreCase("main")) {
125                     traversal.pushContext("x");
126                 } else {
127                     for (Iterator i = traversal.getContexts(); i.hasNext();) {
128                         sb.append(i.next());
129                     }
130                     sb.append(element.getNodeName()).append("|");
131                 }
132                 return true;
133             }
134 
135             @Override
136             public void processTextNode(NodeUtils.PreOrderTraversal traversal, Node textNode) {
137             }
138         });
139         // pre [ 1281655 ] [patch] result
140         String expected = "zero|xfirst|xsecond|xxnormal|xxsimple|xafter|end|";
141         // new result
142         // expected="HTML|HEAD|ZERO|xFIRST|xSECOND|xxNORMAL|xxSIMPLE|xAFTER|END|";
143         String got = sb.toString().toLowerCase(Locale.ENGLISH);
144         assertTrue(got.endsWith(expected), "Traversal result");
145     }
