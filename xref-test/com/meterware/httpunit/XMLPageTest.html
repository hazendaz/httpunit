98      @Test
99      void traversal() throws Exception {
100         defineResource("SimplePage.xml",
101                 "<?xml version='1.0' ?><zero><main><first><second/></first><main><normal/><simple/></main><after/></main><end/></zero>",
102                 "text/xml");
103         WebConversation wc = new WebConversation();
104         WebRequest request = new GetMethodWebRequest(getHostPath() + "/SimplePage.xml");
105         WebResponse simplePage = wc.getResponse(request);
106         NodeUtils.PreOrderTraversal pot = new NodeUtils.PreOrderTraversal(simplePage.getDOM());
107         final StringBuilder sb = new StringBuilder();
108         pot.perform(new NodeUtils.NodeAction() {
109             @Override
110             public boolean processElement(NodeUtils.PreOrderTraversal traversal, Element element) {
111                 if (element.getNodeName().toLowerCase().equals("main")) {
112                     traversal.pushContext("x");
113                 } else {
114                     for (Iterator i = traversal.getContexts(); i.hasNext();) {
115                         sb.append(i.next());
116                     }
117                     sb.append(element.getNodeName()).append("|");
118                 }
119                 return true;
120             }
121 
122             @Override
123             public void processTextNode(NodeUtils.PreOrderTraversal traversal, Node textNode) {
124             }
125         });
126         // pre [ 1281655 ] [patch] result
127         String expected = "zero|xfirst|xsecond|xxnormal|xxsimple|xafter|end|";
128         // new result
129         // expected="HTML|HEAD|ZERO|xFIRST|xSECOND|xxNORMAL|xxSIMPLE|xAFTER|END|";
130         String got = sb.toString().toLowerCase();
131         assertTrue(got.endsWith(expected), "Traversal result");
132     }
