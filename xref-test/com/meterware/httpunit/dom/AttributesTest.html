46      private DocumentImpl _document;

51      @BeforeEach
52      void setUp() throws Exception {
53          _document = DocumentImpl.createDocument();
54          _element = _document.createElement("zork");
55          _heightAttribute = _document.createAttribute("height");
56          _weightAttribute = _document.createAttribute("weight");
57      }

59      /**
60       * Verifies that we can create an attributes node and verify it.
61       */
62      @Test
63      void testAttributeCreation() throws Exception {
64          assertSame(_document, _heightAttribute.getOwnerDocument(), "Owner document");
65          assertEquals("height", _heightAttribute.getName(), "Name");
66          assertEquals("height", _heightAttribute.getNodeName(), "Node name");
67          assertEquals(Node.ATTRIBUTE_NODE, _heightAttribute.getNodeType(), "Node type");
68          assertNull(_heightAttribute.getAttributes(), "Attributes should not have attributes");
69          assertEquals("", _heightAttribute.getNodeValue(), "Initial attribute value");
70          assertEquals("", _heightAttribute.getValue(), "Initial Value");
71          assertFalse(_heightAttribute.getSpecified(), "Should not be marked as specified");
72  
73          _heightAttribute.setNodeValue("an example");
74          assertEquals("an example", _heightAttribute.getNodeValue(), "Node Value after nodevalue update");
75          assertEquals("an example", _heightAttribute.getValue(), "Value after nodevalue update");
76          _heightAttribute.setValue("another one");
77          assertEquals("another one", _heightAttribute.getNodeValue(), "Node Value after value update");
78          assertEquals("another one", _heightAttribute.getValue(), "Value after value update");
79          assertTrue(_heightAttribute.getSpecified(), "Should now be marked as specified");
80      }
81  
82      /**
83       * Verifies that we can set unique attribute nodes on an element and retrieve and remove them.
84       */
85      @Test
86      void testSimpleAttrNodeAssignment() throws Exception {
87          assertFalse(_element.hasAttributes(), "Element should report no attributes");
88          assertNull(_heightAttribute.getOwnerElement(), "Onwer element should not be set before assignment");
89          _element.setAttributeNode(_heightAttribute);
90          _element.setAttributeNode(_weightAttribute);
91          assertSame(_element, _heightAttribute.getOwnerElement(), "owner element");
92          assertTrue(_element.hasAttributes(), "Element should acknowledge having attributes");
93  
94          NamedNodeMap attributes = _element.getAttributes();
95          assertNotNull(attributes, "No attributes returned");
96          assertAttributesInMap(attributes, new Attr[] { _heightAttribute, _weightAttribute });
97          assertSame(_heightAttribute, _element.getAttributeNode("height"), "height attribute");
98          assertSame(_weightAttribute, _element.getAttributeNode("weight"), "weight attribute");
99  
100         _element.removeAttributeNode(_heightAttribute);
101         assertAttributesInMap(_element.getAttributes(), new Attr[] { _weightAttribute });
102         assertNull(_element.getAttributeNode("height"), "height attribute should be gone");
103         assertSame(_weightAttribute, _element.getAttributeNode("weight"), "weight attribute");
104 
105         assertNull(_heightAttribute.getOwnerElement(), "Onwer element should not be set after removal");
106     }

122     /**
123      * Verifies that setting an attribute node removes any older matching attribute node.
124      */
125     @Test
126     void testSetReplacementAttributeNode() throws Exception {
127         _element.setAttributeNode(_heightAttribute);
128         Attr newHeight = _document.createAttribute("height");
129         _element.setAttributeNode(newHeight);
130         assertSame(_element, newHeight.getOwnerElement(), "owner element");
131         assertNull(_heightAttribute.getOwnerElement(), "Onwer element should not be set after removal");
132 
133         assertAttributesInMap(_element.getAttributes(), new Attr[] { newHeight });
134         assertSame(newHeight, _element.getAttributeNode("height"), "height attribute");
135     }

145     /**
146      * Verifies that we can set and get attributes by value.
147      */
148     @Test
149     void testSimpleAttributes() throws Exception {
150         _element.setAttribute("height", "3");
151         _element.setAttribute("width", "really wide");
152         assertAttributesInMap(_element.getAttributes(),
153                 new NVPair[] { new NVPair("height", "3"), new NVPair("width", "really wide") });
154         assertTrue(_element.hasAttribute("height"), "Did not recognize height attribute");
155         assertFalse(_element.hasAttribute("color"), "Should not have claimed attribute 'color' was present");
156         assertEquals("really wide", _element.getAttribute("width"), "width attribute");
157 
158         _element.removeAttribute("height");
159         assertFalse(_element.hasAttribute("height"), "Height attribute should be gone now");
160     }

162     static class NVPair {
163         private String _name;
164         private String _value;
165 
166         public NVPair(String name, String value) {
167             _name = name;
168             _value = value;
169         }
170 
171         public String getName() {
172             return _name;
173         }
174 
175         public NVPair(Attr attrNode) {
176             this(attrNode.getName(), attrNode.getValue());
177         }
178 
179         @Override
180         public String toString() {
181             return "NV Pair: {" + _name + "," + _value + "}";
182         }
183 
184         @Override
185         public boolean equals(Object obj) {
186             return getClass().equals(obj.getClass()) && equals((NVPair) obj);
187         }
188 
189         private boolean equals(NVPair obj) {
190             return _name.equals(obj._name) && _value.equals(obj._value);
191         }
192     }
193 
194     private void assertAttributesInMap(NamedNodeMap attributes, NVPair[] expectedAttributes) {
195         assertEquals(expectedAttributes.length, attributes.getLength(), "Number of known attribute nodes");
196 
197         List attributesMissing = new ArrayList();
198         for (int i = 0; i < attributes.getLength(); i++) {
199             attributesMissing.add(new NVPair((Attr) attributes.item(i)));
200         }
201 
202         for (NVPair expectedAttribute : expectedAttributes) {
203             assertTrue(attributesMissing.contains(expectedAttribute),
204                     "Did not find attribute " + expectedAttribute + " in item sequence");
205             assertEquals(expectedAttribute, new NVPair((Attr) attributes.getNamedItem(expectedAttribute.getName())),
206                     "attribute named '" + expectedAttribute.getName() + "' in map");
207             attributesMissing.remove(expectedAttribute);
208         }
209     }
210 
211     /**
212      * Confirms that the map contains the expected attributes.
213      */
214     private void assertAttributesInMap(NamedNodeMap attributes, Attr[] expectedAttributes) {
215         assertEquals(expectedAttributes.length, attributes.getLength(), "Number of known attribute nodes");
216 
217         List attributesMissing = new ArrayList();
218         for (int i = 0; i < attributes.getLength(); i++) {
219             attributesMissing.add(attributes.item(i));
220         }
221 
222         for (Attr expectedAttribute : expectedAttributes) {
223             assertTrue(attributesMissing.contains(expectedAttribute),
224                     "Did not find attribute " + expectedAttribute + " in item sequence");
225             assertSame(expectedAttribute, attributes.getNamedItem(expectedAttribute.getName()),
226                     "attribute named '" + expectedAttribute.getName() + "' in map");
227             attributesMissing.remove(expectedAttribute);
228         }
229     }
