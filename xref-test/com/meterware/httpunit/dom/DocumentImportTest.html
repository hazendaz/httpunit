57      /**
58       * Verifies the importing of an attribute node with no children.
59       *
60       * @throws Exception
61       *             thrown if an error occurs during the test.
62       */
63      @Test
64      void importAttribute() throws Exception {
65          Element element = _document.createElement("rainbow");
66          Attr original = _document.createAttribute("color");
67          element.setAttributeNode(original);
68          original.setValue("red");
69  
70          Attr copy = (Attr) _document.importNode(original, false);
71          assertEquals(Node.ATTRIBUTE_NODE, copy.getNodeType(), "Node type");
72          assertEquals("color", copy.getNodeName(), "Node name");
73          assertNull(copy.getOwnerElement(), "Should have removed the original element");
74          assertEquals("red", copy.getNodeValue(), "Node value");
75          assertTrue(copy.getSpecified(), "Node value should be specified");
76      }

171     /**
172      * Verifies the importing of a simple element with attributes, both supporting namespaces.
173      *
174      * @throws Exception
175      *             thrown if an error occurs during the test.
176      */
177     @Test
178     void importNSElementWithNSAttributes() throws Exception {
179         Element original = _document.createElementNS("http://funnyspace/", "fs:zork");
180         original.setAttributeNS("http://funnyspace/", "fs:version", "2.0");
181         Attr size = _document.createAttributeNS("http://funnyspace/", "fs:interactive");
182         original.setAttributeNode(size);
183         verifyNSElementWithNSAttributes("original", original);
184 
185         Element copy = (Element) _document.importNode(original, /* deep */ false);
186         verifyNSElementWithNSAttributes("copy", copy);
187     }

189     /**
190      * Verify NS element with NS attributes.
191      *
192      * @param comment
193      *            the comment
194      * @param element
195      *            the element
196      */
197     private void verifyNSElementWithNSAttributes(String comment, Element element) {
198         assertEquals(Node.ELEMENT_NODE, element.getNodeType(), comment + " node type");
199         assertEquals("fs:zork", element.getNodeName(), comment + " node name");
200         assertEquals("zork", element.getLocalName(), comment + " local name");
201         assertEquals("http://funnyspace/", element.getNamespaceURI(), comment + " namespace URI");
202         assertEquals("2.0", element.getAttribute("fs:version"), comment + " version attribute");
203         assertTrue(element.hasAttribute("fs:interactive"), comment + " does not have interactive attribute");
204     }

224     /**
225      * Verifies the deep importing of an element with children.
226      *
227      * @throws Exception
228      *             thrown if an error occurs during the test.
229      */
230     @Test
231     void deepImportElementWithChildren() throws Exception {
232         Element original = _document.createElement("zork");
233         original.appendChild(_document.createElement("foo"));
234         original.appendChild(_document.createTextNode("in the middle"));
235         original.appendChild(_document.createElement("bar"));
236 
237         Element copy = (Element) _document.importNode(original, /* deep */ true);
238         assertEquals(Node.ELEMENT_NODE, copy.getNodeType(), "Node type");
239         assertEquals("zork", copy.getNodeName(), "Node name");
240         assertTrue(copy.hasChildNodes(), "copy should have children");
241 
242         NodeList children = copy.getChildNodes();
243         assertEquals(3, children.getLength(), "Number of child nodes");
244 
245         Node child = copy.getFirstChild();
246         verifyNode("1st", child, Node.ELEMENT_NODE, "foo", null);
247         child = child.getNextSibling();
248         verifyNode("2nd", child, Node.TEXT_NODE, "#text", "in the middle");
249         child = child.getNextSibling();
250         verifyNode("3rd", child, Node.ELEMENT_NODE, "bar", null);
251     }
