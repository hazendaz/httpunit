20  package com.meterware.httpunit.dom;
21  
22  import static org.junit.jupiter.api.Assertions.assertEquals;
23  import static org.junit.jupiter.api.Assertions.assertNotNull;
24  import static org.junit.jupiter.api.Assertions.assertNull;
25  
26  import org.junit.jupiter.api.AfterEach;
27  import org.junit.jupiter.api.BeforeEach;
28  import org.junit.jupiter.api.Test;
29  import org.mozilla.javascript.Context;
30  
31  /**
32   * @author <a href="mailto:russgold@httpunit.org">Russell Gold</a>
33   */
34  class DomEventScriptingTest extends AbstractHTMLElementTest {
35  
36      private Context _context;
37      private static final Object[] NO_ARGS = {};
38  
39      @BeforeEach
40      void setUp() throws Exception {
41          _context = Context.enter();
42          _context.initStandardObjects(null);
43      }
44  
45      @AfterEach
46      void tearDown() throws Exception {
47          Context.exit();
48      }
49  
50      /**
51       * Verifies that the 'onload' event for a body element is initially undefined if no corresponding attribute is
52       * defined.
53       */
54      @Test
55      void noOnloadEvent() throws Exception {
56          HTMLBodyElementImpl body = (HTMLBodyElementImpl) createElement("body");
57          assertNull(body.getOnloadEvent(), "Found a default definition for 'onLoad' event");
58      }
59  
60      /**
61       * Verifies that the 'onload' event for a body element is initially defined if a corresponding attribute is defined.
62       */
63      @Test
64      void inlineOnloadEvent() throws Exception {
65          HTMLBodyElementImpl body = (HTMLBodyElementImpl) createElement("body",
66                  new Object[][] { { "onload", "title='here'" } });
67          assertNotNull(body.getOnloadEvent(), "Found no definition for 'onLoad' event");
68          body.getOnloadEvent().call(_context, body, body, NO_ARGS);
69          assertEquals("here", body.getTitle(), "Updated title");
70      }
71  
72  }
