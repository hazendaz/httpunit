43      @Test
44      void testGetDocument() throws Exception {
45          Element element = createElement("body");
46          assertEquals(_htmlDocument, ((Scriptable) element).get("document", null), "Returned document");
47      }

49      @Test
50      void testDocumentGetTitle() throws Exception {
51          _htmlDocument.setTitle("something");
52          assertEquals("something", _htmlDocument.get("title", null), "title");
53  
54          Node body = createElement("body");
55          assertEquals("something", evaluateExpression(body, "document.title"), "title");
56      }
57  
58      @Test
59      void testDocumentPutTitle() throws Exception {
60          _htmlDocument.put("title", _htmlDocument, "right here");
61          assertEquals("right here", _htmlDocument.getTitle(), "title after put");
62  
63          Node body = createElement("body");
64          evaluateExpression(body, "document.title='new value'");
65          assertEquals("new value", _htmlDocument.getTitle(), "title after script");
66      }

72      void testElementPutTitle() throws Exception {

76          scriptableBody.put("title", scriptableBody, "right here");

83      @Test
84      void testBodyAttributes() throws Exception {
85          HTMLBodyElement body = addBodyElement();
86          body.setBgColor("red");
87  
88          assertEquals("red", evaluateExpression(_htmlDocument, "body.bgcolor"), "initial background color");
89  
90          evaluateExpression(_htmlDocument, "body.id='blue'");
91          assertEquals("blue", body.getId(), "revised foreground color");
92      }
93  
94      @Test
95      void testNumericAttributes() throws Exception {
96          HTMLBodyElement body = addBodyElement();
97          HTMLAnchorElementImpl anchor = (HTMLAnchorElementImpl) createElement("a");
98          body.appendChild(anchor);
99          anchor.setTabIndex(4);
100 
101         assertEquals(4, evaluateExpression(anchor, "tabindex"), "initial tab index");
102 
103         evaluateExpression(anchor, "tabindex=6");
104         assertEquals(6, anchor.getTabIndex(), "revised tab index");
105     }

107     private HTMLBodyElement addBodyElement() {
108         HTMLBodyElement body = (HTMLBodyElement) createElement("body");
109         _htmlDocument.setBody(body);
110         return body;
111     }
112 
113     @Test
114     void testCreateElement() throws Exception {
115         Object node = evaluateExpression(_htmlDocument, "createElement( 'a' )");
116         assertNotNull(node, "No node returned");
117         assertTrue(node instanceof HTMLAnchorElement, "Node is not an anchor element");
118     }
119 
120     @Test
121     void testDocumentLinksCollection() throws Exception {
122         TestWindowProxy proxy = new TestWindowProxy(_htmlDocument);
123         proxy.setUrl(new URL("http://localhost"));
124         _htmlDocument.getWindow().setProxy(proxy);
125         HTMLBodyElement body = addBodyElement();
126         appendLink(body, "red", "red.html");
127         appendLink(body, "blue", "blue.html");
128 
129         assertEquals(2, evaluateExpression(_htmlDocument, "links.length"), "number of links");
130         Object second = evaluateExpression(_htmlDocument, "links[1]");
131         assertNotNull(second, "Did not obtain any link object");
132         assertTrue(second instanceof HTMLAnchorElement, "Object is not a link element");
133         assertEquals("blue", ((HTMLAnchorElement) second).getId(), "Link ID");
134 
135         assertEquals("http://localhost/red.html", evaluateExpression(_htmlDocument, "links.red.href"), "red link href");
136     }
137 
138     private void appendLink(HTMLBodyElement body, String id, String href) {
139         HTMLAnchorElement anchor1 = (HTMLAnchorElement) createElement("a");
140         anchor1.setId(id);
141         anchor1.setHref(href);
142         body.appendChild(anchor1);
143     }
144 
145     @Test
146     void testConvertable() throws Exception {
147         assertConvertable(String.class, String.class);
148         assertConvertable(Integer.class, String.class);
149         assertConvertable(String.class, Integer.class);
150         assertConvertable(Short.class, Integer.class);
151         assertConvertable(String.class, Boolean.class);
152         assertConvertable(Byte.class, int.class);
153     }
154 
155     private void assertConvertable(Class valueType, Class parameterType) {
156         assertTrue(ScriptingSupport.isConvertableTo(valueType, parameterType),
157                 valueType.getName() + " should be convertable to " + parameterType.getName());
158     }
159 
160     private Object evaluateExpression(Node node, String expression) {
161         try {
162             Context context = Context.enter();
163             context.initStandardObjects(null);
164             return ((NodeImpl) node).evaluateExpression(expression);
165         } finally {
166             Context.exit();
167         }
168     }
