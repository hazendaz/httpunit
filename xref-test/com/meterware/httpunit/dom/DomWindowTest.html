20  package com.meterware.httpunit.dom;
21  
22  import static org.junit.jupiter.api.Assertions.assertEquals;
23  import static org.junit.jupiter.api.Assertions.assertFalse;
24  import static org.junit.jupiter.api.Assertions.assertNotNull;
25  import static org.junit.jupiter.api.Assertions.assertSame;
26  import static org.junit.jupiter.api.Assertions.assertTrue;
27  
28  import java.net.URL;
29  
30  import org.junit.jupiter.api.Test;
31  import org.w3c.dom.html.HTMLDocument;
32  
33  /**
34   * @author <a href="mailto:russgold@httpunit.org">Russell Gold</a>
35   */
36  class DomWindowTest extends AbstractHTMLElementTest {
37  
38      private TestWindowProxy _proxy;
39  
40      /**
41       * Verifies that we can obtain a window for a document and then retrieve document for that window.
42       */
43      @Test
44      void testDocumentWindowAccess() throws Exception {
45          DomWindow window = _htmlDocument.getWindow();
46          assertSame(_htmlDocument, window.getDocument(), "The original document");
47          assertSame(window, _htmlDocument.getWindow(), "The window upon subsequence request");
48          assertSame(window, window.getWindow(), "The window accessed from itself");
49          assertSame(window, window.getSelf(), "The window's 'self' object");
50      }
51  
52      /**
53       * Verifies that the open method returns a window with an appropriate document.
54       */
55      @Test
56      void testWindowOpen() throws Exception {
57          DomWindow window1 = createMainWindow();
58          DomWindow window2 = window1.open("next.html", "broken", "", false);
59          HTMLDocument document = window2.getDocument();
60          assertNotNull(document, "Window has no associated document");
61          assertEquals("broken (next.html)", document.getTitle(), "Title of document in new window");
62  
63          window2.close();
64          TestWindowProxy.assertLastProxyMethod("close");
65      }
66  
67      /**
68       * Writes to a document should appear in the window's document write buffer.
69       */
70      @Test
71      void testDocumentWrite() throws Exception {
72          DomWindow window1 = createMainWindow();
73          window1.getDocument().write("A simple string");
74          assertEquals("A simple string", window1.getDocumentWriteBuffer(), "Contents of write buffer");
75          window1.discardDocumentWriteBuffer();
76          assertEquals("", window1.getDocumentWriteBuffer(), "Contents of cleared write buffer");
77  
78      }
79  
80      private DomWindow createMainWindow() {
81          DomWindow window = _htmlDocument.getWindow();
82          _proxy = new TestWindowProxy(_htmlDocument);
83          window.setProxy(_proxy);
84          TestWindowProxy.clearProxyCalls();
85          return window;
86      }
87  
88      /**
89       * Verifies that an alert request is sent to the proxy appropriately.
90       */
91      @Test
92      void testAlert() throws Exception {
93          DomWindow window = createMainWindow();
94          window.alert("A little message");
95          TestWindowProxy.assertLastProxyMethod("alert( A little message )");
96      }
97  
98      /**
99       * Verifies that a confirmation request is sent to the proxy and the appropriate answer is returned.
100      */
101     @Test
102     void testConfirm() throws Exception {
103         DomWindow window = createMainWindow();
104         _proxy.setAnswer("no");
105         assertFalse(window.confirm("Time to quit?"), "Should have said no");
106         TestWindowProxy.assertLastProxyMethod("confirm( Time to quit? )");
107         _proxy.setAnswer("yes");
108         assertTrue(window.confirm("Want to stay?"), "Should have said yes");
109         TestWindowProxy.assertLastProxyMethod("confirm( Want to stay? )");
110     }
111 
112     /**
113      * Verifies that a prompt is sent to the proxy and the appropriate answer is returned.
114      */
115     @Test
116     void testPrompt() throws Exception {
117         DomWindow window = createMainWindow();
118         _proxy.setAnswer(null);
119         assertEquals("0", window.prompt("How many choices?", "0"), "User default choice");
120         TestWindowProxy.assertLastProxyMethod("prompt( How many choices? )");
121         _proxy.setAnswer("blue");
122         assertEquals("blue", window.prompt("What is your favorite color?", "yellow"), "Explicit user choice");
123         TestWindowProxy.assertLastProxyMethod("prompt( What is your favorite color? )");
124     }
125 
126     /**
127      * Verifies that writing to and closing a document triggers a replaceText request.
128      */
129     @Test
130     void testTextReplacement() throws Exception {
131         DomWindow window = createMainWindow();
132         window.getDocument().write("A bit of text");
133         window.getDocument().close();
134         assertNotNull(_proxy.getReplacementText(), "No text replacement occurred");
135         assertEquals("A bit of text", _proxy.getReplacementText(), "Replacement text");
136     }
137 
138     /**
139      * Verifies that the window can report its URL, which it obtains via its prozy.
140      */
141     @Test
142     void testWindowUrl() throws Exception {
143         DomWindow window = createMainWindow();
144         _proxy.setUrl(new URL("http://localhost"));
145         assertEquals(new URL("http://localhost"), window.getUrl(), "Window url");
146     }
