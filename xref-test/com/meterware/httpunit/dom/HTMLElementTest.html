123     @Test
124     void emptyFormDefaults() throws Exception {
125         Element element = createElement("form", new String[][] { { "action", "go_here" } });
126         assertTrue(element instanceof HTMLFormElement,
127                 "node should be an HTMLFormElement but is " + element.getClass());
128         assertEquals("FORM", element.getNodeName(), "Tag name");
129 
130         HTMLFormElement form = (HTMLFormElement) element;
131         assertEquals("UNKNOWN", form.getAcceptCharset(), "default character set");
132         assertEquals("go_here", form.getAction(), "specified action");
133         assertEquals(0, form.getElements().getLength(), "number of controls in collection");
134         assertEquals("application/x-www-form-urlencoded", form.getEnctype(), "default form encoding");
135         assertEquals(0, form.getLength(), "number of controls in form");
136         assertEquals("GET", form.getMethod().toUpperCase(), "default method");
137         assertNull(form.getName(), "form name should not be specified by default");
138         assertNull(form.getTarget(), "default target is not null");
139     }

174     @Test
175     void titleElement() throws Exception {
176         Element element = createElement("title");
177         Text text = _htmlDocument.createTextNode("something here");
178         element.appendChild(text);
179 
180         assertTrue(element instanceof HTMLTitleElement,
181                 "node should be an HTMLTitleElement but is " + element.getClass());
182         assertEquals("TITLE", element.getNodeName(), "Tag name");
183 
184         HTMLTitleElement title = (HTMLTitleElement) element;
185         assertEquals("something here", title.getText(), "initial title");
186 
187         title.setText("what it says now");
188         NodeList childNodes = element.getChildNodes();
189         assertEquals(1, childNodes.getLength(), "Number of child nodes");
190         assertTrue(childNodes.item(0) instanceof Text, "Sole child node is not text");
191         assertEquals("what it says now", ((Text) childNodes.item(0)).getData(), "Revised title text");
192         assertEquals("what it says now", title.getText(), "revised title");
193     }
