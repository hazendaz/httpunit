122     @Test
123     void testEmptyFormDefaults() throws Exception {
124         Element element = createElement("form", new String[][] { { "action", "go_here" } });
125         assertTrue(element instanceof HTMLFormElement,
126                 "node should be an HTMLFormElement but is " + element.getClass());
127         assertEquals("FORM", element.getNodeName(), "Tag name");
128 
129         HTMLFormElement form = (HTMLFormElement) element;
130         assertEquals("UNKNOWN", form.getAcceptCharset(), "default character set");
131         assertEquals("go_here", form.getAction(), "specified action");
132         assertEquals(0, form.getElements().getLength(), "number of controls in collection");
133         assertEquals("application/x-www-form-urlencoded", form.getEnctype(), "default form encoding");
134         assertEquals(0, form.getLength(), "number of controls in form");
135         assertEquals("GET", form.getMethod().toUpperCase(), "default method");
136         assertNull(form.getName(), "form name should not be specified by default");
137         assertNull(form.getTarget(), "default target is not null");
138     }

173     @Test
174     void testTitleElement() throws Exception {
175         Element element = createElement("title");
176         Text text = _htmlDocument.createTextNode("something here");
177         element.appendChild(text);
178 
179         assertTrue(element instanceof HTMLTitleElement,
180                 "node should be an HTMLTitleElement but is " + element.getClass());
181         assertEquals("TITLE", element.getNodeName(), "Tag name");
182 
183         HTMLTitleElement title = (HTMLTitleElement) element;
184         assertEquals("something here", title.getText(), "initial title");
185 
186         title.setText("what it says now");
187         NodeList childNodes = element.getChildNodes();
188         assertEquals(1, childNodes.getLength(), "Number of child nodes");
189         assertTrue(childNodes.item(0) instanceof Text, "Sole child node is not text");
190         assertEquals("what it says now", ((Text) childNodes.item(0)).getData(), "Revised title text");
191         assertEquals("what it says now", title.getText(), "revised title");
192     }
