40      /** The form. */
41      private HTMLFormElement _form;

43      /**
44       * Sets the up.
45       *
46       * @throws Exception
47       *             the exception
48       */
49      @BeforeEach
50      void setUp() throws Exception {
51          TestWindowProxy windowProxy = new TestWindowProxy(_htmlDocument);
52          windowProxy.setUrl(new URL("http://localhost/aux.html"));
53  
54          _htmlDocument.getWindow().setProxy(windowProxy);
55          HTMLBodyElement body = (HTMLBodyElement) _htmlDocument.createElement("body");
56          _htmlDocument.appendChild(body);
57  
58          _form = (HTMLFormElement) createElement("form", new String[][] { { "action", "go_here" } });
59          body.appendChild(_form);
60          _form.setMethod("GET");
61          _form.setAction("tryMe");
62      }
63  
64      /**
65       * Verifies that submitting a simple form works.
66       *
67       * @throws Exception
68       *             the exception
69       */
70      @Test
71      void submitFromForm() throws Exception {
72          addInput("text", "name").setValue("master");
73          addInput("checkbox", "second").setChecked(true);
74          _form.submit();
75          assertEquals("submitRequest( GET, http://localhost/tryMe?name=master&second=on, null, null )",
76                  TestWindowProxy.popProxyCall(), "Expected response");
77      }
78  
79      /**
80       * Verifies that submitting a simple form from a button selects that button only.
81       *
82       * @throws Exception
83       *             the exception
84       */
85      @Test
86      void submitFromButton() throws Exception {
87          addInput("text", "name", "master");
88          addInput("checkbox", "second").setChecked(true);
89          addInput("submit", "save", "none");
90          HTMLInputElementImpl button = (HTMLInputElementImpl) addInput("submit", "save", "all");
91          button.doClickAction();
92          assertEquals("submitRequest( GET, http://localhost/tryMe?name=master&second=on&save=all, null, null )",
93                  TestWindowProxy.popProxyCall(), "Expected response");
94      }

213     /**
214      * Adds the select.
215      *
216      * @param name
217      *            the name
218      *
219      * @return the HTML select element
220      */
221     private HTMLSelectElement addSelect(String name) {
222         HTMLSelectElement select = (HTMLSelectElement) _htmlDocument.createElement("select");
223         _form.appendChild(select);
224         select.setName(name);
225         return select;
226     }
227 
228     /**
229      * Adds the option.
230      *
231      * @param select
232      *            the select
233      * @param value
234      *            the value
235      * @param label
236      *            the label
237      *
238      * @return the HTML option element
239      */
240     private HTMLOptionElement addOption(HTMLSelectElement select, String value, String label) {
241         HTMLOptionElement option = (HTMLOptionElement) _htmlDocument.createElement("option");
242         select.appendChild(option);
243         if (value != null) {
244             option.setValue(value);
245         }
246         if (label != null) {
247             select.appendChild(_htmlDocument.createTextNode(label));
248         }
249         return option;
250     }
251 
252     /**
253      * Adds the input.
254      *
255      * @param type
256      *            the type
257      * @param name
258      *            the name
259      *
260      * @return the HTML input element
261      */
262     private HTMLInputElement addInput(String type, String name) {
263         HTMLInputElement element = (HTMLInputElement) _htmlDocument.createElement("input");
264         element.setAttribute("type", type);
265         element.setAttribute("name", name);
266         _form.appendChild(element);
267         return element;
268     }
269 
270     /**
271      * Adds the input.
272      *
273      * @param type
274      *            the type
275      * @param name
276      *            the name
277      * @param value
278      *            the value
279      *
280      * @return the HTML input element
281      */
282     private HTMLInputElement addInput(String type, String name, String value) {
283         HTMLInputElement element = addInput(type, name);
284         element.setValue(value);
285         return element;
286     }
