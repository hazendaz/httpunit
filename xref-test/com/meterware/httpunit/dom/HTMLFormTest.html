190     /**
191      * Checkboxes.
192      *
193      * @throws Exception
194      *             the exception
195      */
196     @Test
197     void checkboxes() throws Exception {
198         String[] values = { "on1", "on2" };
199         boolean[] initialChecked = { false, true };
200 
201         for (HTMLInputElement element : _checkbox) {
202             _form.appendChild(element);
203         }
204 
205         for (int i = 0; i < _checkbox.length; i++) {
206             assertSame(_form, _checkbox[i].getForm(), "Form for control");
207             assertEquals(values[i], _checkbox[i].getValue(), "Initial value " + i);
208             assertEquals(initialChecked[i], _checkbox[i].getChecked(), "Initial checked " + i);
209             assertEquals(initialChecked[i], _checkbox[i].getDefaultChecked(), "Initial default checked " + i);
210         }
211 
212         for (int j = 0; j < _checkbox.length; j++) {
213             _checkbox[j].setChecked(!initialChecked[j]);
214             for (int i = 0; i < _checkbox.length; i++) {
215                 assertEquals(values[i], _checkbox[i].getValue(), "value " + i + " after change " + j);
216                 assertEquals(!initialChecked[j], _checkbox[i].getChecked(), "checked " + i + " after change " + j);
217                 assertEquals(initialChecked[i], _checkbox[i].getDefaultChecked(),
218                         "default checked " + i + " after change " + j);
219             }
220             ((HTMLControl) _checkbox[j]).reset();
221             for (int i = 0; i < _checkbox.length; i++) {
222                 assertEquals(initialChecked[i], _checkbox[i].getChecked(), "Initial checked " + i);
223                 assertEquals(initialChecked[i], _checkbox[i].getDefaultChecked(), "Initial default checked " + i);
224             }
225         }
226 
227         _checkbox[0].click();
228         assertTrue(_checkbox[0].getChecked(), "checkbox 0 after 1st click");
229 
230         _checkbox[0].click();
231         assertFalse(_checkbox[0].getChecked(), "checkbox 0 after 2nd click");
232     }

302     /**
303      * Verifies that controls in the document body after the form but before the next form are included in the form
304      * collection of elements.
305      *
306      * @throws Exception
307      *             the exception
308      */
309     @Test
310     void improperFormElements() throws Exception {
311         HTMLElement[] elements = { _textField, _passwordField, _hiddenField, _textArea };
312         HTMLElement[] improperElements = { _checkbox[0], _checkbox[1], _resetInput, _submitInput };
313         _form.appendChild(_htmlDocument.createElement("i")).appendChild(_htmlDocument.createTextNode("Some controls"));
314         for (HTMLElement element : elements) {
315             _form.appendChild(element);
316         }
317         for (HTMLElement element : improperElements) {
318             _body.appendChild(element);
319         }
320         Element form = _htmlDocument.createElement("form");
321         _body.appendChild(form);
322         form.appendChild(_htmlDocument.createElement("button"));
323 
324         HTMLCollection collection = _form.getElements();
325         assertNotNull(collection, "No collection returned");
326         assertEquals(elements.length + improperElements.length, collection.getLength(), "Number of elements");
327         for (int i = 0; i < elements.length; i++) {
328             assertSame(elements[i], collection.item(i), "Form element " + i);
329             assertSame(_form, ((HTMLControl) elements[i]).getForm(), "Form for element " + i);
330         }
331         for (int i = 0; i < improperElements.length; i++) {
332             int j = elements.length + i;
333             assertSame(improperElements[i], collection.item(j), "Form element " + j);
334             assertSame(_form, ((HTMLControl) improperElements[i]).getForm(), "Form for element " + j);
335         }
336     }
