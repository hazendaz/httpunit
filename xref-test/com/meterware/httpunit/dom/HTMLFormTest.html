146     @Test
147     void testCheckboxes() throws Exception {
148         String[] values = { "on1", "on2" };
149         boolean[] initialChecked = { false, true };
150 
151         for (HTMLInputElement element : _checkbox) {
152             _form.appendChild(element);
153         }
154 
155         for (int i = 0; i < _checkbox.length; i++) {
156             assertSame(_form, _checkbox[i].getForm(), "Form for control");
157             assertEquals(values[i], _checkbox[i].getValue(), "Initial value " + i);
158             assertEquals(initialChecked[i], _checkbox[i].getChecked(), "Initial checked " + i);
159             assertEquals(initialChecked[i], _checkbox[i].getDefaultChecked(), "Initial default checked " + i);
160         }
161 
162         for (int j = 0; j < _checkbox.length; j++) {
163             _checkbox[j].setChecked(!initialChecked[j]);
164             for (int i = 0; i < _checkbox.length; i++) {
165                 assertEquals(values[i], _checkbox[i].getValue(), "value " + i + " after change " + j);
166                 assertEquals(!initialChecked[j], _checkbox[i].getChecked(), "checked " + i + " after change " + j);
167                 assertEquals(initialChecked[i], _checkbox[i].getDefaultChecked(),
168                         "default checked " + i + " after change " + j);
169             }
170             ((HTMLControl) _checkbox[j]).reset();
171             for (int i = 0; i < _checkbox.length; i++) {
172                 assertEquals(initialChecked[i], _checkbox[i].getChecked(), "Initial checked " + i);
173                 assertEquals(initialChecked[i], _checkbox[i].getDefaultChecked(), "Initial default checked " + i);
174             }
175         }
176 
177         _checkbox[0].click();
178         assertEquals(true, _checkbox[0].getChecked(), "checkbox 0 after 1st click");
179 
180         _checkbox[0].click();
181         assertEquals(false, _checkbox[0].getChecked(), "checkbox 0 after 2nd click");
182     }

204         _radio1[0].click();
205         verifyRadioButtons("radio 1 after click", _radio1, new boolean[] { true, false, false });

230     /**
231      * Verifies that controls in the document body after the form but before the next form are included in the form
232      * collection of elements.
233      *
234      * @throws Exception
235      */
236     @Test
237     void testImproperFormElements() throws Exception {
238         HTMLElement[] elements = { _textField, _passwordField, _hiddenField, _textArea };
239         HTMLElement[] improperElements = { _checkbox[0], _checkbox[1], _resetInput, _submitInput };
240         _form.appendChild(_htmlDocument.createElement("i")).appendChild(_htmlDocument.createTextNode("Some controls"));
241         for (HTMLElement element : elements) {
242             _form.appendChild(element);
243         }
244         for (HTMLElement element : improperElements) {
245             _body.appendChild(element);
246         }
247         Element form = _htmlDocument.createElement("form");
248         _body.appendChild(form);
249         form.appendChild(_htmlDocument.createElement("button"));
250 
251         HTMLCollection collection = _form.getElements();
252         assertNotNull(collection, "No collection returned");
253         assertEquals(elements.length + improperElements.length, collection.getLength(), "Number of elements");
254         for (int i = 0; i < elements.length; i++) {
255             assertSame(elements[i], collection.item(i), "Form element " + i);
256             assertSame(_form, ((HTMLControl) elements[i]).getForm(), "Form for element " + i);
257         }
258         for (int i = 0; i < improperElements.length; i++) {
259             int j = elements.length + i;
260             assertSame(improperElements[i], collection.item(j), "Form element " + j);
261             assertSame(_form, ((HTMLControl) improperElements[i]).getForm(), "Form for element " + j);
262         }
263     }

303     private void verifyNamedControlAndValue(String name, HTMLInputElement textField, String expectedValue) {
304         Object o = ((HTMLFormElementImpl) _form).get(name, null);
305         assertTrue(o instanceof ScriptableObject, "Result should be scriptable, is " + o);
306         ScriptableObject control = (ScriptableObject) o;
307         assertSame(textField, control, "control");
308         assertEquals(expectedValue, control.get("value", null), "field value");
309     }
