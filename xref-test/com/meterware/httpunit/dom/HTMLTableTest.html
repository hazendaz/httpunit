20  package com.meterware.httpunit.dom;
21  
22  import static org.junit.jupiter.api.Assertions.assertEquals;
23  import static org.junit.jupiter.api.Assertions.assertTrue;
24  
25  import org.junit.jupiter.api.BeforeEach;
26  import org.junit.jupiter.api.Test;
27  import org.w3c.dom.Element;
28  import org.w3c.dom.Node;
29  import org.w3c.dom.html.HTMLCollection;
30  import org.w3c.dom.html.HTMLTableCellElement;
31  import org.w3c.dom.html.HTMLTableElement;
32  import org.w3c.dom.html.HTMLTableRowElement;
33  
34  /**
35   * @author <a href="mailto:russgold@httpunit.org">Russell Gold</a>
36   */
37  class HTMLTableTest extends AbstractHTMLElementTest {
38  
39      private Element _body;
40      private HTMLTableElement _mainTable;
41      private HTMLTableRowElement[] _htmlMainTableRows = new HTMLTableRowElement[3];
42  
43      @BeforeEach
44      void setUp() throws Exception {
45          _body = _htmlDocument.createElement("body");
46          _htmlDocument.appendChild(_body);
47  
48          _mainTable = (HTMLTableElement) _htmlDocument.createElement("table");
49          _body.appendChild(_mainTable);
50  
51          for (int i = 0; i < _htmlMainTableRows.length; i++) {
52              _htmlMainTableRows[i] = (HTMLTableRowElement) _htmlDocument.createElement("tr");
53              _mainTable.appendChild(_htmlMainTableRows[i]);
54              for (int j = 0; j < 2; j++) {
55                  _htmlMainTableRows[i].appendChild(_htmlDocument.createElement("td"));
56              }
57          }
58      }
59  
60      /**
61       * Verify the construction of table nodes with their attributes.
62       */
63      @Test
64      void testTableNodeCreation() throws Exception {
65          doElementTest("td", HTMLTableCellElement.class,
66                  new Object[][] { { "abbr", "lots" }, { "align", "center" }, { "axis", "age" }, { "bgColor", "red" },
67              { "char", ",", "." /* ch */ }, { "charoff", "20" /* charoff */ },
68              { "colspan", Integer.valueOf(3), Integer.valueOf(1) }, { "headers", "time,age" },
69              { "height", "20" }, { "nowrap", Boolean.TRUE, Boolean.FALSE },
70              { "rowspan", Integer.valueOf(15), Integer.valueOf(1) }, { "scope", "row" },
71              { "valign", "top", "middle" }, { "width", "10" } });
72          doElementTest("th", HTMLTableCellElement.class, new Object[][] { { "abbr", "lots" } });
73          doElementTest("tr", HTMLTableRowElement.class, new Object[][] { { "align", "center" }, { "bgColor", "red" },
74              { "char", ",", "." /* ch */ }, { "charoff", "20" /* charoff */ }, { "valign", "top", "middle" } });
75          doElementTest("table", HTMLTableElement.class,
76                  new Object[][] { { "align", "right", "center" }, { "bgColor", "red" }, { "border", "2" },
77              { "cellpadding", "20" }, { "cellspacing", "20" }, { "frame", "above", "void" },
78              { "rules", "groups", "none" }, { "summary", "blah blah" }, { "width", "5" } });
79      }
80  
81      @Test
82      void testReadTable() throws Exception {
83          HTMLCollection rows = _mainTable.getRows();
84          assertEquals(3, rows.getLength(), "Number of rows in table");
85          for (int i = 0; i < 3; i++) {
86              Node node = rows.item(i);
87              assertTrue(node instanceof HTMLTableRowElement, "Row " + (i + 1) + " is not a table row");
88              HTMLCollection cells = ((HTMLTableRowElement) node).getCells();
89              assertEquals(2, cells.getLength(), "Number of cells in row");
90              for (int j = 0; j < 2; j++) {
91                  assertTrue(cells.item(j) instanceof HTMLTableCellElement,
92                          "Cell (" + (i + 1) + "," + (j + 1) + ") is not a table cell");
93              }
94          }
95      }
96  }
