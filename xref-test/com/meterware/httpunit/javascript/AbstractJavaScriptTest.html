8   package com.meterware.httpunit.javascript;
9   
10  import com.meterware.httpunit.HttpUnitTest;
11  import com.meterware.httpunit.WebConversation;
12  import com.meterware.httpunit.WebResponse;
13  
14  /**
15   * Created by IntelliJ IDEA. User: russgold Date: May 16, 2008 Time: 3:20:40 PM abstract base class for JavaScript tests
16   * of the httpunit framework supplies doTestJavaScript as a default operation for starting tests
17   */
18  public abstract class AbstractJavaScriptTest extends HttpUnitTest {
19  
20      /** The debug HTML. */
21      // set to true to get the static HTML Code on System.err
22      public static boolean debugHTML = false;
23  
24      /**
25       * test the given javaScript code by putting it into a function and calling it as a prerequisite make the html code
26       * snippet available in the body of the page.
27       *
28       * @param script
29       *            - some javascript code to be called in a function
30       * @param html
31       *            - a html code snippet
32       *
33       * @return the web conversation
34       *
35       * @throws Exception
36       *             the exception
37       */
38      public WebConversation doTestJavaScript(String script, String html) throws Exception {
39          defineResource("OnCommand.html",
40                  "<html><head><script language='JavaScript'>\n" + "function javaScriptFunction() {\n" + script + "}\n"
41                          + "</script></head>" + "<body>" + html + "\n"
42                          + "<a href=\"javascript:javaScriptFunction()\">go</a>" + "</body></html>");
43          WebConversation wc = new WebConversation();
44          WebResponse response = wc.getResponse(getHostPath() + "/OnCommand.html");
45          if (debugHTML) {
46              System.err.println(response.getText() + "\n");
47          }
48          response.getLinkWith("go").click();
49          return wc;
50      }
51  
52      /**
53       * test the given javaScript code by putting it into a function and calling it.
54       *
55       * @param script
56       *            the script to test
57       *
58       * @return the web client on which the test was run
59       *
60       * @throws Exception
61       *             the exception
62       */
63      public WebConversation doTestJavaScript(String script) throws Exception {
64          return doTestJavaScript(script, "");
65      }
66  }
