20  package com.meterware.httpunit.javascript;
21  
22  import com.meterware.httpunit.HttpUnitTest;
23  import com.meterware.httpunit.WebConversation;
24  import com.meterware.httpunit.WebResponse;
25  
26  /**
27   * Created by IntelliJ IDEA. User: russgold Date: May 16, 2008 Time: 3:20:40 PM abstract base class for JavaScript tests
28   * of the httpunit framework supplies doTestJavaScript as a default operation for starting tests
29   */
30  public abstract class AbstractJavaScriptTest extends HttpUnitTest {
31      // set to true to get the static HTML Code on System.err
32      public static boolean debugHTML = false;
33  
34      /**
35       * test the given javaScript code by putting it into a function and calling it as a prerequisite make the html code
36       * snippet available in the body of the page
37       *
38       * @param script
39       *            - some javascript code to be called in a function
40       * @param html
41       *            - a html code snippet
42       *
43       * @return
44       *
45       * @throws Exception
46       */
47      public WebConversation doTestJavaScript(String script, String html) throws Exception {
48          defineResource("OnCommand.html",
49                  "<html><head><script language='JavaScript'>\n" + "function javaScriptFunction() {\n" + script + "}\n"
50                          + "</script></head>" + "<body>" + html + "\n"
51                          + "<a href=\"javascript:javaScriptFunction()\">go</a>" + "</body></html>");
52          WebConversation wc = new WebConversation();
53          WebResponse response = wc.getResponse(getHostPath() + "/OnCommand.html");
54          if (debugHTML) {
55              System.err.println(response.getText() + "\n");
56          }
57          response.getLinkWith("go").click();
58          return wc;
59      }
60  
61      /**
62       * test the given javaScript code by putting it into a function and calling it
63       *
64       * @param script
65       *            the script to test
66       *
67       * @return the web client on which the test was run
68       */
69      public WebConversation doTestJavaScript(String script) throws Exception {
70          return doTestJavaScript(script, "");
71      }
72  }
