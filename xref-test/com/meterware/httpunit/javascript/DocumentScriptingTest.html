54                          + "function viewForms() { " + "  alert( \"found \" + document.forms.length + \" form(s)\" );"
55                          + "  alert( getFound( document.realform ) + \"form 'realform'\" );"

61          assertEquals("found 1 form(s)", wc.popNextAlert(), "Alert message");
62          assertEquals("found form 'realform'", wc.popNextAlert(), "Alert message");
63          assertEquals("found form 'forms[\'realform\']'", wc.popNextAlert(), "Alert message");
64          assertEquals("did not find form 'noform'", wc.popNextAlert(), "Alert message");
65          assertNull(wc.getNextAlert(), "Alert should have been removed");
66      }

210         defineResource("somewhere.html?with=values", "you made it!");
211         defineResource("OnCommand.html", "<html><head></head>" + "<body>" + "<a name=target href='nowhere.html'>"
212                 + "<a name=control href='#' onClick=\"document.target.href='somewhere.html?with=values';\">green</a>"

214         WebConversation wc = new WebConversation();
215         WebResponse response = wc.getResponse(getHostPath() + "/OnCommand.html");
216         WebLink link = response.getLinkWithName("target");
217         assertEquals("nowhere.html", link.getURLString(), "initial value");
218         response.getLinkWithName("control").click();
219         assertEquals(getHostPath() + "/somewhere.html?with=values", link.getRequest().getURL().toExternalForm(),
220                 "changed reference");
221         response = link.click();

223     }
