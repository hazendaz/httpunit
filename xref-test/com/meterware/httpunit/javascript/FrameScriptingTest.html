73      /**
74       * Creates and returns the lines of HTML to represent a menu page with a <code>&lt;base ...&gt;</code> tag and one
75       * link. Used where the formulated HTML is to be embedded in JavaScript.
76       *
77       * @param baseUrlString
78       *            URL string to be used in the href of the 'base' tag.
79       * @param linkUrlString
80       *            URL string to be used in the href of the link.
81       *
82       * @return String array representing the lines of HTML.
83       */
84      private String[] getMenuHtmlLines(String baseUrlString, String linkUrlString) {
85          return new String[] { "<html><head><title>Menu</title>",
86                  "<base href=\"" + baseUrlString + "\" target=\"blue\"></base>", "</head>", "<body>",
87                  "This is the menu.<br>", "Link to <a href=\"" + linkUrlString + "\">target page</a>.",
88                  "</body></html>" };
89      }
90  
91      /**
92       * Creates and returns HTML to represent a menu page with a <code>&lt;base ...&gt;</code> tag and one link.
93       *
94       * @param baseUrlString
95       *            URL string to be used in the href of the 'base' tag.
96       * @param linkUrlString
97       *            URL string to be used in the href of the link.
98       *
99       * @return String representing the HTML.
100      */
101     private String getMenuHtml(final String baseUrlString, final String linkUrlString) {
102         final String[] menuLines = getMenuHtmlLines(baseUrlString, linkUrlString);
103         StringBuilder pageBuffer = new StringBuilder();
104         for (String line : menuLines) {
105             pageBuffer.append(line).append("\n");
106         }
107         return pageBuffer.toString();
108     }

110     /**
111      * (Re)defines the web page 'Frames.html', which is a frameset containing 'red' and 'blue' frames.
112      *
113      * @param redSrc
114      *            The source URL string for the 'red' frame.
115      * @param blueSrc
116      *            The source URL string for the 'blue' frame.
117      */
118     private void redefineFrames(String redSrc, String blueSrc) {
119         defineResource("Frames.html",
120                 "<HTML><HEAD><TITLE>Frameset</TITLE></HEAD>" + "<FRAMESET cols=\"20%,80%\">" + "    <FRAME src=\""
121                         + redSrc + "\" name=\"red\">" + "    <FRAME src=\"" + blueSrc + "\" name=\"blue\">"
122                         + "</FRAMESET></HTML>");
123     }

125     /**
126      * Test that an image can be used as the source for a frame within a frameset.
127      *
128      * @throws Exception
129      *             the exception
130      */
131     @Test
132     void imageInFrame() throws Exception {
133         redefineFrames("/image.gif", "Simple.html");
134         _wc.getResponse(getHostPath() + "/Frames.html");
135         WebResponse redFrame = _wc.getFrameContents("red");
136         assertEquals("image/gif", redFrame.getContentType(), "Red frame content-type");
137     }

139     /**
140      * Test that a link within a frame that contains a <code>&lt;base ...&gt;</code> element is handled correctly.
141      *
142      * @throws Exception
143      *             the exception
144      */
145     @Test
146     void frameContainingBaseElement() throws Exception {
147         final String TARGET_TITLE = "Somewhere/Else/Target";
148         defineWebPage(TARGET_TITLE, "This is the target page.");
149         defineWebPage("Main", "This is a simple page.");
150         defineResource("BaseRelLinker.html", getMenuHtml(getHostPath() + "/Somewhere/Near/", "../Else/Target.html"));
151         redefineFrames("BaseRelLinker.html", "Main.html");
152 
153         _wc.getResponse(getHostPath() + "/Frames.html");
154         WebLink link = _wc.getFrameContents("red").getLinks()[0];
155         link.click();
156         assertEquals(TARGET_TITLE, _wc.getFrameContents("blue").getTitle(), "Content of blue frame after link click");
157     }

213     /**
214      * Verifies that when JavaScript overwrites an empty frame, other empty frames stay empty.
215      *
216      * @throws Exception
217      *             the exception
218      */
219     @Test
220     void javaScriptOverwritingBlankFrame() throws Exception {
221         defineResource("Frames.html",
222                 "<HTML><HEAD><TITLE>Initial</TITLE></HEAD>" + "<FRAMESET>" + "    <FRAME src='Writer.html' name='red'>"
223                         + "    <FRAME name=green>" + "    <FRAME name=blue>" + "</FRAMESET></HTML>");
224         defineWebPage("Writer",
225                 "This page overwrites an empty frame. <script language=\"JavaScript\">"
226                         + "  window.parent.green.document.open(\"text/html\")\n"
227                         + "  window.parent.green.document.write(\"Generated <a href=\\\"#\\\">link</a>\")\n"
228                         + "  window.parent.green.document.close()\n" + "</script>");
229 
230         _wc.getResponse(getHostPath() + "/Frames.html");
231         assertEquals(1, _wc.getFrameContents("green").getLinks().length, "Links in green frame");
232         assertEquals(0, _wc.getFrameContents("blue").getLinks().length, "Links in blue frame");
233     }
