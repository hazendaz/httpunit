49          defineWebPage("Simple", "This is a trivial page.");
50          defineResource("image.gif", new byte[] { 0, 1, 0, 1 }, "image/gif");
51          defineWebPage("Menu/Page/Target", "Target page");
52  
53          StringBuilder buff = new StringBuilder();
54          buff.append("This frame contains JavaScript that re-writes the \"red\" frame if it exists.\n");
55          buff.append("<script language=\"JavaScript\">\n");
56          buff.append("if (window.parent.red) {\n");
57          buff.append("  window.parent.red.document.open(\"text/html\")\n");
58          buff.append("  with (window.parent.red.document) {\n");
59          final String[] lines = getMenuHtmlLines(getHostPath() + "/Menu/Page/Main.html", "Target.html");
60          for (String line : lines) {
61              buff.append("    write('").append(line).append("\\n").append("');\n");
62          }
63          buff.append("    close();\n");
64          buff.append("  }\n");
65          buff.append("}\n");
66          buff.append("</script>\n");
67  
68          defineWebPage("frameRewriter", buff.toString());
69      }

71      /**
72       * Creates and returns the lines of HTML to represent a menu page with a <code>&lt;base ...&gt;</code> tag and one
73       * link. Used where the formulated HTML is to be embedded in JavaScript.
74       *
75       * @param baseUrlString
76       *            URL string to be used in the href of the 'base' tag.
77       * @param linkUrlString
78       *            URL string to be used in the href of the link.
79       *
80       * @return String array representing the lines of HTML.
81       */
82      private String[] getMenuHtmlLines(String baseUrlString, String linkUrlString) {
83          return new String[] { "<html><head><title>Menu</title>",
84                  "<base href=\"" + baseUrlString + "\" target=\"blue\"></base>", "</head>", "<body>",
85                  "This is the menu.<br>", "Link to <a href=\"" + linkUrlString + "\">target page</a>.",
86          "</body></html>" };
87      }
88  
89      /**
90       * Creates and returns HTML to represent a menu page with a <code>&lt;base ...&gt;</code> tag and one link.
91       *
92       * @param baseUrlString
93       *            URL string to be used in the href of the 'base' tag.
94       * @param linkUrlString
95       *            URL string to be used in the href of the link.
96       *
97       * @return String representing the HTML.
98       */
99      private String getMenuHtml(final String baseUrlString, final String linkUrlString) {
100         final String[] menuLines = getMenuHtmlLines(baseUrlString, linkUrlString);
101         StringBuilder pageBuffer = new StringBuilder();
102         for (String line : menuLines) {
103             pageBuffer.append(line).append("\n");
104         }
105         return pageBuffer.toString();
106     }

108     /**
109      * (Re)defines the web page 'Frames.html', which is a frameset containing 'red' and 'blue' frames.
110      *
111      * @param redSrc
112      *            The source URL string for the 'red' frame.
113      * @param blueSrc
114      *            The source URL string for the 'blue' frame.
115      */
116     private void redefineFrames(String redSrc, String blueSrc) {
117         defineResource("Frames.html",
118                 "<HTML><HEAD><TITLE>Frameset</TITLE></HEAD>" + "<FRAMESET cols=\"20%,80%\">" + "    <FRAME src=\""
119                         + redSrc + "\" name=\"red\">" + "    <FRAME src=\"" + blueSrc + "\" name=\"blue\">"
120                         + "</FRAMESET></HTML>");
121     }

123     /**
124      * Test that an image can be used as the source for a frame within a frameset.
125      */
126     @Test
127     void testImageInFrame() throws Exception {
128         redefineFrames("/image.gif", "Simple.html");
129         _wc.getResponse(getHostPath() + "/Frames.html");
130         WebResponse redFrame = _wc.getFrameContents("red");
131         assertEquals("image/gif", redFrame.getContentType(), "Red frame content-type");
132     }

134     /**
135      * Test that a link within a frame that contains a <code>&lt;base ...&gt;</code> element is handled correctly.
136      */
137     @Test
138     void testFrameContainingBaseElement() throws Exception {
139         final String TARGET_TITLE = "Somewhere/Else/Target";
140         defineWebPage(TARGET_TITLE, "This is the target page.");
141         defineWebPage("Main", "This is a simple page.");
142         defineResource("BaseRelLinker.html", getMenuHtml(getHostPath() + "/Somewhere/Near/", "../Else/Target.html"));
143         redefineFrames("BaseRelLinker.html", "Main.html");
144 
145         _wc.getResponse(getHostPath() + "/Frames.html");
146         WebLink link = _wc.getFrameContents("red").getLinks()[0];
147         link.click();
148         assertEquals(TARGET_TITLE, _wc.getFrameContents("blue").getTitle(), "Content of blue frame after link click");
149     }

179     /**
180      * Verifies that a javascript URL which triggers refresh to the parent of a frame resolves with something sensible.
181      */
182     @Test
183     void testJavaScriptURLToParentFrame() throws Exception {
184         defineResource("Frames.html", "<HTML><HEAD><TITLE>Initial</TITLE></HEAD>" + "<FRAMESET cols=\"20%,80%\">"
185                 + "    <FRAME src='Linker.html' name='red'>" + "    <FRAME src=Form name=blue>" + "</FRAMESET></HTML>");
186         defineWebPage("Linker",
187                 "This is a trivial page with <a href='javascript:window.parent.location.replace( \"Target\" );'>one link</a>");
188         defineResource("Form", "This is a page with nothing we care about");
189         defineResource("Target", "You made it!", "text/plain");
190 
191         _wc.getResponse(getHostPath() + "/Frames.html");
192         WebResponse result = _wc.getFrameContents("red").getLinkWith("one link").click();
193         assertEquals("You made it!", result.getText(), "Result of click");
194     }

196     /**
197      * Verifies that when JavaScript overwrites an empty frame, other empty frames stay empty.
198      */
199     @Test
200     void testJavaScriptOverwritingBlankFrame() throws Exception {
201         defineResource("Frames.html",
202                 "<HTML><HEAD><TITLE>Initial</TITLE></HEAD>" + "<FRAMESET>" + "    <FRAME src='Writer.html' name='red'>"
203                         + "    <FRAME name=green>" + "    <FRAME name=blue>" + "</FRAMESET></HTML>");
204         defineWebPage("Writer",
205                 "This page overwrites an empty frame. <script language=\"JavaScript\">"
206                         + "  window.parent.green.document.open(\"text/html\")\n"
207                         + "  window.parent.green.document.write(\"Generated <a href=\\\"#\\\">link</a>\")\n"
208                         + "  window.parent.green.document.close()\n" + "</script>");
209 
210         _wc.getResponse(getHostPath() + "/Frames.html");
211         assertEquals(1, _wc.getFrameContents("green").getLinks().length, "Links in green frame");
212         assertEquals(0, _wc.getFrameContents("blue").getLinks().length, "Links in blue frame");
213     }

217      */
218     @Test
219     void testFrameOnLoadEvent() throws Exception {
220         defineWebPage("OneForm", "<form name='form'><input name=text value='nothing special'></form>");
221         defineResource("Frames.html", "<html><frameset onload='alert( green.document.form.text.value );'>"
222                 + "    <frame src='OneForm.html' name='green'>" + "    <frame name=blue>" + "</frameset></htmlL>");
223 
224         _wc.getResponse(getHostPath() + "/Frames.html");
225         assertEquals("nothing special", _wc.popNextAlert(), "On load message");
