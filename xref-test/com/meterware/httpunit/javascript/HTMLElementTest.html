35      /**
36       * I D property.
37       *
38       * @throws Exception
39       *             the exception
40       */
41      @Test
42      void iDProperty() throws Exception {
43          defineResource("start.html", "<html><head><script language='JavaScript'>" + "function showTitle( id ) {"
44                  + "   alert( 'element with id ' + id + ' has title ' + document.getElementById( id ).title );" + "}"
45                  + "function showAll() {" + "    showTitle( 'there' ); showTitle( 'perform' ); showTitle( 'doIt' );"
46                  + "    showTitle( 'grouping' ); showTitle( 'aCell' ); showTitle( 'myDiv' );\n" + "}</script>"
47                  + "</head><body onLoad='showAll();'>"
48                  + "<div id=myDiv title=first><a href='somewhere' id='there' title=second>here</a>"
49                  + "<table id=grouping title=third><tr><td id='aCell' title=fourth>"
50                  + "<form id='perform' title=fifth><input type='submit' id='doIt' title=sixth></form>"
51                  + "</td></tr></table></div>");
52          WebConversation wc = new WebConversation();
53          wc.getResponse(getHostPath() + "/start.html");
54  
55          assertElementTitle(wc, "id", "there", "second");
56          assertElementTitle(wc, "id", "perform", "fifth");
57          assertElementTitle(wc, "id", "doIt", "sixth");
58          assertElementTitle(wc, "id", "grouping", "third");
59          assertElementTitle(wc, "id", "aCell", "fourth");
60          assertElementTitle(wc, "id", "myDiv", "first");
61      }

85      /**
86       * Name property.
87       *
88       * @throws Exception
89       *             the exception
90       */
91      @Test
92      void nameProperty() throws Exception {
93          defineResource("start.html", "<html><head><script language='JavaScript'>" + "function showTitle( name ) {"
94                  + "  var elements = document.getElementsByName( name );" + "  for( i = 0; i < elements.length; i++) {"
95                  + "   alert( 'element with name ' + name + ' has title ' + elements[i].title );" + "  }" + "}"
96                  + "function showAll() {" + "    showTitle( 'there' ); showTitle( 'perform' ); showTitle( 'doIt' );"
97                  + "    showTitle( 'input' );" + "}</script>" + "</head><body onLoad='showAll();'>"
98                  + "<a href='somewhere' name='there' title=second>here</a>"
99                  + "<form name='perform' title=fifth><input type='text' name='input' title='input1'>"
100                 + "<input type='hidden' name='input' title='input2'><input type='submit' name='doIt' title=sixth></form>"
101                 + "</td></tr></table></div></body></html>");
102         WebConversation wc = new WebConversation();
103         wc.getResponse(getHostPath() + "/start.html");
104 
105         assertElementTitle(wc, "name", "there", "second");
106         assertElementTitle(wc, "name", "perform", "fifth");
107         assertElementTitle(wc, "name", "doIt", "sixth");
108         assertElementTitle(wc, "name", "input", "input1");
109         assertElementTitle(wc, "name", "input", "input2");
110     }

113      * Name property with id attribute.

115      * @throws Exception
116      *             the exception

118     @Test
119     void namePropertyWithIdAttribute() throws Exception {
120         defineResource("start.html",
121                 "<html><head><script language='JavaScript'>" + "function showAll() {"
122                         + "    alert( 'element with name there has title '   + document.there.title );"
123                         + "    alert( 'element with name perform has title ' + document.perform.title );"
124                         + "    alert( 'element with name seeme has title '   + document.seeme.title );" + "}</script>"
125                         + "</head><body onLoad='showAll();'>" + "<a href='somewhere' id='there' title=second>here</a>"
126                         + "<form id='perform' title=fifth></form>" + "<img id='seeme' title='haha' src='haha.jpg'>"
127                         + "</td></tr></table></div></body></html>");
128         WebConversation wc = new WebConversation();
129         wc.getResponse(getHostPath() + "/start.html");
130 
131         assertElementTitle(wc, "name", "there", "second");
132         assertElementTitle(wc, "name", "perform", "fifth");
133         assertElementTitle(wc, "name", "seeme", "haha");
134     }

136     /**
137      * Assert element title.
138      *
139      * @param wc
140      *            the wc
141      * @param propertyName
142      *            the property name
143      * @param id
144      *            the id
145      * @param title
146      *            the title
147      */
148     private void assertElementTitle(WebConversation wc, String propertyName, final String id, final String title) {
149         assertEquals("element with " + propertyName + ' ' + id + " has title " + title, wc.popNextAlert(),
150                 "element '" + id + "' message");
151     }
152 
153     /**
154      * Element properties.
155      *
156      * @throws Exception
157      *             the exception
158      */
159     @Test
160     void elementProperties() throws Exception {
161         defineWebPage("start",
162                 "<form name='perform' title=fifth>" + "  <input name='name' maxlength=20 tabindex='1'>" + "</form>");
163         WebConversation wc = new WebConversation();
164         WebResponse response = wc.getResponse(getHostPath() + "/start.html");
165 
166         assertEquals("1",
167                 response.getScriptingHandler().evaluateExpression("document.perform.name.tabindex").toString(),
168                 "tabindex");
169         assertEquals("20",
170                 response.getScriptingHandler().evaluateExpression("document.perform.name.maxlength").toString(),
171                 "maxlength");
172     }
