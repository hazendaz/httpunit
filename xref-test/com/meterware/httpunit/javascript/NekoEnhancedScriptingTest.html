37  /**

45      @Test
46      void embeddedDocumentWrite() throws Exception {
47          defineResource("OnCommandWrite.html",
48                  "<html><head><title>something</title></head>" + "<body>" + "<script language='JavaScript'>"
49                          + "document.write( '<a id=here href=about:blank>' );" + "document.writeln( document.title );"
50                          + "document.write( '</a>' );" + "</script>" + "</body></html>");
51          WebConversation wc = new WebConversation();
52          WebResponse response = wc.getResponse(getHostPath() + "/OnCommandWrite.html");
53          WebLink link = response.getLinkWithID("here");
54          assertNotNull(link, "The link was not found");
55          assertEquals("something", link.getText(), "Link contents");
56      }

71      @Test
72      void unknownScript() throws Exception {
73          defineWebPage("FunkyScript", "<SCRIPT>" + "var stuff='<A href=\"#\">Default JavaScript Working</A><BR>';"
74                  + "document.writeln(stuff);" + "</SCRIPT>" + "<SCRIPT Language='JavaScript'>"
75                  + "var stuff='<A href=\"#\">JavaScript Working</A><BR>';" + "document.writeln(stuff);" + "</SCRIPT>"
76                  + "<SCRIPT Language='JavaScript1.2'>" + "var stuff='<A href=\"#\">JavaScript 1.2 Working</A><BR>';"
77                  + "document.writeln(stuff);" + "</SCRIPT>" + "<SCRIPT Language='VBScript'>" + "Dim stuff"
78                  + "stuff = '<A href=\"#\">VBScript</A><BR>'" + "document.writeln(stuff)" + "</SCRIPT>");
79          WebConversation wc = new WebConversation();
80          WebResponse wr = wc.getResponse(getHostPath() + "/FunkyScript.html");
81          assertNotNull(wr.getLinkWith("Default JavaScript Working"), "No default script link found");
82          assertNotNull(wr.getLinkWith("JavaScript Working"), "No default script link found");
83          assertNotNull(wr.getLinkWith("JavaScript 1.2 Working"), "No default script link found");
84          assertNull(wr.getLinkWith("VBScript"), "VBScript link found");
85      }

87      /**
88       * test no script sections
89       *
90       * @throws Exception
91       */
92      @Test
93      void noScriptSections() throws Exception {
94          defineResource("OnCommand.html",
95                  "<html><head><title>something</title></head>" + "<body>" + "<script language='JavaScript'>"
96                          + "document.write( '<a id=here href=about:blank>' );" + "document.writeln( document.title );"
97                          + "document.write( '</a>' );" + "</script>" + "<noscript>"
98                          + "<a href='#' id='there'>anything</a>" + "</noscript>" + "</body></html>");
99          WebConversation wc = new WebConversation();
100         WebResponse response = wc.getResponse(getHostPath() + "/OnCommand.html");
101         WebLink link = response.getLinkWithID("here");
102         assertNotNull(link, "The link was not found");
103         assertEquals("something", link.getText(), "Link contents");
104         assertNull(response.getLinkWithID("there"), "Should not have found link in noscript");
105 
106         HttpUnitOptions.setScriptingEnabled(false);
107         response = wc.getResponse(getHostPath() + "/OnCommand.html");
108         link = response.getLinkWithID("there");
109         assertNotNull(link, "The link was not found");
110         assertEquals("anything", link.getText(), "Link contents");
111         assertNull(response.getLinkWithID("here"), "Should not have found scripted link");
112     }

114     /**
115      * Verifies that nodes defined before a script section are available to that script section, even if a preceding
116      * script section has caused them to be cached. Currently does not work with JTidy since there is no way to parse
117      * only to a specific position in the document. It may be possible to fix this with some logic changes...
118      */
119     @Test
120     void formsCaching() throws Exception {
121         defineWebPage("OnCommand",
122                 "<form>" + "  <input type='text' name='color' value='blue' >" + "</form>" + "<script type='JavaScript'>"
123                         + "  alert( document.forms[0].color.value );" + "</script>" + "<form>"
124                         + "  <input type='text' name='size' value='3' >" + "</form>" + "<script type='JavaScript'>"
125                         + "  alert( document.forms[1].size.value );" + "</script>");
126         WebConversation wc = new WebConversation();
127         wc.getResponse(getHostPath() + "/OnCommand.html");
128         assertEquals("blue", wc.popNextAlert(), "Message 1");
129         assertEquals("3", wc.popNextAlert(), "Message 2");
130     }

132     /**
133      * Verifies that a script can write part of the frameset.
134      */
135     @Test
136     void scriptedFrames() throws Exception {
137         assertDoesNotThrow(() -> {
138             defineWebPage("OneForm", "<form name='form'><input name=text value='nothing special'></form>");
139             defineResource("Frames.html", "<html><script>" + "  document.write( '<frameset>' )" + "</script>"
140                     + "    <frame src='OneForm.html' name='green'>" + "    <frame name=blue>" + "</frameset></htmlL>");
141 
142             WebConversation wc = new WebConversation();
143             wc.getResponse(getHostPath() + "/Frames.html");
144             assertMatchingSet("Loaded frames", new String[] { "_top", "green", "blue" }, wc.getFrameNames());
145         });
146     }
