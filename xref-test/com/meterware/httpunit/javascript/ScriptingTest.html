190     /**
191      * test Detection of Javascript files that can not be found behaviour pointed out by Dan Lipofsky
192      *
193      * @throws Exception
194      */
195     @Test
196     void badJavascriptFile() {
197         // define xyz.js to create a 404 error
198         // we don't do this - it should be a default behaviour of the Pseudo Server!
199         // defineResource( "xyz.js", "File does not exist: xyz.js", 404);
200         defineResource("OnCommand.html", "<html><head>" + "<script language='JavaScript' src='xyz.js'></script></head>"
201                 + "<body>Hello</body></html>");
202         boolean originalState = HttpUnitOptions.getExceptionsThrownOnErrorStatus();
203         boolean originalScriptState = HttpUnitOptions.getExceptionsThrownOnScriptError();
204         // make sure stackTraces are not printed on Exceptions
205         // uncomment this if you'd actually like to debug the following code
206         boolean oldDebug = HttpUnitUtils.setEXCEPTION_DEBUG(false);
207         AssertionError failure = null;
208         // check 4 combinations of Exception and ScriptError status flags
209         for (int i = 0; i < 4; i++) {
210             boolean throwScriptException = i % 2 == 0; // true on case 0 and 2
211             boolean throwException = i / 2 % 2 == 0; // true on case 0 and 1
212             String testDescription = "case " + i + " throwScriptException=" + throwScriptException + " throwException="
213                     + throwException;
214             HttpUnitOptions.setExceptionsThrownOnErrorStatus(throwException);
215             HttpUnitOptions.setExceptionsThrownOnScriptError(throwScriptException);
216             HttpUnitOptions.clearScriptErrorMessages();
217             WebConversation wc = new WebConversation();
218             try {
219                 wc.getResponse(getHostPath() + "/OnCommand.html");
220                 // WebResponse response = wc.getResponse( getHostPath() + "/xyz.js" );
221                 // assertEquals( 404, response.getResponseCode() );
222                 if (throwScriptException) {
223                     fail("there should have been an exception");
224                 } else {
225                     String[] errMsgs = HttpUnitOptions.getScriptErrorMessages();
226                     assertEquals(1, errMsgs.length, "There should be an error Message");
227                     String errMsg = errMsgs[0];
228                     assertEquals("reponseCode 404 on getIncludedScript for src='xyz.js'", errMsg, testDescription);
229                 }
230             } catch (ScriptException se) {
231                 assertTrue(throwScriptException);
232             } catch (Exception e) {
233                 fail("there should be no exception when throwScriptException is " + throwScriptException);
234             } catch (AssertionError afe) {
235                 // continue looping on failed tests
236                 failure = afe;
237             }
238         }
239         if (failure != null) {
240             throw failure;
241         }
242         // Restore exceptions state
243         HttpUnitOptions.setExceptionsThrownOnErrorStatus(originalState);
244         HttpUnitOptions.setExceptionsThrownOnScriptError(originalScriptState);
245         HttpUnitUtils.setEXCEPTION_DEBUG(oldDebug);
246     }

313     @Test
314     void includeErrorBypass() throws Exception {
315         defineResource("OnBypassCommand.html", "<html><head><script language='JavaScript' src='missingScript.js'>"
316                 + "</script></head>" + "<body>" + "<a href=\"javascript:sayCheese()\">go</a>" + "</body></html>");
317         WebConversation wc = new WebConversation();
318         boolean oldDebug = HttpUnitUtils.setEXCEPTION_DEBUG(false);
319         HttpUnitOptions.setExceptionsThrownOnScriptError(false);
320         WebResponse response = wc.getResponse(getHostPath() + "/OnBypassCommand.html");
321         try {
322             HttpUnitOptions.setExceptionsThrownOnScriptError(true);
323             HttpUnitOptions.clearScriptErrorMessages();
324             response.getLinkWith("go").click();
325             fail("there should have been an exception");
326         } catch (ScriptException se) {
327             fail("Runtime exception is appropriate in this test case since we ignored the loading error");
328         } catch (RuntimeException rte) {
329             // java.lang.RuntimeException: Error clicking link: com.meterware.httpunit.ScriptException: URL
330             // 'javascript:sayCheese()' failed: org.mozilla.javascript.EcmaError: ReferenceError: "sayCheese" is not
331             // defined.
332             assertTrue(rte.getMessage().indexOf("not defined") > 0, "is not defined should be found in message");
333         } finally {
334             HttpUnitUtils.setEXCEPTION_DEBUG(oldDebug);
335         }
336 
337         HttpUnitOptions.setExceptionsThrownOnScriptError(false);
338         HttpUnitOptions.clearScriptErrorMessages();
339         response.getLinkWith("go").click();
340         String messages[] = HttpUnitOptions.getScriptErrorMessages();
341         assertEquals(1, messages.length, "there should be one message");
342         String message = messages[0];
343         assertTrue(message.indexOf("is not defined") > 0, "is not defined should be found");
344     }
