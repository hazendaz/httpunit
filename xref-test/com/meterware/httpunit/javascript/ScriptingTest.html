187     /**
188      * test Detection of Javascript files that can not be found behaviour pointed out by Dan Lipofsky
189      *
190      * @throws Exception
191      */
192     @Test
193     void testBadJavascriptFile() {
194         // define xyz.js to create a 404 error
195         // we don't do this - it should be a default behaviour of the Pseudo Server!
196         // defineResource( "xyz.js", "File does not exist: xyz.js", 404);
197         defineResource("OnCommand.html", "<html><head>" + "<script language='JavaScript' src='xyz.js'></script></head>"
198                 + "<body>Hello</body></html>");
199         boolean originalState = HttpUnitOptions.getExceptionsThrownOnErrorStatus();
200         boolean originalScriptState = HttpUnitOptions.getExceptionsThrownOnScriptError();
201         // make sure stackTraces are not printed on Exceptions
202         // uncomment this if you'd actually like to debug the following code
203         boolean oldDebug = HttpUnitUtils.setEXCEPTION_DEBUG(false);
204         AssertionError failure = null;
205         // check 4 combinations of Exception and ScriptError status flags
206         for (int i = 0; i < 4; i++) {
207             boolean throwScriptException = i % 2 == 0; // true on case 0 and 2
208             boolean throwException = i / 2 % 2 == 0; // true on case 0 and 1
209             String testDescription = "case " + i + " throwScriptException=" + throwScriptException + " throwException="
210                     + throwException;
211             HttpUnitOptions.setExceptionsThrownOnErrorStatus(throwException);
212             HttpUnitOptions.setExceptionsThrownOnScriptError(throwScriptException);
213             HttpUnitOptions.clearScriptErrorMessages();
214             WebConversation wc = new WebConversation();
215             try {
216                 wc.getResponse(getHostPath() + "/OnCommand.html");
217                 // WebResponse response = wc.getResponse( getHostPath() + "/xyz.js" );
218                 // assertEquals( 404, response.getResponseCode() );
219                 if (throwScriptException) {
220                     fail("there should have been an exception");
221                 } else {
222                     String[] errMsgs = HttpUnitOptions.getScriptErrorMessages();
223                     assertEquals(1, errMsgs.length, "There should be an error Message");
224                     String errMsg = errMsgs[0];
225                     assertEquals("reponseCode 404 on getIncludedScript for src='xyz.js'", errMsg, testDescription);
226                 }
227             } catch (ScriptException se) {
228                 assertTrue(throwScriptException);
229             } catch (Exception e) {
230                 fail("there should be no exception when throwScriptException is " + throwScriptException);
231             } catch (AssertionError afe) {
232                 // continue looping on failed tests
233                 failure = afe;
234             }
235         }
236         if (failure != null) {
237             throw failure;
238         }
239         // Restore exceptions state
240         HttpUnitOptions.setExceptionsThrownOnErrorStatus(originalState);
241         HttpUnitOptions.setExceptionsThrownOnScriptError(originalScriptState);
242         HttpUnitUtils.setEXCEPTION_DEBUG(oldDebug);
243     }

310     @Test
311     void testIncludeErrorBypass() throws Exception {
312         defineResource("OnBypassCommand.html", "<html><head><script language='JavaScript' src='missingScript.js'>"
313                 + "</script></head>" + "<body>" + "<a href=\"javascript:sayCheese()\">go</a>" + "</body></html>");
314         WebConversation wc = new WebConversation();
315         boolean oldDebug = HttpUnitUtils.setEXCEPTION_DEBUG(false);
316         HttpUnitOptions.setExceptionsThrownOnScriptError(false);
317         WebResponse response = wc.getResponse(getHostPath() + "/OnBypassCommand.html");
318         try {

320             HttpUnitOptions.clearScriptErrorMessages();
321             response.getLinkWith("go").click();
322             fail("there should have been an exception");
323         } catch (ScriptException se) {
324             fail("Runtime exception is appropriate in this test case since we ignored the loading error");
325         } catch (RuntimeException rte) {
326             // java.lang.RuntimeException: Error clicking link: com.meterware.httpunit.ScriptException: URL
327             // 'javascript:sayCheese()' failed: org.mozilla.javascript.EcmaError: ReferenceError: "sayCheese" is not
328             // defined.
329             assertTrue(rte.getMessage().indexOf("not defined") > 0, "is not defined should be found in message");
330         } finally {
331             HttpUnitUtils.setEXCEPTION_DEBUG(oldDebug);
332         }
333 
334         HttpUnitOptions.setExceptionsThrownOnScriptError(false);
335         HttpUnitOptions.clearScriptErrorMessages();
336         response.getLinkWith("go").click();
337         String messages[] = HttpUnitOptions.getScriptErrorMessages();
338         assertEquals(1, messages.length, "there should be one message");
339         String message = messages[0];
340         assertTrue(message.indexOf("is not defined") > 0, "is not defined should be found");
341     }

449             }
450         });

452         response.getLinks()[0].click();
453

455         final WebWindow openedWindow = (WebWindow) windowsOpened.get(0);
456         assertEquals("You made it!", openedWindow.getCurrentPage().getText(), "New window message");
457         assertEquals("sample", openedWindow.getName(), "New window name");

459         assertEquals("window is not closed", wc.popNextAlert(), "Alert message");

461         assertTrue(openedWindow.isClosed(), "Window was not closed");
