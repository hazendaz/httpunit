224      * test Detection of Javascript files that can not be found behaviour pointed out by Dan Lipofsky.
225      */
226     @Test
227     void badJavascriptFile() {
228         // define xyz.js to create a 404 error
229         // we don't do this - it should be a default behaviour of the Pseudo Server!
230         // defineResource( "xyz.js", "File does not exist: xyz.js", 404);
231         defineResource("OnCommand.html", "<html><head>" + "<script language='JavaScript' src='xyz.js'></script></head>"
232                 + "<body>Hello</body></html>");
233         boolean originalState = HttpUnitOptions.getExceptionsThrownOnErrorStatus();
234         boolean originalScriptState = HttpUnitOptions.getExceptionsThrownOnScriptError();
235         // make sure stackTraces are not printed on Exceptions
236         // uncomment this if you'd actually like to debug the following code
237         boolean oldDebug = HttpUnitUtils.setEXCEPTION_DEBUG(false);
238         AssertionError failure = null;
239         // check 4 combinations of Exception and ScriptError status flags
240         for (int i = 0; i < 4; i++) {
241             boolean throwScriptException = i % 2 == 0; // true on case 0 and 2
242             boolean throwException = i / 2 % 2 == 0; // true on case 0 and 1
243             String testDescription = "case " + i + " throwScriptException=" + throwScriptException + " throwException="
244                     + throwException;
245             HttpUnitOptions.setExceptionsThrownOnErrorStatus(throwException);
246             HttpUnitOptions.setExceptionsThrownOnScriptError(throwScriptException);
247             HttpUnitOptions.clearScriptErrorMessages();
248             WebConversation wc = new WebConversation();
249             try {
250                 wc.getResponse(getHostPath() + "/OnCommand.html");
251                 // WebResponse response = wc.getResponse( getHostPath() + "/xyz.js" );
252                 // assertEquals( 404, response.getResponseCode() );
253                 if (throwScriptException) {
254                     fail("there should have been an exception");
255                 } else {
256                     String[] errMsgs = HttpUnitOptions.getScriptErrorMessages();
257                     assertEquals(1, errMsgs.length, "There should be an error Message");
258                     String errMsg = errMsgs[0];
259                     assertEquals("reponseCode 404 on getIncludedScript for src='xyz.js'", errMsg, testDescription);
260                 }
261             } catch (ScriptException se) {
262                 assertTrue(throwScriptException);
263             } catch (Exception e) {
264                 fail("there should be no exception when throwScriptException is " + throwScriptException);
265             } catch (AssertionError afe) {
266                 // continue looping on failed tests
267                 failure = afe;
268             }
269         }
270         if (failure != null) {
271             throw failure;
272         }
273         // Restore exceptions state
274         HttpUnitOptions.setExceptionsThrownOnErrorStatus(originalState);
275         HttpUnitOptions.setExceptionsThrownOnScriptError(originalScriptState);
276         HttpUnitUtils.setEXCEPTION_DEBUG(oldDebug);
277     }
