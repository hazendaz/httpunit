211     /**
212      * test Detection of Javascript files that can not be found behaviour pointed out by Dan Lipofsky.
213      */
214     @Test
215     void badJavascriptFile() {
216         // define xyz.js to create a 404 error
217         // we don't do this - it should be a default behaviour of the Pseudo Server!
218         // defineResource( "xyz.js", "File does not exist: xyz.js", 404);
219         defineResource("OnCommand.html", "<html><head>" + "<script language='JavaScript' src='xyz.js'></script></head>"
220                 + "<body>Hello</body></html>");
221         boolean originalState = HttpUnitOptions.getExceptionsThrownOnErrorStatus();
222         boolean originalScriptState = HttpUnitOptions.getExceptionsThrownOnScriptError();
223         // make sure stackTraces are not printed on Exceptions
224         // uncomment this if you'd actually like to debug the following code
225         boolean oldDebug = HttpUnitUtils.setEXCEPTION_DEBUG(false);
226         AssertionError failure = null;
227         // check 4 combinations of Exception and ScriptError status flags
228         for (int i = 0; i < 4; i++) {
229             boolean throwScriptException = i % 2 == 0; // true on case 0 and 2
230             boolean throwException = i / 2 % 2 == 0; // true on case 0 and 1
231             String testDescription = "case " + i + " throwScriptException=" + throwScriptException + " throwException="
232                     + throwException;
233             HttpUnitOptions.setExceptionsThrownOnErrorStatus(throwException);
234             HttpUnitOptions.setExceptionsThrownOnScriptError(throwScriptException);
235             HttpUnitOptions.clearScriptErrorMessages();
236             WebConversation wc = new WebConversation();
237             try {
238                 wc.getResponse(getHostPath() + "/OnCommand.html");
239                 // WebResponse response = wc.getResponse( getHostPath() + "/xyz.js" );
240                 // assertEquals( 404, response.getResponseCode() );
241                 if (throwScriptException) {
242                     fail("there should have been an exception");
243                 } else {
244                     String[] errMsgs = HttpUnitOptions.getScriptErrorMessages();
245                     assertEquals(1, errMsgs.length, "There should be an error Message");
246                     String errMsg = errMsgs[0];
247                     assertEquals("reponseCode 404 on getIncludedScript for src='xyz.js'", errMsg, testDescription);
248                 }
249             } catch (ScriptException se) {
250                 assertTrue(throwScriptException);
251             } catch (Exception e) {
252                 fail("there should be no exception when throwScriptException is " + throwScriptException);
253             } catch (AssertionError afe) {
254                 // continue looping on failed tests
255                 failure = afe;
256             }
257         }
258         if (failure != null) {
259             throw failure;
260         }
261         // Restore exceptions state
262         HttpUnitOptions.setExceptionsThrownOnErrorStatus(originalState);
263         HttpUnitOptions.setExceptionsThrownOnScriptError(originalScriptState);
264         HttpUnitUtils.setEXCEPTION_DEBUG(oldDebug);
265     }

370             HttpUnitUtils.setEXCEPTION_DEBUG(oldDebug);
