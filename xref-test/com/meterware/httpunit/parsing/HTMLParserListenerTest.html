48      @Test
49      void badHTMLPage() throws Exception {
50          defineResource("BadPage.html",
51                  "<html>" + "<head><title>A Sample Page</head>\n" + "<body><p><b>Wrong embedded tags</p></b>\n"
52                          + "have <a blef=\"other.html?a=1&b=2\">an invalid link</A>\n"
53                          + "<IMG SRC=\"/images/arrow.gif\" WIDTH=1 HEIGHT=4>\n" + "<unknownTag>bla</unknownTag>"
54                          + "</body></html>\n");
55  
56          final ErrorHandler errorHandler = new ErrorHandler(/* expectProblems */true);
57          try {
58              WebConversation wc = new WebConversation();
59              HTMLParserFactory.addHTMLParserListener(errorHandler);
60              WebRequest request = new GetMethodWebRequest(getHostPath() + "/BadPage.html");
61              wc.getResponse(request);
62              assertTrue(errorHandler.foundProblems(), "Should have found problems");
63              assertEquals(request.getURL(), errorHandler.getBadURL(), "Expected URL");
64          } finally {
65              HTMLParserFactory.removeHTMLParserListener(errorHandler);
66          }
67      }

108     @Test
109     void headMethodWebRequest2() throws Exception {
110         defineResource("SimplePage.html",
111                 "<html><head><title>A Sample Page</title></head>\n" + "<body>Hello</body></html>\n");
112         HttpUnitOptions.setExceptionsThrownOnErrorStatus(true);
113         try {
114             HTMLParserFactory.setParserWarningsEnabled(true);
115             HTMLParserFactory.setHTMLParser(new NekoHTMLParser() {
116                 @Override
117                 public void parse(URL pageURL, String pageText, DocumentAdapter adapter) {
118                     System.err.println("Parsing URL=" + pageURL + "\n" + pageText);
119                     fail("Should not be parsing a HEAD request");
120                 }
121             });
122             HTMLParserFactory.addHTMLParserListener(new HTMLParserListener() {
123                 @Override
124                 public void error(URL url, String msg, int line, int column) {
125                     System.err.println("ERROR @url=" + url + ": (" + line + ", " + column + "):" + msg);
126                 }
127 
128                 @Override
129                 public void warning(URL url, String msg, int line, int column) {
130                     System.err.println("WARN @url=" + url + ": (" + line + ", " + column + "):" + msg);
131                 }
132 
133             });
134             WebConversation wc = new WebConversation();
135             // create a HeadMethodWebRequest
136             // see http://www.w3.org/Protocols/rfc2616/rfc2616-sec9.html for
137             // definition
138             WebRequest request = new HeadMethodWebRequest(getHostPath() + "/SimplePage.html");
139             WebResponse simplePage = wc.getResponse(request);
140             String text = simplePage.getText();
141             // no body should be returned
142             assertEquals("", text);
143         } finally {
144             HTMLParserFactory.reset();
145         }
146     }

148     static private class ErrorHandler implements HTMLParserListener {
149 
150         private boolean _expectProblems;
151         private boolean _foundProblems;
152         private URL _badURL;
153 
154         public ErrorHandler(boolean expectProblems) {
155             _expectProblems = expectProblems;
156         }
157 
158         @Override
159         public void warning(URL url, String msg, int line, int column) {
160             _foundProblems = true;
161             _badURL = url;
162         }
163 
164         @Override
165         public void error(URL url, String msg, int line, int column) {
166             assertTrue(_expectProblems, msg + " at line " + line + ", column " + column);
167             _foundProblems = true;
168             _badURL = url;
169         }
170 
171         public URL getBadURL() {
172             return _badURL;
173         }
174 
175         public boolean foundProblems() {
176             return _foundProblems;
177         }
178     }
