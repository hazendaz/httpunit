428         support.defineResource("/largeRequest", new PseudoServlet() {
429             @Override
430             public WebResource getPostResponse() {
431                 return new WebResource(super.getBody(), super.getHeader("CONTENT-TYPE"));
432             }
433         });
434 
435         Socket sock = new Socket("localhost", getHostPort());
436         sock.setKeepAlive(true);
437         sock.setTcpNoDelay(true);
438         sock.setSoTimeout(5000);
439 
440         byte[] requestData = generateLongMIMEPostData().getBytes(StandardCharsets.UTF_8);
441         String requestLine = "POST /largeRequest HTTP/1.1\r\n";
442         String hostHeader = "localhost:" + valueOf(getHostPort()) + "\r\n";
443         String clHeader = "Content-Length: " + valueOf(requestData.length) + "\r\n";
444         String conHeader = "Connection: Keep-Alive, TE\r\n";
445         String teHeader = "TE: trailers, deflate, gzip, compress\r\n";
446         String soapHeader = "SOAPAction: \"\"\r\n";
447         String accHeader = "Accept-Encoding: gzip, x-gzip, compress, x-compress\r\n";
448         String ctHeader = "Content-Type: multipart/related; type=\"text/xml\"; boundary=\"--MIME_Boundary\"\r\n";
449         String eoh = "\r\n";
450 
451         BufferedOutputStream out = new BufferedOutputStream(sock.getOutputStream());
452         out.write(requestLine.getBytes(StandardCharsets.UTF_8));
453         out.write(hostHeader.getBytes(StandardCharsets.UTF_8));
454         out.write(conHeader.getBytes(StandardCharsets.UTF_8));
455         out.write(teHeader.getBytes(StandardCharsets.UTF_8));
456         out.write(accHeader.getBytes(StandardCharsets.UTF_8));
457         out.write(soapHeader.getBytes(StandardCharsets.UTF_8));
458         out.write(ctHeader.getBytes(StandardCharsets.UTF_8));
459         out.write(clHeader.getBytes(StandardCharsets.UTF_8));
460         out.write(eoh.getBytes(StandardCharsets.UTF_8));
461 
462         // Send some of the request data
463         out.write(requestData, 0, 200);
464 
465         // Flush the stream and pause to simulate factors that would delay the request data
466         out.flush();
467         Thread.sleep(500);
468 
469         // Write the remaining request data
470         out.write(requestData, 200, requestData.length - 200);
471         out.flush();
472 
473         // Read the response
474         BufferedInputStream in = new BufferedInputStream(sock.getInputStream());
475         int count = 0;
476         while (in.read() != -1 && ++count < requestData.length) {
477 
478         }
479 
480         // Close the connection
481         sock.close();
482     }
