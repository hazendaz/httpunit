120     void testDateHeaderAccess() throws Exception {
121         WebRequest wr = new GetMethodWebRequest("http://localhost/simple");
122         String dateStr = "Mon, 26 Jul 1997 05:00:00 GMT";
123         Date testDate = new Date(dateStr);
124         wr.setHeaderField("Expires", dateStr);
125         HttpServletRequest request = new ServletUnitHttpRequest(NULL_SERVLET_REQUEST, wr, _context, new Hashtable(),
126                 NO_MESSAGE_BODY);
127         String dateStrRequest = request.getHeader("Expires");
128         assertEquals(dateStrRequest, dateStr, "Expires header field");
129         // invalid date headers return -1
130         long requestDate = request.getDateHeader("invalid");
131         assertEquals(-1, requestDate, "invalid date header field");
132         // valid date header field return the millisecs
133         requestDate = request.getDateHeader("Expires");
134         assertEquals(requestDate, testDate.getTime(), "Expires date header field");
135     }
