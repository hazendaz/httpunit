119     void dateHeaderAccess() throws Exception {
120         WebRequest wr = new GetMethodWebRequest("http://localhost/simple");
121         String dateStr = "Mon, 26 Jul 1997 05:00:00 GMT";
122         Date testDate = new Date(dateStr);
123         wr.setHeaderField("Expires", dateStr);
124         HttpServletRequest request = new ServletUnitHttpRequest(NULL_SERVLET_REQUEST, wr, _context, new Hashtable<>(),
125                 NO_MESSAGE_BODY);
126         String dateStrRequest = request.getHeader("Expires");
127         assertEquals(dateStrRequest, dateStr, "Expires header field");
128         // invalid date headers return -1
129         long requestDate = request.getDateHeader("invalid");
130         assertEquals(-1, requestDate, "invalid date header field");
131         // valid date header field return the millisecs
132         requestDate = request.getDateHeader("Expires");
133         assertEquals(requestDate, testDate.getTime(), "Expires date header field");
134     }
