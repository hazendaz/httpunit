46      /**
47       * Simple response.
48       *
49       * @throws Exception
50       *             the exception
51       */
52      @Test
53      void simpleResponse() throws Exception {
54          ServletUnitHttpResponse servletResponse = new ServletUnitHttpResponse();
55          servletResponse.setContentType("text/html");
56          servletResponse.setContentLength(65);
57          PrintWriter pw = servletResponse.getWriter();
58          pw.println("<html><head><title>Sample Page</title></head><body></body></html>");
59  
60          WebResponse response = new ServletUnitWebResponse(null, FrameSelector.TOP_FRAME, null, servletResponse);
61          assertEquals(HttpServletResponse.SC_OK, response.getResponseCode(), "Status code");
62          assertEquals("text/html", response.getContentType(), "Content type");
63          assertEquals("Sample Page", response.getTitle(), "Title");
64          assertEquals(65, response.getContentLength(), "Content length");
65          assertEquals("ISO-8859-1", response.getCharacterSet(), "Content encoding");
66          assertEquals("text/html; charset=ISO-8859-1", response.getHeaderField("Content-type"), "Content header");
67      }

77          String hebrewTitle = "\u05d0\u05d1\u05d2\u05d3";
78          String page = "<html><head><title>" + hebrewTitle + "</title></head>\n" + "<body>This has no data\n"
79                  + "</body></html>\n";
80          ServletUnitHttpResponse servletResponse = new ServletUnitHttpResponse();
81          servletResponse.setContentType("text/html; charset=UTF-8");
82          PrintWriter pw = servletResponse.getWriter();
83          pw.print(page);
84          pw.close();
85  
86          WebResponse response = new ServletUnitWebResponse(null, FrameSelector.TOP_FRAME, null, servletResponse);
87          assertEquals("UTF-8", response.getCharacterSet(), "Character set");
88          assertEquals(hebrewTitle, response.getTitle(), "Title");
89      }

240     /**
241      * Single headers.
242      *
243      * @throws Exception
244      *             the exception
245      */
246     @Test
247     void singleHeaders() throws Exception {
248         ServletUnitHttpResponse servletResponse = new ServletUnitHttpResponse();
249         servletResponse.setContentType("text/html");
250 
251         assertFalse(servletResponse.containsHeader("foo"), "header foo wrongly detected");
252         servletResponse.setHeader("foo", "bar");
253         String headerValue = servletResponse.getHeaderField("foo");
254         assertEquals("bar", headerValue, "header is wrong");
255         assertTrue(servletResponse.containsHeader("foo"), "header foo not detected");
256 
257         servletResponse.setHeader("foo", "baz");
258         headerValue = servletResponse.getHeaderField("foo");
259         assertEquals("baz", headerValue, "header is wrong");
260 
261         servletResponse.setIntHeader("three", 3);
262         headerValue = servletResponse.getHeaderField("three");
263         assertEquals("3", headerValue, "int header is wrong");
264 
265         // use RFC1123_DATE_SPEC formatter
266         SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy HH:mm:ss z");
267         Date d = df.parse("12/9/1969 12:00:00 GMT");
268         servletResponse.setDateHeader("date", d.getTime());
269         headerValue = servletResponse.getHeaderField("date");
270         assertEquals("Tue, 09 Dec 1969 12:00:00 GMT", headerValue, "date header is wrong");
271     }

273     /**
274      * Multiple headers.
275      *
276      * @throws Exception
277      *             the exception
278      */
279     @Test
280     void multipleHeaders() throws Exception {
281         ServletUnitHttpResponse servletResponse = new ServletUnitHttpResponse();
282         servletResponse.setContentType("text/html");
283 
284         // RFC1123_DATE_SPEC format
285         SimpleDateFormat df = new SimpleDateFormat("MM/dd/yyyy HH:mm:ss z");
286         Date date = df.parse("12/9/1969 12:00:00 GMT");
287 
288         servletResponse.addHeader("list", "over-rideme");
289         servletResponse.setHeader("list", "foo");
290         servletResponse.addIntHeader("list", 3);
291         servletResponse.addDateHeader("list", date.getTime());
292         String[] headerList = servletResponse.getHeaderFields("list");
293         assertEquals("foo", headerList[0], "header is wrong");
294         assertEquals("3", headerList[1], "header is wrong");
295         assertEquals("Tue, 09 Dec 1969 12:00:00 GMT", headerList[2], "header is wrong");
296 
297         servletResponse.setHeader("list", "monkeyboy");
298         headerList = servletResponse.getHeaderFields("list");
299         assertEquals(1, headerList.length, "setHeader did not replace the list header");
300         assertEquals("monkeyboy", headerList[0], "header is wrong");
301     }
