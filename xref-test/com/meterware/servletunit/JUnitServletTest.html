193     void someFailuresXMLFormat() throws Exception {
194         ServletUnitClient client = newClient();

196         WebResponse wr = client.getResponse("http://localhost/JUnit?format=xml&test=" + FailingTests.class.getName());

198         DocumentBuilder builder = HttpUnitUtils.newParser();
199         Document document = builder.parse(wr.getInputStream());
200         Element element = document.getDocumentElement();
201         assertEquals("testsuite", element.getNodeName(), "document element name");
202         assertEquals("3", element.getAttribute("tests"), "number of tests");
203         assertEquals("2", element.getAttribute("failures"), "number of failures");
204         assertEquals("0", element.getAttribute("errors"), "number of errors");
205         NodeList nl = element.getElementsByTagName("testcase");
206         verifyElementWithNameHasFailureNode("testAddition", nl, /* failed */ "failure", true);
207         verifyElementWithNameHasFailureNode("testSubtraction", nl, /* failed */ "failure", true);
208         verifyElementWithNameHasFailureNode("testMultiplication", nl, /* failed */ "failure", false);
209     }

235     /**
236      * Verify element with name has failure node.
237      *
238      * @param name
239      *            the name
240      * @param nl
241      *            the nl
242      * @param nodeName
243      *            the node name
244      * @param failed
245      *            the failed
246      */
247     private void verifyElementWithNameHasFailureNode(String name, NodeList nl, String nodeName, boolean failed) {
248         for (int i = 0; i < nl.getLength(); i++) {
249             Element element = (Element) nl.item(i);
250             if (element.getAttribute("name").indexOf(name) >= 0) {
251                 if (failed) {
252                     assertEquals(1, element.getElementsByTagName(nodeName).getLength(),
253                             "no " + nodeName + " element found for test '" + name + "'");
254                 } else {
255                     assertEquals(0, element.getElementsByTagName(nodeName).getLength(),
256                             "unexpected " + nodeName + " element found for test '" + name + "'");
257                 }
258                 return;
259             }
260         }
261         if (failed) {
262             fail("No test result found for '" + name + "'");
263         }
264     }

270      *             the exception
271      */
272     @Test
273     void scriptedServletAccess() throws Exception {
274         WebXMLString wxs = new WebXMLString();
275         Properties params = new Properties();
276         params.setProperty("color", "red");
277         params.setProperty("age", "12");
278         wxs.addServlet("simple", "/SimpleServlet", SimpleGetServlet.class, params);
279         wxs.addServlet("/JUnit", TestRunnerServlet.class);
280 
281         MyFactory._runner = _runner = new ServletRunner(wxs.asInputStream());
282         ServletUnitClient client = _runner.newClient();
283         WebResponse wr = client.getResponse("http://localhost/JUnit?test=" + ServletAccessTestClass.class.getName());
284 
285         final WebTable resultsTable = wr.getTableWithID("results");
286         assertNotNull(resultsTable, "Did not find results table");
287         final String[][] results = resultsTable.asText();
288         assertEquals("OK", results[0][2], "Status");
289     }
