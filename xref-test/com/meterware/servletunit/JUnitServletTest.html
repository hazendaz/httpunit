142     @Test
143     void someFailuresXMLFormat() throws Exception {
144         ServletUnitClient client = newClient();
145 
146         WebResponse wr = client.getResponse("http://localhost/JUnit?format=xml&test=" + FailingTests.class.getName());
147         assertEquals("text/xml", wr.getContentType(), "Content type");
148         DocumentBuilder builder = HttpUnitUtils.newParser();
149         Document document = builder.parse(wr.getInputStream());
150         Element element = document.getDocumentElement();
151         assertEquals("testsuite", element.getNodeName(), "document element name");
152         assertEquals("3", element.getAttribute("tests"), "number of tests");
153         assertEquals("2", element.getAttribute("failures"), "number of failures");
154         assertEquals("0", element.getAttribute("errors"), "number of errors");
155         NodeList nl = element.getElementsByTagName("testcase");
156         verifyElementWithNameHasFailureNode("testAddition", nl, /* failed */ "failure", true);
157         verifyElementWithNameHasFailureNode("testSubtraction", nl, /* failed */ "failure", true);
158         verifyElementWithNameHasFailureNode("testMultiplication", nl, /* failed */ "failure", false);
159     }

170         assertEquals("testsuite", element.getNodeName(), "document element name");
171         assertEquals("2", element.getAttribute("tests"), "number of tests");
172         assertEquals("0", element.getAttribute("failures"), "number of failures");
173         assertEquals("1", element.getAttribute("errors"), "number of errors");
174         NodeList nl = element.getElementsByTagName("testcase");
175         verifyElementWithNameHasFailureNode("testAddition", nl, /* failed */ "error", true);
176         verifyElementWithNameHasFailureNode("testMultiplication", nl, /* failed */ "error", false);
177     }
178 
179     private void verifyElementWithNameHasFailureNode(String name, NodeList nl, String nodeName, boolean failed) {
180         for (int i = 0; i < nl.getLength(); i++) {
181             Element element = (Element) nl.item(i);
182             if (element.getAttribute("name").indexOf(name) >= 0) {
183                 if (failed) {
184                     assertEquals(1, element.getElementsByTagName(nodeName).getLength(),
185                             "no " + nodeName + " element found for test '" + name + "'");
186                 } else {
187                     assertEquals(0, element.getElementsByTagName(nodeName).getLength(),
188                             "unexpected " + nodeName + " element found for test '" + name + "'");
189                 }
190                 return;
191             }
192         }
193         if (failed) {
194             fail("No test result found for '" + name + "'");
195         }
196     }

198     @Test
199     void scriptedServletAccess() throws Exception {
200         WebXMLString wxs = new WebXMLString();
201         Properties params = new Properties();
202         params.setProperty("color", "red");
203         params.setProperty("age", "12");
204         wxs.addServlet("simple", "/SimpleServlet", SimpleGetServlet.class, params);
205         wxs.addServlet("/JUnit", TestRunnerServlet.class);
206 
207         MyFactory._runner = _runner = new ServletRunner(wxs.asInputStream());
208         ServletUnitClient client = _runner.newClient();
209         WebResponse wr = client.getResponse("http://localhost/JUnit?test=" + ServletAccessTestClass.class.getName());
210 
211         final WebTable resultsTable = wr.getTableWithID("results");
212         assertNotNull(resultsTable, "Did not find results table");
213         final String[][] results = resultsTable.asText();
214         assertEquals("OK", results[0][2], "Status");
215     }

217     private ServletUnitClient newClient() {
218         _runner = new ServletRunner();
219         MyFactory._runner = _runner;
220         _runner.registerServlet("/JUnit", TestRunnerServlet.class.getName());
221         return _runner.newClient();
222     }
