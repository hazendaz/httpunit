55              EventVerifier verifyContext = new ServletContextEventVerifier();

57              wxs.addContextListener(ListenerClass1.class);
58              wxs.addContextListener(ListenerClass2.class);
59  
60              clearEvents();
61              expectEvent("startup", ListenerClass1.class, verifyContext);
62              expectEvent("startup", ListenerClass2.class, verifyContext);
63              ServletRunner sr = new ServletRunner(wxs.asInputStream());
64              verifyEvents();
65  
66              clearEvents();
67              expectEvent("shutdown", ListenerClass2.class, verifyContext);
68              expectEvent("shutdown", ListenerClass1.class, verifyContext);
69              sr.shutDown();
70              verifyEvents();
71          });
72      }

145         assertDoesNotThrow(() -> {
146             WebXMLString wxs = new WebXMLString();
147             wxs.addServlet("/SimpleServlet", WebXMLTest.SimpleGetServlet.class);
148             HttpSessionAttributeEventVerifier verifyAttribute = new HttpSessionAttributeEventVerifier();

154             ServletRunner sr = new ServletRunner(wxs.asInputStream());
155

157             InvocationContext ic = client.newInvocation("http://localhost/SimpleServlet");
158             HttpSession session = ic.getRequest().getSession();
159             verifyEvents();
160 
161             verifyAttribute.expect("one", Integer.valueOf(1));
162             expectEvent("added", ListenerClass5.class, verifyAttribute);
163             expectEvent("added", ListenerClass6.class, verifyAttribute);
164             session.setAttribute("one", Integer.valueOf(1));
165             verifyEvents();
166 
167             expectEvent("replaced", ListenerClass5.class, verifyAttribute);
168             expectEvent("replaced", ListenerClass6.class, verifyAttribute);
169             session.setAttribute("one", "I");
170             verifyEvents();
171 
172             verifyAttribute.expect("one", "I");
173             expectEvent("removed", ListenerClass5.class, verifyAttribute);
174             expectEvent("removed", ListenerClass6.class);
175             session.removeAttribute("one");
176             verifyEvents();
177 
178             sr.shutDown();
179         });
