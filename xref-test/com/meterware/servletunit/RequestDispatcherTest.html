92      /**
93       * Request dispatcher parameters.
94       *
95       * @throws Exception
96       *             the exception
97       */
98      @Test
99      void requestDispatcherParameters() throws Exception {
100         InvocationContext ic = _runner.newClient()
101                 .newInvocation("http://localhost/sample/" + outerServletName + "?param=original&param1=first");
102 
103         final HttpServletRequest request = ic.getRequest();
104         final HttpServletResponse response = ic.getResponse();
105         RequestDispatcherServlet servlet = (RequestDispatcherServlet) ic.getServlet();
106         RequestDispatcher rd = servlet.getServletContext()
107                 .getRequestDispatcher("/" + innerServletName + "?param=revised&param2=new");
108 
109         assertEquals("original", request.getParameter("param"), "param");
110         assertEquals("first", request.getParameter("param1"), "param1");
111         assertNull(request.getParameter("param2"), "param2 should not be defined");
112 
113         ic.pushIncludeRequest(rd, request, response);
114 
115         final HttpServletRequest innerRequest = ic.getRequest();
116         assertEquals("revised", innerRequest.getParameter("param"), "param in included servlet");
117         assertEquals("first", innerRequest.getParameter("param1"), "param1 in included servlet");
118         assertEquals("new", innerRequest.getParameter("param2"), "param2 in included servlet");
119 
120         assertEquals(IncludedServlet.class, ic.getServlet().getClass(), "Included servlet class");
121 
122         ic.popRequest();
123 
124         final HttpServletRequest restoredRequest = ic.getRequest();
125         assertEquals("original", restoredRequest.getParameter("param"), "reverted param");
126         assertEquals("first", restoredRequest.getParameter("param1"), "reverted param1");
127         assertNull(restoredRequest.getParameter("param2"), "reverted param2 should not be defined");
128         assertEquals(RequestDispatcherServlet.class, ic.getServlet().getClass(), "Included servlet class");
129     }

291     /**
292      * The Class IncludedServlet.
293      */
294     static class IncludedServlet extends HttpServlet {
295 
296         /** The Constant serialVersionUID. */
297         private static final long serialVersionUID = 1L;
298 
299         /** The desired request uri. */
300         static String DESIRED_REQUEST_URI = "localhost/subdir/pagename.jsp";
301 
302         /** The desired servlet path. */
303         static String DESIRED_SERVLET_PATH = "/subdir/pagename.jsp";
304 
305         /** The desired query string. */
306         static String DESIRED_QUERY_STRING = "param=value&param2=value";
307 
308         /** The desired output. */
309         static String DESIRED_OUTPUT = DESIRED_REQUEST_URI + DESIRED_QUERY_STRING + DESIRED_SERVLET_PATH;
310 
311         @Override
312         public void service(HttpServletRequest request, HttpServletResponse response) throws IOException {
313             response.setContentType("text/plain");
314             String requestUri = (String) request.getAttribute(REQUEST_URI);
315             String queryString = (String) request.getAttribute(QUERY_STRING);
316             String servletPath = (String) request.getAttribute(SERVLET_PATH);
317             PrintWriter pw = response.getWriter();
318             pw.write(blankIfNull(requestUri));
319             pw.write(blankIfNull(queryString));
320             pw.write(blankIfNull(servletPath));
321             pw.close();
322         }
323 
324         /**
325          * Blank if null.
326          *
327          * @param s
328          *            the s
329          *
330          * @return the string
331          */
332         private String blankIfNull(String s) {
333             return s == null ? "" : s;
334         }
