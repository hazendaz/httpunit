70      @Test
71      void requestDispatcherParameters() throws Exception {
72          InvocationContext ic = _runner.newClient()
73                  .newInvocation("http://localhost/sample/" + outerServletName + "?param=original&param1=first");
74  
75          final HttpServletRequest request = ic.getRequest();
76          final HttpServletResponse response = ic.getResponse();
77          RequestDispatcherServlet servlet = (RequestDispatcherServlet) ic.getServlet();
78          RequestDispatcher rd = servlet.getServletContext()
79                  .getRequestDispatcher("/" + innerServletName + "?param=revised&param2=new");
80  
81          assertEquals("original", request.getParameter("param"), "param");
82          assertEquals("first", request.getParameter("param1"), "param1");
83          assertNull(request.getParameter("param2"), "param2 should not be defined");
84  
85          ic.pushIncludeRequest(rd, request, response);
86  
87          final HttpServletRequest innerRequest = ic.getRequest();
88          assertEquals("revised", innerRequest.getParameter("param"), "param in included servlet");
89          assertEquals("first", innerRequest.getParameter("param1"), "param1 in included servlet");
90          assertEquals("new", innerRequest.getParameter("param2"), "param2 in included servlet");
91  
92          assertEquals(IncludedServlet.class, ic.getServlet().getClass(), "Included servlet class");
93  
94          ic.popRequest();
95  
96          final HttpServletRequest restoredRequest = ic.getRequest();
97          assertEquals("original", restoredRequest.getParameter("param"), "reverted param");
98          assertEquals("first", restoredRequest.getParameter("param1"), "reverted param1");
99          assertNull(restoredRequest.getParameter("param2"), "reverted param2 should not be defined");
100         assertEquals(RequestDispatcherServlet.class, ic.getServlet().getClass(), "Included servlet class");
101     }

218     static class RequestDispatcherServlet extends HttpServlet {
219 
220         private static final long serialVersionUID = 1L;
221 
222         @Override
223         public void doGet(HttpServletRequest request, HttpServletResponse response)
224                 throws ServletException, IOException {
225             RequestDispatcher dispatcher = getServletContext()
226                     .getRequestDispatcher("/subdir/pagename.jsp?param=value&param2=value");
227             dispatcher.forward(request, response);
228         }

237     static class IncludedServlet extends HttpServlet {
238 
239         private static final long serialVersionUID = 1L;
240         static String DESIRED_REQUEST_URI = "localhost/subdir/pagename.jsp";
241         static String DESIRED_SERVLET_PATH = "/subdir/pagename.jsp";
242         static String DESIRED_QUERY_STRING = "param=value&param2=value";
243         static String DESIRED_OUTPUT = DESIRED_REQUEST_URI + DESIRED_QUERY_STRING + DESIRED_SERVLET_PATH;
244 
245         @Override
246         public void service(HttpServletRequest request, HttpServletResponse response) throws IOException {
247             response.setContentType("text/plain");
248             String requestUri = (String) request.getAttribute(REQUEST_URI);
249             String queryString = (String) request.getAttribute(QUERY_STRING);
250             String servletPath = (String) request.getAttribute(SERVLET_PATH);
251             PrintWriter pw = response.getWriter();
252             pw.write(blankIfNull(requestUri));
253             pw.write(blankIfNull(queryString));
254             pw.write(blankIfNull(servletPath));
255             pw.close();
256         }
257 
258         private String blankIfNull(String s) {
259             return s == null ? "" : s;
260         }
