96      /**
97       * check that the dtd is on the CLASSPATH.
98       *
99       * @return true, if is dtd on classpath
100      */
101     public static boolean isDtdOnClasspath() {
102         if (dtdStream == null) {
103             dtdStream = WebXMLString.class.getClassLoader().getResource(dtd);
104         }
105         return dtdStream != null;
106     }

108     /**
109      * return me as a Text/String containing the XML descriptor according to the Sun Microsystems's specification if
110      * running on Eclipse shorten the DOCTYPE specification as a work-around for a bogus java.net.MalformedURLException
111      * (that is in fact caused by a null-pointer exception in handling the dtd part)
112      *
113      * @return the string
114      */
115     String asText() {
116         StringBuilder result = new StringBuilder("<?xml version='1.0' encoding='UTF-8'?>\n");
117         String doctype = "<!DOCTYPE web-app PUBLIC '-//Sun Microsystems, Inc.//DTD Web Application 2.3//EN' 'http://java.sun.com/dtd/"
118                 + dtd + "'>";
119         // if the dtd file is not on the CLASSPATH there will be nasty java.net.MalformedURLException problems
120         // work - around if dtd is not in CLASSPATH so that only the one test case
121         // that checks the isDtdOnClassPath condition will fail
122         if (!isDtdOnClasspath()) {
123             doctype = "<!DOCTYPE web-app>";
124         }
125         result.append(doctype);
126         result.append("<web-app>\n");
127         // result.append( "<web-app version='2.4' xmlns='http://java.sun.com/xml/ns/j2ee'\n " );
128         // result.append( " xmlns:xsi='http://www.w3.org/2001/XMLSchema-instance'\n" );
129         // result.append( " xsi:schemaLocation='http://java.sun.com/xml/ns/j2ee
130         // http://java.sun.com/xml/ns/j2ee/web-app_2_4.xsd'>\n" );
131 
132         if (_displayName != null) {
133             result.append("  <display-name>").append(_displayName).append("</display-name>");
134         }
135         for (Iterator i = _contextParams.entrySet().iterator(); i.hasNext();) {
136             Map.Entry entry = (Map.Entry) i.next();
137             result.append("  <context-param>\n    <param-name>").append(entry.getKey());
138             result.append("</param-name>\n    <param-value>").append(entry.getValue())
139                     .append("</param-value>\n  </context-param>\n");
140         }
141         for (int i = 0; i < _filters.size(); i++) {
142             Object name = _filterNames.get(i);
143             result.append("  <filter>\n    <filter-name>").append(name).append("</filter-name>\n");
144             result.append("    <filter-class>").append(((Class) _filters.get(i)).getName()).append("</filter-class>\n");
145             appendParams(result, "init-param", (Hashtable) _filterParams.get(name));
146             result.append("  </filter>\n");
147         }
148         for (int i = 0; i < _filters.size(); i++) {
149             result.append("  <filter-mapping>\n    <filter-name>").append(_filterNames.get(i))
150                     .append("</filter-name>\n");
151             result.append("    ").append(_filterMappings.get(_filterNames.get(i))).append("\n  </filter-mapping>\n");
152         }
153         for (Object _listener : _listeners) {
154             Class aClass = (Class) _listener;
155             result.append("  <listener><listener-class>").append(aClass.getName());
156             result.append("</listener-class></listener>\n");
157         }
158         for (int i = _servlets.size() - 1; i >= 0; i--) {
159             Object name = _servletNames.get(i);
160             result.append("  <servlet>\n    <servlet-name>").append(name).append("</servlet-name>\n");
161             result.append("    <servlet-class>").append(((Class) _servlets.get(i)).getName())
162                     .append("</servlet-class>\n");
163             appendParams(result, "init-param", (Hashtable) _initParams.get(name));
164             appendLoadOnStartup(result, _loadOnStartup.get(name));
165             result.append("  </servlet>\n");
166         }
167         for (int i = _mappings.size() - 1; i >= 0; i--) {
168             result.append("  <servlet-mapping>\n    <servlet-name>").append(_servletNames.get(i))
169                     .append("</servlet-name>\n");
170             result.append("    <url-pattern>").append(_mappings.get(i))
171                     .append("</url-pattern>\n  </servlet-mapping>\n");
172         }
173         for (Enumeration e = _resources.elements(); e.hasMoreElements();) {
174             result.append(((WebResourceSpec) e.nextElement()).asText());
175         }
176         result.append(_loginConfig);
177         result.append("</web-app>");
178         return result.toString();
179     }

181     /**
182      * Append load on startup.
183      *
184      * @param result
185      *            the result
186      * @param startupOrder
187      *            the startup order
188      */
189     private void appendLoadOnStartup(StringBuilder result, Object startupOrder) {
190         if (startupOrder == null) {
191             return;
192         }
193         result.append("    <load-on-startup");
194         if (startupOrder instanceof Number) {
195             result.append(">").append(startupOrder).append("</load-on-startup>\n");
196         } else {
197             result.append("/>\n");
198         }
199     }

201     /**
202      * Append params.
203      *
204      * @param result
205      *            the result
206      * @param tagName
207      *            the tag name
208      * @param params
209      *            the params
210      */
211     private void appendParams(StringBuilder result, String tagName, Hashtable params) {
212         if (params == null) {
213             return;
214         }
215         for (Iterator it = params.entrySet().iterator(); it.hasNext();) {
216             Map.Entry entry = (Map.Entry) it.next();
217             result.append("    <").append(tagName).append(">\n      <param-name>").append(entry.getKey());
218             result.append("</param-name>\n      <param-value>").append(entry.getValue())
219                     .append("</param-value>\n    </");
220             result.append(tagName).append(">\n");
221         }
222     }

445     private WebResourceSpec getWebResource(String resourceName) {
446         WebResourceSpec result = (WebResourceSpec) _resources.get(resourceName);

479     String asText() {
480         StringBuilder sb = new StringBuilder();
481         sb.append("  <security-constraint>\n");
482         sb.append("    <web-resource-collection>\n");
483         sb.append("      <web-resource-name>").append(_name).append("</web-resource-name>\n");
484         for (Iterator i = _urls.iterator(); i.hasNext();) {
485             sb.append("      <url-pattern>").append(i.next()).append("</url-pattern>\n");
486         }
487         sb.append("    </web-resource-collection>\n");
488         sb.append("    <auth-constraint>\n");
489         for (Iterator i = _roles.iterator(); i.hasNext();) {
490             sb.append("      <role-name>").append(i.next()).append("</role-name>\n");
491         }
492         sb.append("    </auth-constraint>\n");
493         sb.append("  </security-constraint>\n");
494         return sb.toString();
495     }
