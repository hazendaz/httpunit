20  package com.meterware.httpunit;
21  
22  import java.util.Map;
23  import java.util.Properties;
24  
25  /**
26   * This exception is thrown when an unauthorized request is made for a page that requires authentication.
27   **/
28  public class AuthorizationRequiredException extends RuntimeException {
29  
30      private static final long serialVersionUID = 1L;
31      public static AuthorizationRequiredException createBasicAuthenticationRequiredException(String realm) {
32          Properties props = new Properties();
33          props.put("realm", realm);
34          return new AuthorizationRequiredException("Basic", props);
35      }
36  
37      static AuthorizationRequiredException createException(String scheme, Map properties) {
38          return new AuthorizationRequiredException(scheme, properties);
39      }
40  
41      private AuthorizationRequiredException(String scheme, Map properties) {
42          _scheme = scheme;
43          _properties = properties;
44      }
45  
46      @Override
47      public String getMessage() {
48          return _scheme + " authentication required: " + _properties;
49      }
50  
51      /**
52       * Returns the name of the <a href="http://www.freesoft.org/CIE/RFC/Orig/rfc2617.txt">authentication scheme</a>.
53       *
54       * @return the scheme
55       **/
56      public String getAuthenticationScheme() {
57          return _scheme;
58      }
59  
60      /**
61       * Returns the named authentication parameter. For Basic authentication, the only parameter is "realm".
62       *
63       * @param parameterName
64       *            the name of the parameter to fetch
65       *
66       * @return the parameter, without quotes
67       **/
68      public String getAuthenticationParameter(String parameterName) {
69          return unQuote((String) _properties.get(parameterName));
70      }
71  
72      private String unQuote(String value) {
73          if (value == null || value.length() <= 1 || !value.startsWith("\"") || !value.endsWith("\"")) {
74              return value;
75          }
76  
77          return value.substring(1, value.length() - 1);
78      }
79  
80      // ------------------------------------- private members ------------------------------------------
81  
82      private String _scheme;
83      private Map _properties;
84  }
