20  package com.meterware.httpunit;
21  
22  /**
23   * A class which represents the properties of a web client.
24   *
25   * @author <a href="mailto:russgold@httpunit.org">Russell Gold</a>
26   **/
27  public class ClientProperties {
28  
29      /**
30       * Returns the current defaults for newly created web clients.
31       */
32      public static ClientProperties getDefaultProperties() {
33          return _defaultProperties;
34      }
35  
36      /**
37       * Specifies the ID information for a client.
38       */
39      public void setApplicationID(String applicationName, String applicationCodeName, String applicationVersion) {
40          _applicationCodeName = applicationCodeName;
41          _applicationName = applicationName;
42          _applicationVersion = applicationVersion;
43      }
44  
45      public String getApplicationCodeName() {
46          return _applicationCodeName;
47      }
48  
49      public void setApplicationCodeName(String applicationCodeName) {
50          _applicationCodeName = applicationCodeName;
51      }
52  
53      public String getApplicationName() {
54          return _applicationName;
55      }
56  
57      public void setApplicationName(String applicationName) {
58          _applicationName = applicationName;
59      }
60  
61      public String getApplicationVersion() {
62          return _applicationVersion;
63      }
64  
65      public void setApplicationVersion(String applicationVersion) {
66          _applicationVersion = applicationVersion;
67      }
68  
69      /**
70       * Returns the user agent identification. Unless this has been set explicitly, it will default to the application
71       * code name followed by a slash and the application version.
72       */
73      public String getUserAgent() {
74          return _userAgent != null ? _userAgent : _applicationCodeName + '/' + _applicationVersion;
75      }
76  
77      public void setUserAgent(String userAgent) {
78          _userAgent = userAgent;
79      }
80  
81      public String getPlatform() {
82          return _platform;
83      }
84  
85      public void setPlatform(String platform) {
86          _platform = platform;
87      }
88  
89      /**
90       * A shortcut for setting both availableScreenWidth and availableScreenHeight at one time.
91       */
92      public void setAvailableScreenSize(int width, int height) {
93          _availWidth = width;
94          _availHeight = height;
95      }
96  
97      public int getAvailableScreenWidth() {
98          return _availWidth;
99      }
100 
101     public void setAvailableScreenWidth(int availWidth) {
102         _availWidth = availWidth;
103     }
104 
105     public int getAvailHeight() {
106         return _availHeight;
107     }
108 
109     public void setAvailHeight(int availHeight) {
110         _availHeight = availHeight;
111     }
112 
113     /**
114      * Returns true if the client should accept and transmit cookies. The default is to accept them.
115      */
116     public boolean isAcceptCookies() {
117         return _acceptCookies;
118     }
119 
120     /**
121      * Specifies whether the client should accept and send cookies.
122      */
123     public void setAcceptCookies(boolean acceptCookies) {
124         _acceptCookies = acceptCookies;
125     }
126 
127     /**
128      * Returns true if the client will accept GZIP encoding of responses. The default is to accept GZIP encoding.
129      **/
130     public boolean isAcceptGzip() {
131         return _acceptGzip;
132     }
133 
134     /**
135      * Specifies whether the client will accept GZIP encoded responses. The default is true.
136      */
137     public void setAcceptGzip(boolean acceptGzip) {
138         _acceptGzip = acceptGzip;
139     }
140 
141     /**
142      * get Maximum number of redirect requests
143      *
144      * @return it
145      */
146     public int getMaxRedirects() {
147         return _maxRedirects;
148     }
149 
150     /**
151      * set the maximum number of redirects
152      *
153      * @param maxRedirects
154      */
155     public void setMaxRedirects(int maxRedirects) {
156         _maxRedirects = maxRedirects;
157     }
158 
159     /**
160      * Returns true if the client should automatically follow page redirect requests (status 3xx). By default, this is
161      * true.
162      **/
163     public boolean isAutoRedirect() {
164         return _autoRedirect;
165     }
166 
167     /**
168      * Determines whether the client should automatically follow page redirect requests (status 3xx). By default, this
169      * is true in order to simulate normal browser operation.
170      **/
171     public void setAutoRedirect(boolean autoRedirect) {
172         _autoRedirect = autoRedirect;
173     }
174 
175     /**
176      * Returns true if the client should automatically follow page refresh requests. By default, this is false, so that
177      * programs can verify the redirect page presented to users before the browser switches to the new page.
178      **/
179     public boolean isAutoRefresh() {
180         return _autoRefresh;
181     }
182 
183     /**
184      * Specifies whether the client should automatically follow page refresh requests. By default, this is false, so
185      * that programs can verify the redirect page presented to users before the browser switches to the new page.
186      * Setting this to true can cause an infinite loop on pages that refresh themselves.
187      **/
188     public void setAutoRefresh(boolean autoRefresh) {
189         _autoRefresh = autoRefresh;
190     }
191 
192     public boolean isIframeSupported() {
193         return _iframeSupported;
194     }
195 
196     public void setIframeSupported(boolean iframeSupported) {
197         _iframeSupported = iframeSupported;
198     }
199 
200     /**
201      * @deprecated since 1.8 see BR 2595566 - name of getter is a typo
202      *
203      * @see getOverrideContentType
204      *
205      * @return the overriding content type
206      */
207     @Deprecated
208     public String getOverrideContextType() {
209         return getOverrideContentType();
210     }
211 
212     /**
213      * @deprecated since 1.8 see BR 2595566 - name of setter is a typo
214      *
215      * @param overrideContentType
216      *            the content type
217      *
218      * @see setOverrideContentType
219      */
220     @Deprecated
221     public void setOverrideContextType(String overrideContentType) {
222         setOverrideContentType(overrideContentType);
223     }
224 
225     /**
226      * Returns the content type (if any) to use instead of the one specified by the server. Defaults to null.
227      *
228      * @return the overriding content type, or null if none is specified.
229      */
230     public String getOverrideContentType() {
231         return _overrideContentType;
232     }
233 
234     /**
235      * All responses to this client will use the specified content type rather than the one specified by the server.
236      * Setting this to "text/html" will force all reponses to be interpreted as HTML.
237      *
238      * @param overrideContentType
239      *            the new override to apply to context types.
240      */
241     public void setOverrideContentType(String overrideContentType) {
242         _overrideContentType = overrideContentType;
243     }
244 
245     /**
246      * Specifies a listener for DNS requests from the client.
247      *
248      * @param dnsListener
249      *            the new listener.
250      */
251     public void setDnsListener(DNSListener dnsListener) {
252         _dnsListener = dnsListener;
253     }
254 
255     /**
256      * Returns the listener for DNS requests to be used by the client.
257      *
258      * @return the currently specified DNS listener, or null if none is specified.
259      */
260     DNSListener getDnsListener() {
261         return _dnsListener;
262     }
263 
264     /**
265      * @return the whether Referer information should be stripped from the header
266      */
267     public boolean isSendReferer() {
268         return _sendReferer;
269     }
270 
271     /**
272      * set whether Referer information should be stripped
273      *
274      * @param referer
275      *            the _sendReferer to set
276      */
277     public void setSendReferer(boolean referer) {
278         _sendReferer = referer;
279     }
280 
281     ClientProperties cloneProperties() {
282         return new ClientProperties(this);
283     }
284 
285     private String _applicationCodeName = "httpunit";
286     private String _applicationName = "HttpUnit";
287     private String _applicationVersion = "1.5";
288     private String _userAgent;
289     private String _platform = "Java";
290     private String _overrideContentType = null;
291     private int _availWidth = 800;
292     private int _availHeight = 600;
293     private int _maxRedirects = 5;
294 
295     private boolean _iframeSupported = true;
296     private boolean _acceptCookies = true;
297     private boolean _acceptGzip = true;
298     private boolean _autoRedirect = true;
299     private boolean _autoRefresh = false;
300 
301     private DNSListener _dnsListener;
302     private boolean _sendReferer;
303 
304     private static ClientProperties _defaultProperties = new ClientProperties();
305 
306     /**
307      * default Constructor
308      */
309     private ClientProperties() {
310         _sendReferer = true;
311     }
312 
313     /**
314      * copy constructor
315      *
316      * @param source
317      *            - the ClientProperties to copy from
318      */
319     private ClientProperties(ClientProperties source) {
320         _applicationCodeName = source._applicationCodeName;
321         _applicationName = source._applicationName;
322         _applicationVersion = source._applicationVersion;
323         _userAgent = source._userAgent;
324         _platform = source._platform;
325         _overrideContentType = source._overrideContentType;
326         _iframeSupported = source._iframeSupported;
327         _acceptCookies = source._acceptCookies;
328         _acceptGzip = source._acceptGzip;
329         _autoRedirect = source._autoRedirect;
330         _autoRefresh = source._autoRefresh;
331         _sendReferer = source._sendReferer;
332         _maxRedirects = source._maxRedirects;
333     }
334 
335 }
