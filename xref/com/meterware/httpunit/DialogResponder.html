20  package com.meterware.httpunit;
21  
22  /**
23   * Interface for an object to supply user responses to dialogs.
24   *
25   * @author <a href="mailto:russgold@acm.org">Russell Gold</a>
26   **/
27  public interface DialogResponder {
28  
29      /**
30       * Invoked when the user agent needs to display a confirmation dialog. This method should return true to accept the
31       * proposed action or false to reject it.
32       */
33      boolean getConfirmation(String confirmationPrompt);
34  
35      /**
36       * Invoked when the user agent needs to display a generic dialog and obtain a user response. This method should
37       * return the user's answer.
38       */
39      String getUserResponse(String prompt, String defaultResponse);
40  
41  }
