20  package com.meterware.httpunit;
21  
22  import java.util.HashMap;
23  import java.util.Iterator;
24  import java.util.Map;
25  
26  import org.w3c.dom.Node;
27  
28  /**
29   * @author <a href="mailto:russgold@httpunit,org">Russell Gold</a>
30   */
31  class ElementRegistry {
32  
33      private Map _map = new HashMap();
34  
35      /**
36       * Registers an HttpUnit element for a node.
37       *
38       * @return the registered element
39       */
40      Object registerElement(Node node, HTMLElement htmlElement) {
41          _map.put(node, htmlElement);
42          return htmlElement;
43      }
44  
45      /**
46       * Returns the HttpUnit element associated with the specified DOM element, if any.
47       */
48      Object getRegisteredElement(Node node) {
49          return _map.get(node);
50      }
51  
52      Iterator iterator() {
53          return _map.values().iterator();
54      }
55  
56      boolean hasNode(Node node) {
57          return _map.containsKey(node);
58      }
59  }
