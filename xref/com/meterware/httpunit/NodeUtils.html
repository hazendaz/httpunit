33  /**
34   * Some common utilities for manipulating DOM nodes.

36  public class NodeUtils {

38      /**
39       * get the attribute with the given name from the given node as an int value.
40       *
41       * @param node
42       *            - the node to look in
43       * @param attributeName
44       *            - the attribute's name to look for
45       * @param defaultValue
46       *            the default value
47       *
48       * @return - the value - defaultValue as default
49       */
50      public static int getAttributeValue(Node node, String attributeName, int defaultValue) {
51          NamedNodeMap nnm = node.getAttributes();
52          Node attribute = nnm.getNamedItem(attributeName);
53          if (attribute == null) {
54              return defaultValue;
55          }
56          try {
57              return Integer.parseInt(attribute.getNodeValue());
58          } catch (NumberFormatException e) {
59              return defaultValue;
60          }
61      }
62  
63      /**
64       * get the attribute with the given name from the given node.
65       *
66       * @param node
67       *            - the node to look in
68       * @param attributeName
69       *            - the attribute's name to look for
70       *
71       * @return - the value - "" as default
72       */
73      public static String getNodeAttribute(Node node, String attributeName) {
74          return getNodeAttribute(node, attributeName, "");
75      }
76  
77      /**
78       * get the attribute with the given name from the given node.
79       *
80       * @param node
81       *            - the node to look in
82       * @param attributeName
83       *            - the attribute's name to look for
84       * @param defaultValue
85       *            the default value
86       *
87       * @return - the value - defaultValue as default
88       */
89      public static String getNodeAttribute(Node node, String attributeName, String defaultValue) {
90          NamedNodeMap attributes = node.getAttributes();
91          if (attributes == null) {
92              return defaultValue;
93          }
94  
95          Node attribute = attributes.getNamedItem(attributeName);
96          return attribute == null ? defaultValue : attribute.getNodeValue();
97      }

99      /**
100      * set the attribute with the given attribute to the given value in the given node.
101      *
102      * @param node
103      *            the node
104      * @param attributeName
105      *            - the attribute's name to look for
106      * @param value
107      *            - the value to set
108      */
109     static void setNodeAttribute(Node node, String attributeName, String value) {
110         ((Element) node).setAttributeNS(null, attributeName, value);
111     }

121     static void removeNodeAttribute(Node node, String attributeName) {

139     /**
140      * common Node action methods.
141      */
142     interface NodeAction {
143 
144         /**
145          * Does appropriate processing on specified element. Will return false if the subtree below the element should
146          * be skipped.
147          *
148          * @param traversal
149          *            the traversal
150          * @param element
151          *            the element
152          *
153          * @return true, if successful
154          */
155         boolean processElement(PreOrderTraversal traversal, Element element);
156 
157         /**
158          * Processes a text node.
159          *
160          * @param traversal
161          *            the traversal
162          * @param textNode
163          *            the text node
164          */
165         void processTextNode(PreOrderTraversal traversal, Node textNode);
166     }
167 
168     /**
169      * Converts the DOM trees rooted at the specified nodes to text, ignoring any HTML tags.
170      *
171      * @param rootNodes
172      *            the root nodes
173      *
174      * @return the string
175      */
176     public static String asText(NodeList rootNodes) {
177         final StringBuilder sb = new StringBuilder(HttpUnitUtils.DEFAULT_TEXT_BUFFER_SIZE);
178         NodeAction action = new NodeAction() {
179             @Override
180             public boolean processElement(PreOrderTraversal traversal, Element node) {
181                 String nodeName = node.getNodeName();
182                 if (nodeName.equalsIgnoreCase("p") || nodeName.equalsIgnoreCase("br")
183                         || nodeName.equalsIgnoreCase("tr")) {
184                     sb.append("\n");
185                 } else if (nodeName.equalsIgnoreCase("td") || nodeName.equalsIgnoreCase("th")) {
186                     sb.append(" | ");
187                 } else if (nodeName.equalsIgnoreCase("img") && HttpUnitOptions.getImagesTreatedAsAltText()) {
188                     sb.append(getNodeAttribute(node, "alt"));
189                 }
190                 return true;
191             }
192 
193             @Override
194             public void processTextNode(PreOrderTraversal traversal, Node textNode) {
195                 sb.append(HTMLParserFactory.getHTMLParser().getCleanedText(textNode.getNodeValue()));
196             }
197         };
198         new PreOrderTraversal(rootNodes).perform(action);
199         return sb.toString();
200     }

202     /**
203      * The Class PreOrderTraversal.
204      */
205     static class PreOrderTraversal {
206 
207         /** The pending nodes. */
208         private Stack _pendingNodes = new Stack();
209 
210         /** The traversal context. */
211         private Stack _traversalContext = new Stack();
212 
213         /** The Constant POP_CONTEXT. */
214         private static final Object POP_CONTEXT = new Object();
215 
216         /**
217          * Instantiates a new pre order traversal.
218          *
219          * @param rootNodes
220          *            the root nodes
221          */
222         public PreOrderTraversal(NodeList rootNodes) {
223             pushNodeList(rootNodes);
224         }
225 
226         /**
227          * Instantiates a new pre order traversal.
228          *
229          * @param rootNode
230          *            the root node
231          */
232         public PreOrderTraversal(Node rootNode) {
233             pushNodeList(rootNode.getLastChild());
234         }
235 
236         /**
237          * Push base context.
238          *
239          * @param context
240          *            the context
241          */
242         public void pushBaseContext(Object context) {
243             _traversalContext.push(context);
244         }
245 
246         /**
247          * Push context.
248          *
249          * @param context
250          *            the context
251          */
252         public void pushContext(Object context) {
253             _traversalContext.push(context);
254             _pendingNodes.push(POP_CONTEXT);
255         }
256 
257         /**
258          * Gets the contexts.
259          *
260          * @return the contexts
261          */
262         public Iterator getContexts() {
263             Stack stack = _traversalContext;
264             return getTopDownIterator(stack);
265         }
266 
267         /**
268          * Gets the root context.
269          *
270          * @return the root context
271          */
272         public Object getRootContext() {
273             return _traversalContext.firstElement();
274         }
275 
276         /**
277          * Gets the top down iterator.
278          *
279          * @param stack
280          *            the stack
281          *
282          * @return the top down iterator
283          */
284         private Iterator getTopDownIterator(final Stack stack) {
285             return new Iterator() {
286                 private ListIterator _forwardIterator = stack.listIterator(stack.size());
287 
288                 @Override
289                 public boolean hasNext() {
290                     return _forwardIterator.hasPrevious();
291                 }
292 
293                 @Override
294                 public Object next() {
295                     return _forwardIterator.previous();
296                 }
297 
298                 @Override
299                 public void remove() {
300                     _forwardIterator.remove();
301                 }
302             };
303         }
304 
305         /**
306          * Returns the most recently pushed context which implements the specified class. Will return null if no
307          * matching context is found.
308          *
309          * @param matchingClass
310          *            the matching class
311          *
312          * @return the closest context
313          */
314         public Object getClosestContext(Class matchingClass) {
315             for (int i = _traversalContext.size() - 1; i >= 0; i--) {
316                 Object o = _traversalContext.elementAt(i);
317                 if (matchingClass.isInstance(o)) {
318                     return o;
319                 }
320             }
321             return null;
322         }
323 
324         /**
325          * Perform.
326          *
327          * @param action
328          *            the action
329          */
330         public void perform(NodeAction action) {
331             while (!_pendingNodes.empty()) {
332                 final Object object = _pendingNodes.pop();
333                 if (object == POP_CONTEXT) {
334                     _traversalContext.pop();
335                 } else {
336                     Node node = (Node) object;
337                     if (node.getNodeType() == Node.TEXT_NODE) {
338                         action.processTextNode(this, node);
339                     } else if (node.getNodeType() != Node.ELEMENT_NODE) {
340                         continue;
341                     } else {
342                         action.processElement(this, (Element) node);
343                     }
344                     pushNodeList(node.getLastChild());
345                 }
346             }
347         }
348 
349         /**
350          * Push node list.
351          *
352          * @param nl
353          *            the nl
354          */
355         private void pushNodeList(NodeList nl) {
356             if (nl != null) {
357                 for (int i = nl.getLength() - 1; i >= 0; i--) {
358                     _pendingNodes.push(nl.item(i));
359                 }
360             }
361         }
362 
363         /**
364          * Push node list.
365          *
366          * @param lastChild
367          *            the last child
368          */
369         private void pushNodeList(Node lastChild) {
370             for (Node node = lastChild; node != null; node = node.getPreviousSibling()) {
371                 _pendingNodes.push(node);
372             }
373         }
374     }
