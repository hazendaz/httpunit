20  package com.meterware.httpunit;
21  
22  import java.net.URL;
23  
24  import org.w3c.dom.html.HTMLTableCellElement;
25  
26  /**
27   * A single cell in an HTML table.
28   **/
29  public class TableCell extends BlockElement {
30  
31      private HTMLTableCellElement _element;
32  
33      /**
34       * Returns the number of columns spanned by this cell.
35       **/
36      public int getColSpan() {
37          return _element.getColSpan();
38      }
39  
40      /**
41       * Returns the number of rows spanned by this cell.
42       **/
43      public int getRowSpan() {
44          return _element.getRowSpan();
45      }
46  
47      /**
48       * Returns the text value of this cell.
49       *
50       * @deprecated as of 1.6, use #getText()
51       */
52      @Deprecated
53      public String asText() {
54          return getText();
55      }
56  
57      // ---------------------------------------- package methods -----------------------------------------
58  
59      TableCell(WebResponse response, FrameSelector frame, HTMLTableCellElement element, URL url, String parentTarget,
60              String characterSet) {
61          super(response, frame, url, parentTarget, element, characterSet);
62          _element = element;
63      }
64  
65  }
