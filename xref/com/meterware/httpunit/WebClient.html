444     /**
445      * Returns the value of all current header fields.
446      **/
447     protected Dictionary getHeaderFields(URL targetURL) {
448         Hashtable result = (Hashtable) _headers.clone();
449         result.put("User-Agent", getClientProperties().getUserAgent());
450         if (getClientProperties().isAcceptGzip()) {
451             result.put("Accept-Encoding", "gzip");
452         }
453         AddHeaderIfNotNull(result, "Cookie", _cookieJar.getCookieHeaderField(targetURL));
454         if (_authorizationString == null) {
455             _authorizationString = _fixedAuthorizationString;
456         }
457         AddHeaderIfNotNull(result, "Authorization", _authorizationString);
458         AddHeaderIfNotNull(result, "Proxy-Authorization", _proxyAuthorizationString);
459         _authorizationString = null;
460         return result;
461     }
462 
463     private void AddHeaderIfNotNull(Hashtable result, final String headerName, final String headerValue) {
464         if (headerValue != null) {
465             result.put(headerName, headerValue);
466         }

523     void updateFrameContents(WebWindow requestWindow, String requestTarget, WebResponse response,
524             RequestContext requestContext) throws IOException, SAXException {
525         if (response.getFrame() == FrameSelector.NEW_FRAME) {
526             WebWindow window = new WebWindow(this, requestWindow.getCurrentPage());
527             if (!WebRequest.NEW_WINDOW.equalsIgnoreCase(requestTarget)) {
528                 window.setName(requestTarget);
529             }
530             response.setFrame(window.getTopFrame());
531             window.updateFrameContents(response, requestContext);
532             _openWindows.add(window);
533             reportWindowOpened(window);
534         } else if (response.getFrame().getWindow() != null && response.getFrame().getWindow() != requestWindow) {
535             response.getFrame().getWindow().updateFrameContents(response, requestContext);
536         } else {
537             if (response.getFrame() == FrameSelector.TOP_FRAME) {
538                 response.setFrame(requestWindow.getTopFrame());
539             }
540             requestWindow.updateFrameContents(response, requestContext);
541         }
542     }

544     void close(WebWindow window) {
545         if (!_openWindows.contains(window)) {
546             throw new IllegalStateException("Window is already closed");
547         }
548         _openWindows.remove(window);
549         if (_openWindows.isEmpty()) {
550             _openWindows.add(new WebWindow(this));
551         }
552         if (window.equals(_mainWindow)) {
553             _mainWindow = (WebWindow) _openWindows.get(0);
554         }
555         reportWindowClosed(window);
556     }

623     /**
624      * Examines the headers in the response and throws an exception if appropriate.
625      *
626      * @parm response - the response to validate
627      **/
628     private void validateHeaders(WebResponse response) throws HttpException {
629         HttpException exception = null;
630         if (response.getResponseCode() == HttpURLConnection.HTTP_INTERNAL_ERROR) {
631             exception = new HttpInternalErrorException(response.getURL());
632         } else if (response.getResponseCode() == HttpURLConnection.HTTP_NOT_FOUND) {
633             exception = new HttpNotFoundException(response.getResponseMessage(), response.getURL());
634         } else if (response.getResponseCode() >= HttpURLConnection.HTTP_BAD_REQUEST) {
635             exception = new HttpException(response.getResponseCode(), response.getResponseMessage(), response.getURL());
636         }
637         // is there an exception?
638         if (exception != null) {
639             // see feature request [ 914314 ] Add HttpException.getResponse for better reporting
640             exception.setResponse(response);
641             // shall we ignore errors?
642             if (!getExceptionsThrownOnErrorStatus()) {
643                 return;
644             }
645             throw exception;
646         }
647     }

660     /**
661      * Sends a request and returns a response after dealing with any authentication challenge. If challenged and able to
662      * respond, resends the request after setting the authentication header (which will apply only for the that
663      * request).
664      *
665      * @param request
666      *            the original request
667      * @param targetFrame
668      *            the frame into which the result will be stored
669      *
670      * @return a response from the server
671      *
672      * @throws IOException
673      *             if an exception (including authorization failure) occurs
674      */
675     WebResponse createResponse(WebRequest request, FrameSelector targetFrame) throws IOException {
676         WebResponse response = newResponse(request, targetFrame);
677         AuthenticationChallenge challenge = new AuthenticationChallenge(this, request,
678                 response.getHeaderField("WWW-Authenticate"));
679         if (!challenge.needToAuthenticate()) {
680             return response;
681         }
682         setOnetimeAuthenticationHeader(challenge.createAuthenticationHeader());
683         WebResponse response2 = newResponse(request, targetFrame);
684         if (response2.getHeaderField("WWW-Authenticate") != null && getExceptionsThrownOnErrorStatus()) {
685             throw AuthenticationChallenge.createException(response2.getHeaderField("WWW-Authenticate"));
686         }
687         return response2;
688     }

690     private void setOnetimeAuthenticationHeader(String authorizationHeader) {
691         _authorizationString = authorizationHeader;
692     }

696     static public class HeaderDictionary extends Hashtable {
697 
698         private static final long serialVersionUID = 1L;
699 
700         public void addEntries(Dictionary source) {
701             for (Enumeration e = source.keys(); e.hasMoreElements();) {
702                 Object key = e.nextElement();
703                 put(key, source.get(key));
704             }
705         }
706 
707         @Override
708         public boolean containsKey(Object key) {
709             return super.containsKey(matchPreviousFieldName(key.toString()));
710         }
711 
712         @Override
713         public Object get(Object fieldName) {
714             return super.get(matchPreviousFieldName(fieldName.toString()));
715         }
716 
717         @Override
718         public Object put(Object fieldName, Object fieldValue) {
719             fieldName = matchPreviousFieldName(fieldName.toString());
720             Object oldValue = super.get(fieldName);
721             if (fieldValue == null) {
722                 remove(fieldName);
723             } else {
724                 super.put(fieldName, fieldValue);
725             }
726             return oldValue;
727         }
728 
729         /**
730          * If a matching field name with different case is already known, returns the older name. Otherwise, returns the
731          * specified name.
732          **/
733         private String matchPreviousFieldName(String fieldName) {
734             for (Enumeration e = keys(); e.hasMoreElements();) {
735                 String key = (String) e.nextElement();
736                 if (key.equalsIgnoreCase(fieldName)) {
737                     return key;
738                 }
739             }
740             return fieldName;
741         }
742 
743     }
