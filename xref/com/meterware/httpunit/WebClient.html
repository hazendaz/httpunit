83          }

440     /**
441      * Returns the value of all current header fields.
442      **/
443     protected Dictionary getHeaderFields(URL targetURL) {
444         Hashtable result = (Hashtable) _headers.clone();
445         result.put("User-Agent", getClientProperties().getUserAgent());
446         if (getClientProperties().isAcceptGzip()) {
447             result.put("Accept-Encoding", "gzip");
448         }
449         AddHeaderIfNotNull(result, "Cookie", _cookieJar.getCookieHeaderField(targetURL));
450         if (_authorizationString == null) {
451             _authorizationString = _fixedAuthorizationString;
452         }
453         AddHeaderIfNotNull(result, "Authorization", _authorizationString);
454         AddHeaderIfNotNull(result, "Proxy-Authorization", _proxyAuthorizationString);
455         _authorizationString = null;
456         return result;
457     }

459     private void AddHeaderIfNotNull(Hashtable result, final String headerName, final String headerValue) {
460         if (headerValue != null) {

462         }

519     void updateFrameContents(WebWindow requestWindow, String requestTarget, WebResponse response,
520             RequestContext requestContext) throws IOException, SAXException {
521         if (response.getFrame() == FrameSelector.NEW_FRAME) {
522             WebWindow window = new WebWindow(this, requestWindow.getCurrentPage());
523             if (!WebRequest.NEW_WINDOW.equalsIgnoreCase(requestTarget)) {
524                 window.setName(requestTarget);
525             }
526             response.setFrame(window.getTopFrame());
527             window.updateFrameContents(response, requestContext);
528             _openWindows.add(window);
529             reportWindowOpened(window);
530         } else if (response.getFrame().getWindow() != null && response.getFrame().getWindow() != requestWindow) {
531             response.getFrame().getWindow().updateFrameContents(response, requestContext);
532         } else {
533             if (response.getFrame() == FrameSelector.TOP_FRAME) {
534                 response.setFrame(requestWindow.getTopFrame());
535             }
536             requestWindow.updateFrameContents(response, requestContext);
537         }
538     }

540     void close(WebWindow window) {
541         if (!_openWindows.contains(window)) {
542             throw new IllegalStateException("Window is already closed");
543         }
544         _openWindows.remove(window);
545         if (_openWindows.isEmpty()) {
546             _openWindows.add(new WebWindow(this));
547         }
548         if (window.equals(_mainWindow)) {
549             _mainWindow = (WebWindow) _openWindows.get(0);
550         }
551         reportWindowClosed(window);
552     }

619     /**
620      * Examines the headers in the response and throws an exception if appropriate.
621      *
622      * @parm response - the response to validate
623      **/
624     private void validateHeaders(WebResponse response) throws HttpException {
625         HttpException exception = null;
626         if (response.getResponseCode() == HttpURLConnection.HTTP_INTERNAL_ERROR) {
627             exception = new HttpInternalErrorException(response.getURL());
628         } else if (response.getResponseCode() == HttpURLConnection.HTTP_NOT_FOUND) {
629             exception = new HttpNotFoundException(response.getResponseMessage(), response.getURL());
630         } else if (response.getResponseCode() >= HttpURLConnection.HTTP_BAD_REQUEST) {
631             exception = new HttpException(response.getResponseCode(), response.getResponseMessage(), response.getURL());
632         }
633         // is there an exception?
634         if (exception != null) {
635             // see feature request [ 914314 ] Add HttpException.getResponse for better reporting
636             exception.setResponse(response);
637             // shall we ignore errors?
638             if (!getExceptionsThrownOnErrorStatus()) {
639                 return;
640             }
641             throw exception;
642         }
643     }

656     /**
657      * Sends a request and returns a response after dealing with any authentication challenge. If challenged and able to
658      * respond, resends the request after setting the authentication header (which will apply only for the that
659      * request).
660      *
661      * @param request
662      *            the original request
663      * @param targetFrame
664      *            the frame into which the result will be stored
665      *
666      * @return a response from the server
667      *
668      * @throws IOException
669      *             if an exception (including authorization failure) occurs
670      */
671     WebResponse createResponse(WebRequest request, FrameSelector targetFrame) throws IOException {
672         WebResponse response = newResponse(request, targetFrame);
673         AuthenticationChallenge challenge = new AuthenticationChallenge(this, request,
674                 response.getHeaderField("WWW-Authenticate"));
675         if (!challenge.needToAuthenticate()) {
676             return response;
677         }
678         setOnetimeAuthenticationHeader(challenge.createAuthenticationHeader());
679         WebResponse response2 = newResponse(request, targetFrame);
680         if (response2.getHeaderField("WWW-Authenticate") != null && getExceptionsThrownOnErrorStatus()) {
681             throw AuthenticationChallenge.createException(response2.getHeaderField("WWW-Authenticate"));
682         }
683         return response2;
684     }
685 
686     private void setOnetimeAuthenticationHeader(String authorizationHeader) {
687         _authorizationString = authorizationHeader;
688     }

692     static public class HeaderDictionary extends Hashtable {
693 
694         private static final long serialVersionUID = 1L;
695 
696         public void addEntries(Dictionary source) {
697             for (Enumeration e = source.keys(); e.hasMoreElements();) {
698                 Object key = e.nextElement();
699                 put(key, source.get(key));
700             }
701         }
702 
703         @Override
704         public boolean containsKey(Object key) {
705             return super.containsKey(matchPreviousFieldName(key.toString()));
706         }
707 
708         @Override
709         public Object get(Object fieldName) {
710             return super.get(matchPreviousFieldName(fieldName.toString()));
711         }
712 
713         @Override
714         public Object put(Object fieldName, Object fieldValue) {
715             fieldName = matchPreviousFieldName(fieldName.toString());
716             Object oldValue = super.get(fieldName);
717             if (fieldValue == null) {
718                 remove(fieldName);
719             } else {
720                 super.put(fieldName, fieldValue);
721             }
722             return oldValue;
723         }
724 
725         /**
726          * If a matching field name with different case is already known, returns the older name. Otherwise, returns the
727          * specified name.
728          **/
729         private String matchPreviousFieldName(String fieldName) {
730             for (Enumeration e = keys(); e.hasMoreElements();) {
731                 String key = (String) e.nextElement();
732                 if (key.equalsIgnoreCase(fieldName)) {
733                     return key;
734                 }
735             }
736             return fieldName;
737         }
738 
739     }
