289     /**
290      * Creates and returns a web request which will simulate the submission of this form by pressing the specified
291      * button. If the button is null, simulates the pressing of the default button.
292      *
293      * @param button
294      *            - the submitbutton to be pressed - may be null
295      * @param x
296      *            - the x position
297      * @param y
298      *            - the y position
299      **/
300     public WebRequest getRequest(SubmitButton button, int x, int y) {
301         if (button == null) {
302             button = getDefaultButton();
303         }
304 
305         if (HttpUnitOptions.getParameterValuesValidated()) {
306             if (button == null) {
307                 throw new IllegalUnnamedSubmitButtonException();
308             }
309             if (button.isFake()) {
310                 // bypass checks
311             } else if (!getSubmitButtonVector().contains(button)) {
312                 throw new IllegalSubmitButtonException(button);
313             } else if (!button.wasEnabled()) {
314                 // this is too late for the check of isDisabled()
315                 // onclick has already been done ...
316                 // [ 1289151 ] Order of events in button.click() is wrong
317                 button.throwDisabledException();
318             }
319         }
320 
321         SubmitButton[] buttons = getSubmitButtons();
322         for (SubmitButton button2 : buttons) {
323             button2.setPressed(false);
324         }
325         button.setPressed(true);
326 
327         if (getMethod().equalsIgnoreCase("post")) {
328             return new PostMethodWebRequest(this, button, x, y);
329         }
330         return new GetMethodWebRequest(this, WebRequest.newParameterHolder(this), button, x, y);
331     }

342     /**
343      * Creates and returns a web request which includes the specified button and position. If no button is specified,
344      * will include the default button, if any. No parameter validation will be done on the returned request and no
345      * scripts will be run when it is submitted.
346      **/
347     public WebRequest newUnvalidatedRequest(SubmitButton button, int x, int y) {
348         if (button == null) {
349             button = getDefaultButton();
350         }
351 
352         SubmitButton[] buttons = getSubmitButtons();
353         for (SubmitButton button2 : buttons) {
354             button2.setPressed(false);
355         }
356         button.setPressed(true);
357 
358         if (getMethod().equalsIgnoreCase("post")) {
359             return new PostMethodWebRequest(this, new UncheckedParameterHolder(this), button, x, y);
360         }
361         return new GetMethodWebRequest(this, new UncheckedParameterHolder(this), button, x, y);
362     }

606     /**
607      * Sets the multiple values of a parameter in this form. This is generally used when there are multiple controls
608      * with the same name in the form.
609      *
610      * @param name
611      * @param values
612      */
613     @Override
614     public void setParameter(String name, final String[] values) {
615         FormParameter parameter = getParameter(name);
616         if (parameter.isUnknown()) {
617             throw new NoSuchParameterException(name);
618         }
619         if (parameter.isFileParameter()) {
620             if (values != NO_VALUES) {
621                 throw new InvalidFileParameterException(name, values);
622             }
623             parameter.setFiles(new UploadFileSpec[0]);
624         } else {
625             parameter.setValues(values);
626         }
627 
628     }

784          *            - the name of the property to get
785          *
786          * @return the Object for the property
787          */
788         @Override
789         public Object get(String propertyName) {
790             if (propertyName.equals("target")) {
791                 return getTarget();
792             }
793             if (propertyName.equals("action")) {
794                 return getAction();
795             }
796             if (propertyName.equals("length")) {
797                 return Integer.valueOf(getFormControls().length);
798             }
799             final FormParameter parameter = getParameter(propertyName);
800             if (!parameter.isUnknown()) {
801                 return parameter.getScriptableObject();
802             }
803             FormControl control = getControlWithID(propertyName);
804             return control == null ? super.get(propertyName) : control.getScriptingHandler();
805         }
806 
807         /**
808          * Sets the value of the named property. Will throw a runtime exception if the property does not exist or cannot
809          * accept the specified value.
810          *
811          * @param propertyName
812          *            - the name of the property
813          * @param value
814          *            - the new value
815          **/
816         @Override
817         public void set(String propertyName, Object value) {
818             if (propertyName.equals("target")) {
819                 setTargetAttribute(value.toString());
820             } else if (propertyName.equals("action")) {
821                 setAction(value.toString());
822             } else if (propertyName.equals("name")) {
823                 getElement().setAttribute("name", value.toString());
824             } else if (value instanceof String) {
825                 setParameterValue(propertyName, (String) value);
826             } else if (value instanceof Number) {
827                 setParameterValue(propertyName, HttpUnitUtils.trimmedValue((Number) value));
828             } else {
829                 super.set(propertyName, value);
830             }
831         }
832 
833         @Override
834         public void setParameterValue(String name, String value) {
835             final Object scriptableObject = getParameter(name).getScriptableObject();
836             if (scriptableObject instanceof ScriptableDelegate) {
837                 ((ScriptableDelegate) scriptableObject).set("value", value);
838             } else if (scriptableObject instanceof ScriptableDelegate[]) {
839                 ((ScriptableDelegate[]) scriptableObject)[0].set("value", value);
840             }
841         }

970     /**
971      * Returns a map of parameter name to form parameter objects. Each form parameter object represents the set of form
972      * controls with a particular name. Unnamed parameters are ignored.
973      */
974     private Map getFormParameters() {
975         Map formParameters = new HashMap();
976         loadFormParameters(formParameters, getPresetParameters());
977         loadFormParameters(formParameters, getFormControls());
978         return formParameters;
979     }
980 
981     private void loadFormParameters(Map formParameters, FormControl[] controls) {
982         for (FormControl control : controls) {
983             if (control.getName().length() == 0) {
984                 continue;
985             }
986             FormParameter parameter = (FormParameter) formParameters.get(control.getName());
987             if (parameter == null) {
988                 parameter = new FormParameter();
989                 formParameters.put(control.getName(), parameter);
990             }
991             parameter.addControl(control);
992         }
993     }
