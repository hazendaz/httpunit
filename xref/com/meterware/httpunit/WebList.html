20  package com.meterware.httpunit;
21  
22  import com.meterware.httpunit.scripting.ScriptableDelegate;
23  
24  import java.net.URL;
25  import java.util.ArrayList;
26  
27  import org.w3c.dom.Element;
28  
29  /**
30   * Represents an HTML list. Experimental.
31   *
32   * @author <a href="mailto:russgold@httpunit.org">Russell Gold</a>
33   *
34   * @since 1.6
35   **/
36  public class WebList extends HTMLElementBase {
37  
38      /** Indicator of an ordered list (HTML tag &lt;ol&gt;) */
39      public static final int ORDERED_LIST = 1;
40  
41      /** Indicator of a bullet list (HTML tag &lt;ul&gt;) */
42      public static final int BULLET_LIST = 2;
43  
44      private WebResponse _response;
45      private FrameSelector _frame;
46      private URL _baseURL;
47      private String _baseTarget;
48      private String _characterSet;
49  
50      private ArrayList _items = new ArrayList();
51      private int _listType;
52  
53      public WebList(WebResponse response, FrameSelector frame, URL baseURL, String baseTarget, Element element,
54              String characterSet) {
55          super(element);
56          if (element.getNodeName().equalsIgnoreCase("ol")) {
57              _listType = ORDERED_LIST;
58          } else if (element.getNodeName().equalsIgnoreCase("ul")) {
59              _listType = BULLET_LIST;
60          }
61          _response = response;
62          _frame = frame;
63          _baseURL = baseURL;
64          _baseTarget = baseTarget;
65          _characterSet = characterSet;
66      }
67  
68      public int getListType() {
69          return _listType;
70      }
71  
72      public TextBlock[] getItems() {
73          return (TextBlock[]) _items.toArray(new TextBlock[_items.size()]);
74      }
75  
76      @Override
77      public ScriptableDelegate newScriptable() {
78          return new HTMLElementScriptable(this);
79      }
80  
81      @Override
82      public ScriptableDelegate getParentDelegate() {
83          return _response.getDocumentScriptable();
84      }
85  
86      TextBlock addNewItem(Element element) {
87          TextBlock listItem = new TextBlock(_response, _frame, _baseURL, _baseTarget, element, _characterSet);
88          _items.add(listItem);
89          return listItem;
90      }
91  }
