137     /**
138      * Submits a web request and returns a response, using all state developed so far as stored in cookies as requested
139      * by the server. see patch [ 1155415 ] Handle redirect instructions which can lead to a loop
140      *
141      * @exception SAXException
142      *                thrown if there is an error parsing the retrieved page
143      *
144      * @return the WebResponse or null
145      **/
146     public WebResponse getResponse(WebRequest request) throws IOException, SAXException {
147         // Need to have some sort of ExecuteAroundMethod to ensure that the
148         // redirects data structure gets cleared down upon exit - not
149         // straightforward, since this could be a recursive call
150         if (_isInitialRequest) {
151             _initialRequest = request;
152             _isInitialRequest = false;
153         }
154 
155         WebResponse result = null;
156 
157         try {
158             final RequestContext requestContext = new RequestContext();
159             final WebResponse response = getSubframeResponse(request, requestContext);
160             requestContext.runScripts();
161             // javascript might replace the response in its frame
162             result = response == null ? null : response.getWindow().getFrameContents(response.getFrame());
163         } finally {
164             if (null != request && request.equals(_initialRequest)) {
165                 _redirects.clear();
166                 _initialRequest = null;
167                 _isInitialRequest = true;
168             }
169         }
170         return result;
171     }

173     /**
174      * get a Response from a SubFrame
175      *
176      * @param request
177      * @param requestContext
178      *
179      * @return the WebResponse or null
180      *
181      * @throws IOException
182      * @throws SAXException
183      */
184     WebResponse getSubframeResponse(WebRequest request, RequestContext requestContext)
185             throws IOException, SAXException {
186         WebResponse response = getResource(request);
187 
188         return response == null ? null : updateWindow(request.getTarget(), response, requestContext);
189     }

191     /**
192      * Updates this web client based on a received response. This includes updating cookies and frames.
193      **/
194     WebResponse updateWindow(String requestTarget, WebResponse response, RequestContext requestContext)
195             throws IOException, SAXException {
196         _client.updateClient(response);
197         if (getClient().getClientProperties().isAutoRefresh() && response.getRefreshRequest() != null) {
198             WebRequest request = response.getRefreshRequest();
199             return getResponse(request);
200         }
201         if (shouldFollowRedirect(response)) {
202             delay(HttpUnitOptions.getRedirectDelay());
203             return getResponse(new RedirectWebRequest(response));
204         }
205         _client.updateFrameContents(this, requestTarget, response, requestContext);
206         return response;
207     }

209     /**
210      * Returns the resource specified by the request. Does not update the window or load included framesets. May return
211      * null if the resource is a JavaScript URL which would normally leave the client unchanged.
212      */
213     public WebResponse getResource(WebRequest request) throws IOException {
214         _client.tellListeners(request);
215 
216         WebResponse response = null;
217         String urlString = request.getURLString().trim();
218         FrameSelector targetFrame = _frameContents.getTargetFrame(request);
219         if (urlString.startsWith("about:")) {
220             response = new DefaultWebResponse(_client, targetFrame, null, "");
221         } else if (!HttpUnitUtils.isJavaScriptURL(urlString)) {
222             response = _client.createResponse(request, targetFrame);
223         } else {
224             ScriptingHandler handler = request.getSourceScriptingHandler();
225             if (handler == null) {
226                 handler = getCurrentPage().getScriptingHandler();
227             }
228             Object result = handler.evaluateExpression(urlString);
229             if (result != null) {
230                 response = new DefaultWebResponse(_client, targetFrame, request.getURL(), result.toString());
231             }
232         }
233 
234         if (response != null) {
235             _client.tellListeners(response);
236         }
237         return response;
238     }

321     /**
322      * Delays the specified amount of time.
323      **/
324     private void delay(int numMilliseconds) {
325         if (numMilliseconds == 0) {
326             return;
327         }
328         try {
329             Thread.sleep(numMilliseconds);
330         } catch (InterruptedException e) {
331             // ignore the exception
332         }
333     }

335     /**
336      * check whether redirect is configured
337      *
338      * @param response
339      *
340      * @return
341      */
342     private boolean redirectConfigured(WebResponse response) {
343         boolean isAutoredirect = getClient().getClientProperties().isAutoRedirect();
344         boolean hasLocation = response.getHeaderField("Location") != null;
345         int responseCode = response.getResponseCode();
346         return isAutoredirect && responseCode >= HttpURLConnection.HTTP_MOVED_PERM
347                 && responseCode <= HttpURLConnection.HTTP_MOVED_TEMP && hasLocation;
348     }
349 
350     /**
351      * check wether we should follow the redirect given in the response make sure we don't run into a recursion
352      *
353      * @param response
354      *
355      * @return
356      */
357     private boolean shouldFollowRedirect(WebResponse response) {
358         // first check whether redirect is configured for this response
359         // this is the old pre [ 1155415 ] Handle redirect instructions which can lead to a loop
360         // shouldFollowRedirect method - just renamed
361         if (!redirectConfigured(response)) {
362             return false;
363         }
364         // now do the recursion check
365         String redirectLocation = response.getHeaderField("Location");
366 
367         URL url = null;
368 
369         try {
370             if (redirectLocation != null) {
371                 url = new URL(response.getURL(), redirectLocation);
372             }
373         } catch (MalformedURLException e) {
374             // Fall through and allow existing exception handling code deal
375             // with any exception - we don't know at this stage whether it is
376             // a redirect instruction, although it is highly likely, given
377             // there is a location header present in the response!
378         }
379 
380         switch (response.getResponseCode()) {
381         case HttpURLConnection.HTTP_MOVED_PERM:
382         case HttpURLConnection.HTTP_MOVED_TEMP: // Fall through
383             int count = 0;
384             if (null != url) {
385                 Integer value = (Integer) _redirects.get(url);
386                 if (null != value) {
387                     // We have already been instructed to redirect to that
388                     // location in the course of this attempt to resolve the
389                     // resource
390 
391                     count = value.intValue();
392 
393                     int maxRedirects = getClient().getClientProperties().getMaxRedirects();
394 
395                     if (count == maxRedirects) {
396                         throw new RecursiveRedirectionException(url, "Maximum number of redirects exceeded");
397                     }
398                 }
399 
400                 count++;
401                 _redirects.put(url, Integer.valueOf(count));
402             }
403             break;
404         }
405         return redirectLocation != null;
406     }
