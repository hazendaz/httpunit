20  package com.meterware.httpunit;
21  
22  import javax.xml.xpath.XPathConstants;
23  import javax.xml.xpath.XPathExpression;
24  import javax.xml.xpath.XPathExpressionException;
25  import javax.xml.xpath.XPathFactory;
26  
27  import org.w3c.dom.Document;
28  import org.w3c.dom.Node;
29  import org.w3c.dom.NodeList;
30  
31  /**
32   * Provides an HTMLElement Predicate that is capable of matching based on an XPath node specification. This allows for
33   * very advanced matching techniques. THREAD: Instances are not thread safe, each thread should create its own instance
34   * with a specific xpath. (The same instance can be used for multiple documents, each change in document will result in
35   * its internal caches being flushed).
36   *
37   * @author <a href="mailto:edA-qa@disemia.com">edA-qa mort-ora-y</a>
38   * @author <a href="mailto:stephane@mikaty.net">Stephane Mikaty</a>
39   */
40  public class XPathPredicate implements HTMLElementPredicate {
41  
42      /** XPath which dictates matching nodes, from root */
43      private XPathExpression xpath;
44      private String path;
45      // set to true for debugging
46      public static final boolean DEBUG = false;
47  
48      /**
49       * Constructs an HTMLElementPredicate that matches only those elements which match the provided XPath.
50       *
51       * @param path
52       *            [in] XPath specification of valid/matching nodes
53       *
54       * @throws XPathExpressionException
55       *             if the xpath is invalid
56       */
57      public XPathPredicate(String path) throws XPathExpressionException {
58          this.path = path;
59          this.xpath = XPathFactory.newInstance().newXPath().compile(path);
60      }
61  
62      /**
63       * debug Output for node structure
64       *
65       * @param node
66       * @param indent
67       */
68      private void debugOut(Node node, String indent) {
69          System.out.print(indent + node.getNodeName() + ":");
70          System.out.println(indent + node.getNodeValue());
71          NodeList nl = node.getChildNodes();
72          for (int i = 0; i < nl.getLength(); i++) {
73              debugOut(nl.item(i), indent + "\t");
74          }
75      }
76  
77      /**
78       * check whether the given criteria are matched for the given element
79       *
80       * @param someElement
81       *            - the element to check
82       * @param criteria
83       *            - the criteria to check
84       */
85      @Override
86      public boolean matchesCriteria(final Object someElement, final Object criteria) {
87  
88          // this condition should normally be false
89          if (!(someElement instanceof HTMLElement)) {
90              return false;
91          }
92  
93          HTMLElement htmlElement = (HTMLElement) someElement;
94  
95          Node htmlNode = htmlElement.getNode();
96          Document doc = htmlNode.getOwnerDocument();
97          if (DEBUG) {
98              debugOut(doc, "");
99          }
100 
101         NodeList nodes;
102         try {
103             nodes = (NodeList) xpath.evaluate(doc, XPathConstants.NODESET);
104             final int nodeCount = nodes.getLength();
105             for (int i = 0; i < nodeCount; i++) {
106                 if (nodes.item(i).equals(htmlNode)) {
107                     return true;
108                 }
109             }
110         } catch (XPathExpressionException e) {
111             throw new RuntimeException("unable to evaluate xpath '" + path + "'", e);
112         }
113         return false;
114     }
115 
116 }
