45  public class SelectionFormControl extends FormControl {

47      private final boolean _multiSelect;
48      private final boolean _listBox;
49  
50      private Options _selectionOptions;

52      @Override
53      public String getType() {
54          return isMultiValued() ? MULTIPLE_TYPE : SINGLE_TYPE;
55      }

57      public SelectionFormControl(WebForm form, HTMLSelectElementImpl element) {
58          super(form, element);
59          if (!element.getNodeName().equalsIgnoreCase("select")) {
60              throw new RuntimeException("Not a select element");
61          }
62  
63          int size = NodeUtils.getAttributeValue(element, "size", 0);
64          _multiSelect = NodeUtils.isNodeAttributePresent(element, "multiple");
65          _listBox = size > 1 || _multiSelect && size != 1;
66  
67          _selectionOptions = _listBox ? (Options) new MultiSelectOptions(element)
68                  : (Options) new SingleSelectOptions(element);
69      }
70  
71      @Override
72      public String[] getValues() {
73          return _selectionOptions.getSelectedValues();
74      }

76      @Override
77      public String[] getOptionValues() {
78          return _selectionOptions.getValues();
79      }
80  
81      @Override
82      public String[] getDisplayedOptions() {
83          return _selectionOptions.getDisplayedText();
84      }

94      class Scriptable extends FormControl.Scriptable {
95  
96          /**
97           * get the Object with the given property name
98           *
99           * @param propertyName
100          *            - the name of the property to get
101          *
102          * @return the Object for the property
103          */
104         @Override
105         public Object get(String propertyName) {
106             if (propertyName.equalsIgnoreCase("options")) {
107                 return _selectionOptions;
108             }
109             if (propertyName.equalsIgnoreCase("length")) {
110                 return Integer.valueOf(getOptionValues().length);
111             }
112             if (propertyName.equalsIgnoreCase("value")) {
113                 return getSelectedValue();
114             }
115             if (propertyName.equalsIgnoreCase("selectedIndex")) {
116                 return Integer.valueOf(_selectionOptions.getFirstSelectedIndex());
117             }
118             return super.get(propertyName);
119         }
120 
121         /**
122          * get the Object at the given index
123          *
124          * @param index
125          *            - the index of the object to get
126          *
127          * @return the object at the given index
128          */
129         @Override
130         public Object get(int index) {
131             return _selectionOptions.get(index);
132         }
133 
134         private String getSelectedValue() {
135             String[] values = getValues();
136             return values.length == 0 ? "" : values[0];
137         }
138 
139         /**
140          * set the property with the given name to the given value
141          *
142          * @param propertyName
143          *            - the name of the property to set
144          * @param value
145          *            - the value to assign to the property
146          */
147         @Override
148         public void set(String propertyName, Object value) {
149             if (propertyName.equalsIgnoreCase("value")) {
150                 ArrayList values = new ArrayList();
151                 values.add(value);
152                 _selectionOptions.claimUniqueValues(values);
153             } else if (propertyName.equalsIgnoreCase("selectedIndex")) {
154                 if (!(value instanceof Number)) {
155                     throw new RuntimeException("selectedIndex must be set to an integer");
156                 }
157                 _selectionOptions.setSelectedIndex(((Number) value).intValue());
158             } else if (propertyName.equalsIgnoreCase("length")) {
159                 _selectionOptions.setLength(((Number) value).intValue());
160             } else {
161                 super.set(propertyName, value);
162             }
163         }
164     }

166     @Override
167     public ScriptableDelegate newScriptable() {
168         return new Scriptable();
169     }

172         if (isReadOnly()) {
173             required.put(getName(), getValues());
174         }
175     }

177     @Override
178     protected void addValues(ParameterProcessor processor, String characterSet) throws IOException {
179         if (isDisabled()) {
180             return;
181         }
182         for (int i = 0; i < getValues().length; i++) {
183             processor.addParameter(getName(), getValues()[i], characterSet);
184         }
185     }

187     @Override
188     protected void claimUniqueValue(List values) {
189         boolean changed = _selectionOptions.claimUniqueValues(values);
190         if (changed) {
191             sendOnChangeEvent();
192         }
193     }

195     @Override
196     protected void reset() {
197         _selectionOptions.reset();
198     }

287     public abstract class Options extends ScriptableDelegate implements SelectionOptions {
288 
289         private Option[] _options;
290 
291         Options(Node selectionNode) {
292             // BR [ 1843978 ] Accessing Options in a form is in lower case
293             // calls for uppercase "option" here ... pending as of 2007-12-30
294             NodeList nl = ((Element) selectionNode).getElementsByTagName("OPTION");
295 
296             _options = new Option[nl.getLength()];
297             for (int i = 0; i < _options.length; i++) {
298                 final String displayedText = getValue(nl.item(i).getFirstChild()).trim();
299                 _options[i] = new Option(displayedText, getOptionValue(nl.item(i), displayedText),
300                         nl.item(i).getAttributes().getNamedItem("selected") != null);
301                 _options[i].setIndex(this, i);
302             }
303         }
304 
305         /**
306          * claim unique values from the given list of values
307          *
308          * @param values
309          *            - the list of values
310          *
311          * @return
312          */
313         boolean claimUniqueValues(List values) {
314             return claimUniqueValues(values, _options);
315         }
316 
317         protected abstract boolean claimUniqueValues(List values, Option[] options);
318 
319         /**
320          * report if there are no matches be aware of [ 1100437 ] Patch for ClassCastException in FormControl TODO
321          * implement patch if test get's available
322          *
323          * @param values
324          */
325         final protected void reportNoMatches(List values) {
326             if (!_listBox) {
327                 throw new IllegalParameterValueException(getName(), values, getOptionValues());
328             }
329         }
330 
331         String[] getSelectedValues() {
332             ArrayList list = new ArrayList();
333             for (Option _option : _options) {
334                 _option.addValueIfSelected(list);
335             }
336             if (!_listBox && list.isEmpty() && _options.length > 0) {
337                 list.add(_options[0].getValue());
338             }
339             return (String[]) list.toArray(new String[list.size()]);
340         }
341 
342         void reset() {
343             for (Option _option : _options) {
344                 _option.reset();
345             }
346         }
347 
348         String[] getDisplayedText() {
349             String[] displayedText = new String[_options.length];
350             for (int i = 0; i < displayedText.length; i++) {
351                 displayedText[i] = _options[i].getText();
352             }
353             return displayedText;
354         }
355 
356         String[] getValues() {
357             String[] values = new String[_options.length];
358             for (int i = 0; i < values.length; i++) {
359                 values[i] = _options[i].getValue();
360             }
361             return values;
362         }
363 
364         /**
365          * Selects the matching item and deselects the others.
366          **/
367         void setSelectedIndex(int index) {
368             for (int i = 0; i < _options.length; i++) {
369                 _options[i]._selected = i == index;
370             }
371         }
372 
373         /**
374          * Returns the index of the first item selected, or -1 if none is selected.
375          */
376         int getFirstSelectedIndex() {
377             for (int i = 0; i < _options.length; i++) {
378                 if (_options[i].isSelected()) {
379                     return i;
380                 }
381             }
382             return noOptionSelectedIndex();
383         }
384 
385         protected abstract int noOptionSelectedIndex();
386 
387         @Override
388         public int getLength() {
389             return _options.length;
390         }
391 
392         /**
393          * Modified by gklopp - 12/19/2005 [ 1396835 ] Javascript : length of a select element cannot be increased Bug
394          * corrected : The length can be greater than the original length
395          */
396         @Override
397         public void setLength(int length) {
398             if (length < 0) {
399                 return;
400             }
401             Option[] newArray = new Option[length];
402             System.arraycopy(_options, 0, newArray, 0, Math.min(length, _options.length));
403             for (int i = _options.length; i < length; i++) {
404                 newArray[i] = new Option();
405             }
406             _options = newArray;
407         }
408 
409         @Override
410         public void put(int i, SelectionOption option) {
411             if (i < 0) {
412                 return;
413             }
414 
415             if (option == null) {
416                 if (i >= _options.length) {
417                     return;
418                 }
419                 deleteOptionsEntry(i);
420             } else {
421                 if (i >= _options.length) {
422                     i = _options.length;
423                     expandOptionsArray();
424                 }
425                 _options[i] = (Option) option;
426                 _options[i].setIndex(this, i);
427                 if (option.isSelected()) {
428                     ensureUniqueOption(_options, i);
429                 }
430             }
431         }
432 
433         protected abstract void ensureUniqueOption(Option[] options, int i);
434 
435         private void deleteOptionsEntry(int i) {
436             Option[] newArray = new Option[_options.length - 1];
437             System.arraycopy(_options, 0, newArray, 0, i);
438             System.arraycopy(_options, i + 1, newArray, i, newArray.length - i);
439             _options = newArray;
440         }
441 
442         private void expandOptionsArray() {
443             Option[] newArray = new Option[_options.length + 1];
444             System.arraycopy(_options, 0, newArray, 0, _options.length);
445             _options = newArray;
446         }
447 
448         /**
449          * get the Object at the given index check that the index is not out of bounds
450          *
451          * @param index
452          *            - the index of the object to get
453          *
454          * @throws RuntimeException
455          *             if index is out of bounds
456          *
457          * @since [ 1124057 ] Out of Bounds Exception should be avoided
458          */
459         @Override
460         public Object get(int index) {
461             // if the index is out of bounds
462             if (index < 0 || index >= _options.length) {
463                 // create a user friendly error message
464                 StringBuilder msg = new StringBuilder("invalid index ").append(index).append(" for Options ");
465                 // by listing all possible options
466                 for (int i = 0; i < _options.length; i++) {
467                     msg.append(_options[i]._text);
468                     if (i < _options.length - 1) {
469                         msg.append(",");
470                     }
471                 } // for
472                 // now throw a RunTimeException that would
473                 // have happened anyways with a less friendly message
474                 throw new RuntimeException(msg.toString());
475             } // if
476             return _options[index];
477         } // get
478 
479         /** Invoked when an option is set true. **/
480         void optionSet(int i) {
481             ensureUniqueOption(_options, i);
482         }
483 
484         private String getOptionValue(Node optionNode, String displayedText) {
485             NamedNodeMap nnm = optionNode.getAttributes();
486             if (nnm.getNamedItem("value") != null) {
487                 return getValue(nnm.getNamedItem("value"));
488             }
489             return displayedText;
490         }
491 
492         private String getValue(Node node) {
493             return node == null ? "" : emptyIfNull(node.getNodeValue());
494         }
495     }

497     class SingleSelectOptions extends Options {
498 
499         public SingleSelectOptions(Node selectionNode) {
500             super(selectionNode);
501         }
502 
503         @Override
504         protected void ensureUniqueOption(Option[] options, int i) {
505             for (int j = 0; j < options.length; j++) {
506                 options[j]._selected = i == j;
507             }
508         }
509 
510         @Override
511         protected int noOptionSelectedIndex() {
512             return 0;
513         }
514 
515         /**
516          * claim the values be aware of [ 1100437 ] Patch for ClassCastException in FormControl TODO implement patch if
517          * test get's available - the (String) cast might fail
518          */
519         @Override
520         protected boolean claimUniqueValues(List values, Option[] options) {
521             boolean changed = false;
522             for (int i = 0; i < values.size(); i++) {
523                 String value = (String) values.get(i);
524                 for (int j = 0; j < options.length; j++) {
525                     boolean selected = value.equals(options[j].getValue());
526                     if (selected != options[j].isSelected()) {
527                         changed = true;
528                     }
529                     options[j].setSelected(selected);
530                     if (selected) {
531                         values.remove(value);
532                         for (++j; j < options.length; j++) {
533                             options[j].setSelected(false);
534                         }
535                         return changed;
536                     }
537                 }
538             }
539             reportNoMatches(values);
540             return changed;
541         }
542     }

544     class MultiSelectOptions extends Options {
545 
546         public MultiSelectOptions(Node selectionNode) {
547             super(selectionNode);
548         }
549 
550         @Override
551         protected void ensureUniqueOption(Option[] options, int i) {
552         }
553 
554         @Override
555         protected int noOptionSelectedIndex() {
556             return -1;
557         }
558 
559         @Override
560         protected boolean claimUniqueValues(List values, Option[] options) {
561             boolean changed = false;
562             for (Option option : options) {
563                 final boolean newValue = values.contains(option.getValue());
564                 if (newValue != option.isSelected()) {
565                     changed = true;
566                 }
567                 option.setSelected(newValue);
568                 if (newValue) {
569                     values.remove(option.getValue());
570                 }
571             }
572             return changed;
573         }
