8   package com.meterware.httpunit.cookies;
9   
10  /**
11   * An interface for classes which can listen for cookies being rejected and the reason.
12   */
13  public interface CookieListener {
14  
15      /** Indicates that the cookie was accepted. **/
16      int ACCEPTED = 0;
17  
18      /** Indicates that the domain attribute has only one dot. **/
19      int DOMAIN_ONE_DOT = 2;
20  
21      /** Indicates that the domain attribute is not a suffix of the source domain issuing the cookie. **/
22      int DOMAIN_NOT_SOURCE_SUFFIX = 3;
23  
24      /** Indicates that the source domain has an extra dot beyond those defined in the domain attribute. **/
25      int DOMAIN_TOO_MANY_LEVELS = 4;
26  
27      /** Indicates that the source path does not begin with the path attribute. **/
28      int PATH_NOT_PREFIX = 5;
29  
30      /**
31       * Invoked when a cookie is rejected by HttpUnit.
32       *
33       * @param cookieName
34       *            the cookie name
35       * @param reason
36       *            the reason
37       * @param attribute
38       *            the attribute
39       */
40      void cookieRejected(String cookieName, int reason, String attribute);
41  
42  }
