20  package com.meterware.httpunit.dom;
21  
22  import org.w3c.dom.CharacterData;
23  import org.w3c.dom.DOMException;
24  
25  /**
26   * @author <a href="mailto:russgold@httpunit.org">Russell Gold</a>
27   **/
28  abstract public class CharacterDataImpl extends NodeImpl implements CharacterData {
29  
30      private static final long serialVersionUID = 1L;
31      private String _data;
32  
33      protected void initialize(DocumentImpl ownerDocument, String data) {
34          super.initialize(ownerDocument);
35          _data = data;
36      }
37  
38      @Override
39      public String getData() throws DOMException {
40          return _data;
41      }
42  
43      @Override
44      public void setData(String data) throws DOMException {
45          if (data == null) {
46              data = "";
47          }
48          _data = data;
49      }
50  
51      @Override
52      public int getLength() {
53          return _data.length();
54      }
55  
56      @Override
57      public String substringData(int offset, int count) throws DOMException {
58          return null;
59      }
60  
61      @Override
62      public void appendData(String arg) throws DOMException {
63      }
64  
65      @Override
66      public void insertData(int offset, String arg) throws DOMException {
67      }
68  
69      @Override
70      public void deleteData(int offset, int count) throws DOMException {
71      }
72  
73      @Override
74      public void replaceData(int offset, int count, String arg) throws DOMException {
75      }
76  
77  }
