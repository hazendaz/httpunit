73      @Override
74      public Object get(String propertyName, Scriptable scriptable) {
75          if (propertyName.equals("document")) {
76              return this;
77          }
78  
79          Object result = super.get(propertyName, scriptable);
80          if (result != NOT_FOUND) {
81              return result;
82          }
83  
84          Element element = getElementById(propertyName);
85          if (element != null) {
86              return element;
87          }
88  
89          NodeList elements = getElementsByName(propertyName);
90          if (elements.getLength() >= 1) {
91              return elements.item(0);
92          }
93  
94          return ScriptingSupport.getNamedProperty(this, getJavaPropertyName(propertyName), scriptable);
95      }
96  
97      @Override
98      public void put(String propertyName, Scriptable initialObject, Object value) {
99          ScriptingSupport.setNamedProperty(this, getJavaPropertyName(propertyName), value);
100     }

144     /**
145      * Gets the title element.
146      *
147      * @return the title element
148      */
149     private HTMLTitleElement getTitleElement() {
150         HTMLTitleElement result = null;
151         NodeList titleNodes = getElementsByTagName("title");
152         for (int i = 0; i < titleNodes.getLength(); i++) {
153             Node node = titleNodes.item(i);
154             if (node instanceof HTMLTitleElement) {
155                 result = (HTMLTitleElement) node;
156             }
157         }
158         return result;
159     }

161     /**
162      * Gets the head element.
163      *
164      * @return the head element
165      */
166     private HTMLHeadElement getHeadElement() {
167         NodeList headNodes = getElementsByTagName("head");
168         for (int i = 0; i < headNodes.getLength(); i++) {
169             Node node = headNodes.item(i);
170             if (node instanceof HTMLHeadElement) {
171                 return (HTMLHeadElement) node;
172             }
173         }
174 
175         HTMLHeadElement head = (HTMLHeadElement) createElement("head");
176         getHtmlElement().appendChild(head);
177         return head;
178     }

344      * Gets the elements by name.
345      *
346      * @param elementName
347      *            the element name
348      *
349      * @return the elements by name
350      */
351     @Override
352     public NodeList getElementsByName(String elementName) {
353         ArrayList<HTMLElementImpl> elements = new ArrayList<HTMLElementImpl>();
354         for (Iterator<?> each = preOrderIterator(); each.hasNext();) {
355             Node node = (Node) each.next();
356             if (!(node instanceof HTMLElementImpl)) {
357                 continue;
358             }
359             HTMLElementImpl element = (HTMLElementImpl) node;
360             if (elementName.equals(element.getAttributeWithNoDefault("name"))) {
361                 elements.add(element);
362             }
363         }
364         return new NodeListImpl(elements);
365     }

388         HTMLDocumentImpl copy = new HTMLDocumentImpl();

495     /**
496      * Gets the base url.
497      *
498      * @return the base url
499      */
500     URL getBaseUrl() {
501         NodeList list = getElementsByTagName("base");
502         if (list.getLength() == 0) {
503             return getWindow().getUrl();
504         }
505 
506         HTMLBaseElement base = (HTMLBaseElement) list.item(0);
507         try {
508             return new URL(getWindow().getUrl(), base.getHref());
509         } catch (MalformedURLException e) {
510             return getWindow().getUrl();
511         }
