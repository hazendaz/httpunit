57      @Override
58      public Object get(String propertyName, Scriptable scriptable) {
59          if (propertyName.equals("document")) {
60              return this;
61          }
62  
63          Object result = super.get(propertyName, scriptable);
64          if (result != NOT_FOUND) {
65              return result;
66          }
67  
68          Element element = getElementById(propertyName);
69          if (element != null) {
70              return element;
71          }
72  
73          NodeList elements = getElementsByName(propertyName);
74          if (elements.getLength() >= 1) {
75              return elements.item(0);
76          }
77  
78          return ScriptingSupport.getNamedProperty(this, getJavaPropertyName(propertyName), scriptable);
79      }
80  
81      @Override
82      public void put(String propertyName, Scriptable initialObject, Object value) {
83          ScriptingSupport.setNamedProperty(this, getJavaPropertyName(propertyName), value);
84      }

117     @Override

119         HTMLTitleElement result = getTitleElement();
120         return result == null ? "" : result.getText();
121     }

123     private HTMLTitleElement getTitleElement() {
124         HTMLTitleElement result = null;
125         NodeList titleNodes = getElementsByTagName("title");
126         for (int i = 0; i < titleNodes.getLength(); i++) {
127             Node node = titleNodes.item(i);
128             if (node instanceof HTMLTitleElement) {
129                 result = (HTMLTitleElement) node;
130             }
131         }
132         return result;
133     }
134 
135     private HTMLHeadElement getHeadElement() {
136         NodeList headNodes = getElementsByTagName("head");
137         for (int i = 0; i < headNodes.getLength(); i++) {
138             Node node = headNodes.item(i);
139             if (node instanceof HTMLHeadElement) {
140                 return (HTMLHeadElement) node;
141             }
142         }
143 
144         HTMLHeadElement head = (HTMLHeadElement) createElement("head");
145         getHtmlElement().appendChild(head);
146         return head;
147     }
148 
149     private HTMLHtmlElement getHtmlElement() {
150         NodeList htmlNodes = getElementsByTagName("html");
151         for (int i = 0; i < htmlNodes.getLength(); i++) {
152             Node node = htmlNodes.item(i);
153             if (node instanceof HTMLHtmlElement) {
154                 return (HTMLHtmlElement) node;
155             }
156         }
157 
158         HTMLHtmlElement html = (HTMLHtmlElement) createElement("html");
159         appendChild(html);
160         return html;
161     }

163     @Override

165         HTMLTitleElement titleElement = getTitleElement();
166         if (titleElement != null) {

168         } else {
169             titleElement = (HTMLTitleElement) createElement("title");
170             titleElement.setText(title);
171             getHeadElement().appendChild(titleElement);
172         }

241     @Override
242     public NodeList getElementsByName(String elementName) {
243         ArrayList<HTMLElementImpl> elements = new ArrayList<HTMLElementImpl>();
244         for (Iterator<?> each = preOrderIterator(); each.hasNext();) {
245             Node node = (Node) each.next();
246             if (!(node instanceof HTMLElementImpl)) {
247                 continue;
248             }
249             HTMLElementImpl element = (HTMLElementImpl) node;
250             if (elementName.equals(element.getAttributeWithNoDefault("name"))) {
251                 elements.add(element);
252             }
253         }
254         return new NodeListImpl(elements);
255     }

276     @Override
277     public Node cloneNode(boolean deep) {
278         HTMLDocumentImpl copy = new HTMLDocumentImpl();
279 
280         if (deep) {
281             copy.importChildren(this, copy);
282             copy._documentElement = copy.getHtmlElement();
283         }
284         return copy;
285     }

287     private static HTMLElementImpl getExemplar(String tagName) {
288         HTMLElementImpl impl = (HTMLElementImpl) _exemplars.get(tagName.toLowerCase());
289         if (impl == null) {
290             impl = new HTMLElementImpl();
291         }
292         return impl;
293     }

335     public DomWindow getWindow() {
336         // if there is now window yet
337         if (_window == null) {

356     URL getBaseUrl() {
357         NodeList list = getElementsByTagName("base");
358         if (list.getLength() == 0) {
359             return getWindow().getUrl();
360         }
361 
362         HTMLBaseElement base = (HTMLBaseElement) list.item(0);
363         try {
364             return new URL(getWindow().getUrl(), base.getHref());
365         } catch (MalformedURLException e) {
366             return getWindow().getUrl();
367         }
