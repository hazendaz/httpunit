20  package com.meterware.httpunit.dom;
21  
22  import org.w3c.dom.Attr;
23  import org.w3c.dom.NodeList;
24  import org.w3c.dom.html.HTMLElement;
25  
26  /**
27   * @author <a href="mailto:russgold@httpunit.org">Russell Gold</a>
28   */
29  public class HTMLElementImpl extends ElementImpl implements HTMLElement {
30  
31      private static final long serialVersionUID = 1L;
32      public final static String UNSPECIFIED_ATTRIBUTE = null;
33  
34      ElementImpl create() {
35          return new HTMLElementImpl();
36      }
37  
38      public void click() {
39          doClickAction();
40      }
41  
42      public void doClickAction() {
43      }
44  
45      @Override
46      public String getId() {
47          return getAttributeWithNoDefault("id");
48      }
49  
50      @Override
51      public void setId(String id) {
52          setAttribute("id", id);
53      }
54  
55      @Override
56      public String getTitle() {
57          return getAttributeWithNoDefault("title");
58      }
59  
60      @Override
61      public void setTitle(String title) {
62          setAttribute("title", title);
63      }
64  
65      @Override
66      public String getLang() {
67          return getAttributeWithNoDefault("lang");
68      }
69  
70      @Override
71      public void setLang(String lang) {
72          setAttribute("lang", lang);
73      }
74  
75      @Override
76      public String getDir() {
77          return getAttributeWithNoDefault("dir");
78      }
79  
80      @Override
81      public void setDir(String dir) {
82          setAttribute("dir", dir);
83      }
84  
85      @Override
86      public String getClassName() {
87          return getAttributeWithNoDefault("class");
88      }
89  
90      @Override
91      public void setClassName(String className) {
92          setAttribute("class", className);
93      }
94  
95      @Override
96      public NodeList getElementsByTagName(String name) {
97          return super.getElementsByTagName(((HTMLDocumentImpl) getOwnerDocument()).toNodeCase(name));
98      }
99  
100     // ---------------------------------------------- protected methods
101     // -----------------------------------------------------
102 
103     final protected String getAttributeWithDefault(String attributeName, String defaultValue) {
104         if (hasAttribute(attributeName)) {
105             return getAttribute(attributeName);
106         }
107         return defaultValue;
108     }
109 
110     final protected String getAttributeWithNoDefault(String attributeName) {
111         if (hasAttribute(attributeName)) {
112             return getAttribute(attributeName);
113         }
114         return UNSPECIFIED_ATTRIBUTE;
115     }
116 
117     protected boolean getBooleanAttribute(String name) {
118         Attr attr = getAttributeNode(name);
119         return attr != null && !attr.getValue().equalsIgnoreCase("false");
120     }
121 
122     protected int getIntegerAttribute(String name) {
123         String value = getAttribute(name);
124         return value.length() == 0 ? 0 : Integer.parseInt(value);
125     }
126 
127     protected int getIntegerAttribute(String name, int defaultValue) {
128         String value = getAttribute(name);
129         return value.length() == 0 ? defaultValue : Integer.parseInt(value);
130     }
131 
132     protected void setAttribute(String name, boolean disabled) {
133         setAttribute(name, disabled ? "true" : "false");
134     }
135 
136     protected void setAttribute(String name, int value) {
137         setAttribute(name, Integer.toString(value));
138     }
139 
140     HTMLDocumentImpl getHtmlDocument() {
141         return (HTMLDocumentImpl) getOwnerDocument();
142     }
143 
144 }
