53      @Override
54      public Object get(String propertyName, Scriptable scriptable) {
55          HTMLCollection elements = getElements();
56          for (int i = 0; i < elements.getLength(); i++) {
57              Node node = elements.item(i);
58              NamedNodeMap attributes = node.getAttributes();
59              AttrImpl nameAttribute = (AttrImpl) attributes.getNamedItem("name");
60              if (nameAttribute != null && propertyName.equals(nameAttribute.getValue())) {
61                  return node;
62              }
63              AttrImpl idAttribute = (AttrImpl) attributes.getNamedItem("id");
64              if (idAttribute != null && propertyName.equals(idAttribute.getValue())) {
65                  return node;
66              }
67          }
68          return super.get(propertyName, scriptable);
69      }

151     @Override
152     public HTMLCollection getElements() {
153         ArrayList elements = new ArrayList();
154         String[] names = { "INPUT", "TEXTAREA", "BUTTON", "SELECT" };
155         for (Iterator each = preOrderIteratorAfterNode(); each.hasNext();) {
156             Node node = (Node) each.next();
157             if (node instanceof HTMLFormElement) {
158                 break;
159             }
160 
161             if (node.getNodeType() != ELEMENT_NODE) {
162                 continue;
163             }
164             String tagName = ((Element) node).getTagName();
165             for (String name : names) {
166                 if (tagName.equalsIgnoreCase(name)) {
167                     elements.add(node);
168                 }
169             }
170         }
171         return HTMLCollectionImpl.createHTMLCollectionImpl(new NodeListImpl(elements));
172     }

179     @Override
180     public void reset() {
181         HTMLCollection elements = getElements();
182         for (int i = 0; i < elements.getLength(); i++) {
183             Node node = elements.item(i);
184             if (node instanceof HTMLControl) {
185                 ((HTMLControl) node).reset();
186             }
187         }
188     }

190     @Override
191     public void submit() {
192         doSubmitAction();
193     }
194 
195     /**
196      * Handles the actual form submission - does not handle the "submit" event.
197      */
198     void doSubmitAction() {
199         try {
200             if ("get".equalsIgnoreCase(getMethod())) {
201                 getDomWindow().submitRequest(this, getMethod(), getEffectiveUrl(), getTarget(), new byte[0]);
202             } else if ("post".equalsIgnoreCase(getMethod())) {
203                 getDomWindow().submitRequest(this, getMethod(), getAction(), getTarget(), new byte[0]);
204             }
205         } catch (Exception e) {
206             throw new RuntimeException("Error submitting form: " + e);
207         } finally {
208             silenceSubmitButtons();
209         }
210     }
211 
212     private void silenceSubmitButtons() {
213         HTMLCollection controls = getElements();
214         for (int i = 0; i < controls.getLength(); i++) {
215             ((HTMLControl) controls.item(i)).silenceSubmitButton();
216         }
217     }
218 
219     private String getEffectiveUrl() throws IOException {
220         StringBuilder spec = new StringBuilder(getAction());
221         if ("get".equalsIgnoreCase(getMethod())) {
222             URLEncodedString parameters = new URLEncodedString();
223             HTMLCollection controls = getElements();
224             for (int i = 0; i < controls.getLength(); i++) {
225                 ((HTMLControl) controls.item(i)).addValues(parameters, "us-ascii");
226             }
227             if (spec.indexOf("?") >= 0 && !spec.toString().endsWith("?")) {
228                 spec.append('&');
229             } else {
230                 spec.append('?');
231             }
232             spec.append(parameters.getString());
233         }
234         return new URL(getDomWindow().getUrl(), spec.toString()).toExternalForm();
235     }
236 
237     private DomWindow getDomWindow() {
238         return ((HTMLDocumentImpl) getOwnerDocument()).getWindow();
239     }
