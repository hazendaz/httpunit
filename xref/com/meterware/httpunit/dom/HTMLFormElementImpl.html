54      @Override
55      public Object get(String propertyName, Scriptable scriptable) {
56          HTMLCollection elements = getElements();
57          for (int i = 0; i < elements.getLength(); i++) {
58              Node node = elements.item(i);
59              NamedNodeMap attributes = node.getAttributes();
60              AttrImpl nameAttribute = (AttrImpl) attributes.getNamedItem("name");
61              if (nameAttribute != null && propertyName.equals(nameAttribute.getValue())) {
62                  return node;
63              }
64              AttrImpl idAttribute = (AttrImpl) attributes.getNamedItem("id");
65              if (idAttribute != null && propertyName.equals(idAttribute.getValue())) {
66                  return node;
67              }
68          }
69          return super.get(propertyName, scriptable);
70      }

209     /**
210      * Gets the elements.
211      *
212      * @return the elements
213      */
214     @Override
215     public HTMLCollection getElements() {
216         ArrayList elements = new ArrayList<>();
217         String[] names = { "INPUT", "TEXTAREA", "BUTTON", "SELECT" };
218         for (Iterator each = preOrderIteratorAfterNode(); each.hasNext();) {
219             Node node = (Node) each.next();
220             if (node instanceof HTMLFormElement) {
221                 break;
222             }
223 
224             if (node.getNodeType() != ELEMENT_NODE) {
225                 continue;
226             }
227             String tagName = ((Element) node).getTagName();
228             for (String name : names) {
229                 if (tagName.equalsIgnoreCase(name)) {
230                     elements.add(node);
231                 }
232             }
233         }
234         return HTMLCollectionImpl.createHTMLCollectionImpl(new NodeListImpl(elements));
235     }

269     /**
270      * Handles the actual form submission - does not handle the "submit" event.
271      */
272     void doSubmitAction() {
273         try {
274             if ("get".equalsIgnoreCase(getMethod())) {
275                 getDomWindow().submitRequest(this, getMethod(), getEffectiveUrl(), getTarget(), new byte[0]);
276             } else if ("post".equalsIgnoreCase(getMethod())) {
277                 getDomWindow().submitRequest(this, getMethod(), getAction(), getTarget(), new byte[0]);
278             }
279         } catch (Exception e) {
280             throw new RuntimeException("Error submitting form: " + e);
281         } finally {
282             silenceSubmitButtons();
283         }
284     }

286     /**
287      * Silence submit buttons.
288      */
289     private void silenceSubmitButtons() {
290         HTMLCollection controls = getElements();
291         for (int i = 0; i < controls.getLength(); i++) {
292             ((HTMLControl) controls.item(i)).silenceSubmitButton();
293         }
294     }

296     /**
297      * Gets the effective url.
298      *
299      * @return the effective url
300      *
301      * @throws IOException
302      *             Signals that an I/O exception has occurred.
303      */
304     private String getEffectiveUrl() throws IOException {
305         StringBuilder spec = new StringBuilder(getAction());
306         if ("get".equalsIgnoreCase(getMethod())) {
307             URLEncodedString parameters = new URLEncodedString();
308             HTMLCollection controls = getElements();
309             for (int i = 0; i < controls.getLength(); i++) {
310                 ((HTMLControl) controls.item(i)).addValues(parameters, "us-ascii");
311             }
312             if (spec.indexOf("?") >= 0 && !spec.toString().endsWith("?")) {
313                 spec.append('&');
314             } else {
315                 spec.append('?');
316             }
317             spec.append(parameters.getString());
318         }
319         return new URL(getDomWindow().getUrl(), spec.toString()).toExternalForm();
320     }
