20  package com.meterware.httpunit.dom;
21  
22  import org.w3c.dom.NodeList;
23  import org.w3c.dom.Text;
24  import org.w3c.dom.html.HTMLTitleElement;
25  
26  /**
27   * @author <a href="mailto:russgold@httpunit.org">Russell Gold</a>
28   **/
29  public class HTMLTitleElementImpl extends HTMLElementImpl implements HTMLTitleElement {
30  
31      private static final long serialVersionUID = 1L;
32  
33      @Override
34      ElementImpl create() {
35          return new HTMLTitleElementImpl();
36      }
37  
38      @Override
39      public String getText() {
40          Text contentNode = getContentNode();
41          return contentNode == null ? "" : contentNode.getData();
42      }
43  
44      private Text getContentNode() {
45          NodeList childNodes = getChildNodes();
46          for (int i = 0; i < childNodes.getLength(); i++) {
47              if (childNodes.item(i).getNodeType() == TEXT_NODE) {
48                  return (Text) childNodes.item(i);
49              }
50          }
51          return null;
52      }
53  
54      @Override
55      public void setText(String text) {
56          Text newChild = getOwnerDocument().createTextNode(text);
57          Text oldChild = getContentNode();
58          if (oldChild == null) {
59              appendChild(newChild);
60          } else {
61              replaceChild(newChild, oldChild);
62          }
63      }
64  }
