20  package com.meterware.httpunit.dom;
21  
22  import org.w3c.dom.DOMException;
23  import org.w3c.dom.Node;
24  import org.w3c.dom.ProcessingInstruction;
25  
26  /**
27   * @author <a href="mailto:russgold@httpunit.org">Russell Gold</a>
28   **/
29  public class ProcessingInstructionImpl extends NodeImpl implements ProcessingInstruction {
30  
31      private static final long serialVersionUID = 1L;
32      private String _target;
33      private String _data;
34  
35      public static ProcessingInstruction createProcessingImpl(DocumentImpl ownerDocument, String target, String data) {
36          ProcessingInstructionImpl instruction = new ProcessingInstructionImpl();
37          instruction.initialize(ownerDocument, target, data);
38          return instruction;
39      }
40  
41      private void initialize(DocumentImpl ownerDocument, String target, String data) {
42          super.initialize(ownerDocument);
43          _target = target;
44          _data = data;
45      }
46  
47      public static Node importNode(DocumentImpl document, ProcessingInstruction processingInstruction) {
48          return createProcessingImpl(document, processingInstruction.getTarget(), processingInstruction.getData());
49      }
50  
51      @Override
52      public String getNodeName() {
53          return _target;
54      }
55  
56      @Override
57      public String getNodeValue() throws DOMException {
58          return _data;
59      }
60  
61      @Override
62      public void setNodeValue(String string) throws DOMException {
63          setData(string);
64      }
65  
66      @Override
67      public short getNodeType() {
68          return PROCESSING_INSTRUCTION_NODE;
69      }
70  
71      @Override
72      public String getTarget() {
73          return _target;
74      }
75  
76      @Override
77      public String getData() {
78          return _data;
79      }
80  
81      @Override
82      public void setData(String string) throws DOMException {
83          _data = string;
84      }
85  }
