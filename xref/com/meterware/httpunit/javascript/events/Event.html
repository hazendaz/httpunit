20  package com.meterware.httpunit.javascript.events;
21  
22  import org.mozilla.javascript.ScriptableObject;
23  
24  /**
25   * @author <a href="mailto:rafal@caltha.pl">Rafal Krzewski</a>
26   *
27   * @version $Id$
28   */
29  public abstract class Event extends ScriptableObject {
30      // -- Scriptable interface ------------------------------------------------------------------
31  
32      private static final long serialVersionUID = 1L;
33  
34      /** JavaScript class name. */
35      private static final String JS_CLASS_NAME = "Event";
36  
37      /**
38       * {@inheritDoc}
39       */
40      @Override
41      public String getClassName() {
42          return JS_CLASS_NAME;
43      }
44  
45      // -- DOM2 Events specification -------------------------------------------------------------
46  
47      /**
48       * The current event phase is the capturing phase.
49       */
50      public static final short CAPTURING_PHASE = 1;
51  
52      /**
53       * The event is currently being evaluated at the target EventTarget.
54       */
55      public static final short AT_TARGET = 2;
56  
57      /**
58       * The current event phase is the bubbling phase.
59       */
60      public static final short BUBBLING_PHASE = 3;
61  
62      /**
63       * The name of the event (case-insensitive). The name must be an XML name.
64       */
65      private String type;
66  
67      /**
68       * Used to indicate whether or not an event is a bubbling event. If the event can bubble the value is true, else the
69       * value is false.
70       */
71      private boolean bubbles;
72  
73      /**
74       * Used to indicate whether or not an event can have its default action prevented. If the default action can be
75       * prevented the value is true, else the value is false.
76       */
77      private boolean cancelable;
78  
79      /**
80       * Used to specify the time (in milliseconds relative to the epoch) at which the event was created. Due to the fact
81       * that some systems may not provide this information the value of timeStamp may be not available for all events.
82       * When not available, a value of 0 will be returned. Examples of epoch time are the time of the system start or
83       * 0:0:0 UTC 1st January 1970.
84       */
85      private long timeStamp;
86  
87      /**
88       * The name of the event (case-insensitive). The name must be an XML name.
89       */
90      public String jsGet_type() {
91          return type;
92      }
93  
94      /**
95       * Used to indicate whether or not an event is a bubbling event. If the event can bubble the value is true, else the
96       * value is false.
97       */
98      public boolean jsGet_bubbles() {
99          return bubbles;
100     }
101 
102     /**
103      * Used to specify the time (in milliseconds relative to the epoch) at which the event was created. Due to the fact
104      * that some systems may not provide this information the value of timeStamp may be not available for all events.
105      * When not available, a value of 0 will be returned. Examples of epoch time are the time of the system start or
106      * 0:0:0 UTC 1st January 1970.
107      */
108     public boolean jsGet_cancelable() {
109         return cancelable;
110     }
111 
112     /**
113      * Used to specify the time (in milliseconds relative to the epoch) at which the event was created. Due to the fact
114      * that some systems may not provide this information the value of timeStamp may be not available for all events.
115      * When not available, a value of 0 will be returned. Examples of epoch time are the time of the system start or
116      * 0:0:0 UTC 1st January 1970.
117      */
118     public long jsGet_timeStamp() {
119         return timeStamp;
120     }
121 
122     /**
123      * Used to indicate the EventTarget to which the event was originally dispatched.
124      */
125     public abstract EventTarget jsGet_target();
126 
127     /**
128      * Used to indicate the EventTarget whose EventListeners are currently being processed. This is particularly useful
129      * during capturing and bubbling.
130      */
131     public abstract EventTarget jsGet_currentTarget();
132 
133     /**
134      * Used to indicate which phase of event flow is currently being evaluated.
135      */
136     public abstract short jsGet_eventPhase();
137 
138     /**
139      * The stopPropagation method is used prevent further propagation of an event during event flow. If this method is
140      * called by any EventListener the event will cease propagating through the tree. The event will complete dispatch
141      * to all listeners on the current EventTarget before event flow stops. This method may be used during any stage of
142      * event flow.
143      */
144     public abstract void jsFunction_stopPropagation();
145 
146     /**
147      * If an event is cancelable, the preventDefault method is used to signify that the event is to be canceled, meaning
148      * any default action normally taken by the implementation as a result of the event will not occur. If, during any
149      * stage of event flow, the preventDefault method is called the event is canceled. Any default action associated
150      * with the event will not occur. Calling this method for a non-cancelable event has no effect. Once preventDefault
151      * has been called it will remain in effect throughout the remainder of the event's propagation. This method may be
152      * used during any stage of event flow.
153      */
154     public abstract void jsFunction_preventDefault();
155 
156     /**
157      * The initEvent method is used to initialize the value of an Event created through the DocumentEvent interface.
158      * This method may only be called before the Event has been dispatched via the dispatchEvent method, though it may
159      * be called multiple times during that phase if necessary. If called multiple times the final invocation takes
160      * precedence. If called from a subclass of Event interface only the values specified in the initEvent method are
161      * modified, all other attributes are left unchanged.
162      *
163      * @param eventTypeArg
164      *            Specifies the event type. This type may be any event type currently defined in this specification or a
165      *            new event type.. The string must be an XML name. Any new event type must not begin with any upper,
166      *            lower, or mixed case version of the string "DOM". This prefix is reserved for future DOM event sets.
167      *            It is also strongly recommended that third parties adding their own events use their own prefix to
168      *            avoid confusion and lessen the probability of conflicts with other new events.
169      * @param canBubbleArg
170      *            Specifies whether or not the event can bubble.
171      * @param cancelableArg
172      *            Specifies whether or not the event's default action can be prevented.
173      */
174     public void jsFunction_initEvent(String eventTypeArg, boolean canBubbleArg, boolean cancelableArg) {
175         type = eventTypeArg;
176         bubbles = canBubbleArg;
177         cancelable = cancelableArg;
178     }
179 }
