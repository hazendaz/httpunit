8   package com.meterware.httpunit.javascript.events;
9   
10  import org.mozilla.javascript.Scriptable;
11  
12  /**
13   * The EventListener interface is the primary method for handling events. Users implement the EventListener interface
14   * and register their listener on an EventTarget using the AddEventListener method. The users should also remove their
15   * EventListener from its EventTarget after they have completed using the listener. When a Node is copied using the
16   * cloneNode method the EventListeners attached to the source Node are not attached to the copied Node. If the user
17   * wishes the same EventListeners to be added to the newly created copy the user must add them manually.
18   */
19  public interface EventListener extends Scriptable {
20      /**
21       * This method is called whenever an event occurs of the type for which the EventListener interface was registered.
22       *
23       * @param evt
24       *            The Event contains contextual information about the event. It also contains the stopPropagation and
25       *            preventDefault methods which are used in determining the event's flow and default action.
26       */
27      void jsFunction_handleEvent(Scriptable evt);
28  }
