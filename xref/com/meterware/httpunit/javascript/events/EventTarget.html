20  package com.meterware.httpunit.javascript.events;
21  
22  import org.mozilla.javascript.Scriptable;
23  
24  /**
25   * The EventTarget interface is implemented by all Nodes in an implementation which supports the DOM Event Model.
26   * Therefore, this interface can be obtained by using binding-specific casting methods on an instance of the Node
27   * interface. The interface allows registration and removal of EventListeners on an EventTarget and dispatch of events
28   * to that EventTarget.
29   *
30   * @author W3C
31   *
32   * @version $Id$
33   */
34  public interface EventTarget extends Scriptable {
35      /**
36       * This method allows the registration of event listeners on the event target. If an EventListener is added to an
37       * EventTarget while it is processing an event, it will not be triggered by the current actions but may be triggered
38       * during a later stage of event flow, such as the bubbling phase. If multiple identical EventListeners are
39       * registered on the same EventTarget with the same parameters the duplicate instances are discarded. They do not
40       * cause the EventListener to be called twice and since they are discarded they do not need to be removed with the
41       * removeEventListener method.
42       *
43       * @param type
44       *            The event type for which the user is registering.
45       * @param listener
46       *            The listener parameter takes an interface implemented by the user which contains the methods to be
47       *            called when the event occurs.
48       * @param useCapture
49       *            If true, useCapture indicates that the user wishes to initiate capture. After initiating capture, all
50       *            events of the specified type will be dispatched to the registered EventListener before being
51       *            dispatched to any EventTargets beneath them in the tree. Events which are bubbling upward through the
52       *            tree will not trigger an EventListener designated to use capture.
53       */
54      void jsFunction_addEventListener(String type, Scriptable listener, boolean useCapture);
55  
56      /**
57       * This method allows the removal of event listeners from the event target. If an EventListener is removed from an
58       * EventTarget while it is processing an event, it will not be triggered by the current actions. EventListeners can
59       * never be invoked after being removed. Calling removeEventListener with arguments which do not identify any
60       * currently registered EventListener on the EventTarget has no effect.
61       *
62       * @param type
63       *            Specifies the event type of the EventListener being removed.
64       * @param listener
65       *            The EventListener parameter indicates the EventListener to be removed.
66       * @param useCapture
67       *            Specifies whether the EventListener being removed was registered as a capturing listener or not. If a
68       *            listener was registered twice, one with capture and one without, each must be removed separately.
69       *            Removal of a capturing listener does not affect a non-capturing version of the same listener, and vice
70       *            versa.
71       */
72      void jsFunction_removeEventListener(String type, Scriptable listener, boolean useCapture);
73  
74      /**
75       * This method allows the dispatch of events into the implementations event model. Events dispatched in this manner
76       * will have the same capturing and bubbling behavior as events dispatched directly by the implementation. The
77       * target of the event is the EventTarget on which dispatchEvent is called.
78       *
79       * @param evt
80       *            Specifies the event type, behavior, and contextual information to be used in processing the event.
81       *
82       * @return The return value of dispatchEvent indicates whether any of the listeners which handled the event called
83       *         preventDefault. If preventDefault was called the value is false, else the value is true.
84       *
85       * @throws EventException
86       *             UNSPECIFIED_EVENT_TYPE_ERR: Raised if the Event's type was not specified by initializing the event
87       *             before dispatchEvent was called. Specification of the Event's type as null or an empty string will
88       *             also trigger this exception.
89       */
90      boolean jsFunction_dispatchEvent(Scriptable evt) throws EventException;
91  }
