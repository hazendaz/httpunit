8   package com.meterware.httpunit.javascript.events;
9   
10  import org.mozilla.javascript.Scriptable;
11  
12  /**
13   * The EventTarget interface is implemented by all Nodes in an implementation which supports the DOM Event Model.
14   * Therefore, this interface can be obtained by using binding-specific casting methods on an instance of the Node
15   * interface. The interface allows registration and removal of EventListeners on an EventTarget and dispatch of events
16   * to that EventTarget.
17   */
18  public interface EventTarget extends Scriptable {
19      /**
20       * This method allows the registration of event listeners on the event target. If an EventListener is added to an
21       * EventTarget while it is processing an event, it will not be triggered by the current actions but may be triggered
22       * during a later stage of event flow, such as the bubbling phase. If multiple identical EventListeners are
23       * registered on the same EventTarget with the same parameters the duplicate instances are discarded. They do not
24       * cause the EventListener to be called twice and since they are discarded they do not need to be removed with the
25       * removeEventListener method.
26       *
27       * @param type
28       *            The event type for which the user is registering.
29       * @param listener
30       *            The listener parameter takes an interface implemented by the user which contains the methods to be
31       *            called when the event occurs.
32       * @param useCapture
33       *            If true, useCapture indicates that the user wishes to initiate capture. After initiating capture, all
34       *            events of the specified type will be dispatched to the registered EventListener before being
35       *            dispatched to any EventTargets beneath them in the tree. Events which are bubbling upward through the
36       *            tree will not trigger an EventListener designated to use capture.
37       */
38      void jsFunction_addEventListener(String type, Scriptable listener, boolean useCapture);
39  
40      /**
41       * This method allows the removal of event listeners from the event target. If an EventListener is removed from an
42       * EventTarget while it is processing an event, it will not be triggered by the current actions. EventListeners can
43       * never be invoked after being removed. Calling removeEventListener with arguments which do not identify any
44       * currently registered EventListener on the EventTarget has no effect.
45       *
46       * @param type
47       *            Specifies the event type of the EventListener being removed.
48       * @param listener
49       *            The EventListener parameter indicates the EventListener to be removed.
50       * @param useCapture
51       *            Specifies whether the EventListener being removed was registered as a capturing listener or not. If a
52       *            listener was registered twice, one with capture and one without, each must be removed separately.
53       *            Removal of a capturing listener does not affect a non-capturing version of the same listener, and vice
54       *            versa.
55       */
56      void jsFunction_removeEventListener(String type, Scriptable listener, boolean useCapture);
57  
58      /**
59       * This method allows the dispatch of events into the implementations event model. Events dispatched in this manner
60       * will have the same capturing and bubbling behavior as events dispatched directly by the implementation. The
61       * target of the event is the EventTarget on which dispatchEvent is called.
62       *
63       * @param evt
64       *            Specifies the event type, behavior, and contextual information to be used in processing the event.
65       *
66       * @return The return value of dispatchEvent indicates whether any of the listeners which handled the event called
67       *         preventDefault. If preventDefault was called the value is false, else the value is true.
68       *
69       * @throws EventException
70       *             UNSPECIFIED_EVENT_TYPE_ERR: Raised if the Event's type was not specified by initializing the event
71       *             before dispatchEvent was called. Specification of the Event's type as null or an empty string will
72       *             also trigger this exception.
73       */
74      boolean jsFunction_dispatchEvent(Scriptable evt) throws EventException;
75  }
