20  package com.meterware.httpunit.protocol;
21  
22  import java.io.IOException;
23  import java.io.InputStream;
24  import java.io.OutputStream;
25  
26  /**
27   * A POST-method message body which is MIME-encoded. This is used when uploading files, and is selected when the enctype
28   * parameter of a form is set to "multi-part/form-data".
29   **/
30  class MimeEncodedMessageBody extends MessageBody {
31  
32      MimeEncodedMessageBody(String characterSet) {
33          super(characterSet);
34      }
35  
36      /**
37       * Returns the content type of this message body.
38       **/
39      @Override
40      public String getContentType() {
41          return "multipart/form-data; boundary=" + BOUNDARY;
42      }
43  
44      /**
45       * Transmits the body of this request as a sequence of bytes.
46       **/
47      @Override
48      public void writeTo(OutputStream outputStream, ParameterCollection parameters) throws IOException {
49          MimeEncoding encoding = new MimeEncoding(outputStream);
50          parameters.recordParameters(encoding);
51          encoding.sendClose();
52      }
53  
54      private static final String BOUNDARY = "--HttpUnit-part0-aSgQ2M";
55      private static final byte[] CRLF = { 0x0d, 0x0A };
56  
57      private String encode(String string) {
58          char[] chars = string.toCharArray();
59          StringBuilder sb = new StringBuilder(chars.length + 20);
60          for (char element : chars) {
61              if (element == '\\') {
62                  sb.append("\\\\"); // accomodate MS-DOS file paths ??? is this safe??
63              } else {
64                  sb.append(element);
65              }
66          }
67          return sb.toString();
68      }
69  
70      private void writeLn(OutputStream os, String value, String encoding) throws IOException {
71          os.write(value.getBytes(encoding));
72          os.write(CRLF);
73      }
74  
75      private void writeLn(OutputStream os, String value) throws IOException {
76          writeLn(os, value, getCharacterSet());
77      }
78  
79      class MimeEncoding implements ParameterProcessor {
80  
81          public MimeEncoding(OutputStream outputStream) {
82              _outputStream = outputStream;
83          }
84  
85          public void sendClose() throws IOException {
86              writeLn(_outputStream, "--" + BOUNDARY + "--");
87          }
88  
89          /**
90           * add the parameter with the given name value and characterSet
91           *
92           * @param name
93           *            - the name
94           * @param value
95           *            - the value to set
96           * @param characterSet
97           *            - the name of the characterSet to use
98           */
99          @Override
100         public void addParameter(String name, String value, String characterSet) throws IOException {
101             if (name == null || name.isEmpty() || value == null) {
102                 return;
103             }
104             writeLn(_outputStream, "--" + BOUNDARY);
105             writeLn(_outputStream, "Content-Disposition: form-data; name=\"" + name + '"'); // XXX need to handle
106             // non-ascii names here
107             writeLn(_outputStream, "Content-Type: text/plain; charset=" + getCharacterSet());
108             writeLn(_outputStream, "");
109             writeLn(_outputStream, fixLineEndings(value), getCharacterSet());
110         }
111 
112         private static final char CR = 0x0D;
113         private static final char LF = 0x0A;
114 
115         private String fixLineEndings(String value) {
116             StringBuilder sb = new StringBuilder();
117             char[] chars = value.toCharArray();
118             for (int i = 0; i < chars.length; i++) {
119                 if (chars[i] == CR || chars[i] == LF && (i == 0 || chars[i - 1] != CR)) {
120                     sb.append(CR).append(LF);
121                 } else {
122                     sb.append(chars[i]);
123                 }
124             }
125             return sb.toString();
126         }
127 
128         @Override
129         public void addFile(String name, UploadFileSpec spec) throws IOException {
130             byte[] buffer = new byte[8 * 1024];
131 
132             writeLn(_outputStream, "--" + BOUNDARY);
133             writeLn(_outputStream, "Content-Disposition: form-data; name=\"" + encode(name) + "\"; filename=\""
134                     + encode(spec.getFileName()) + '"'); // XXX need to handle non-ascii names here
135             writeLn(_outputStream, "Content-Type: " + spec.getContentType());
136             writeLn(_outputStream, "");
137 
138             InputStream in = spec.getInputStream();
139             int count = 0;
140             do {
141                 _outputStream.write(buffer, 0, count);
142                 count = in.read(buffer, 0, buffer.length);
143             } while (count != -1);
144 
145             in.close();
146             writeLn(_outputStream, "");
147         }
148 
149         private OutputStream _outputStream;
150     }
