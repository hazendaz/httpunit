20  package com.meterware.httpunit.protocol;
21  
22  import java.net.URLEncoder;
23  
24  /**
25   * @author <a href="mailto:russgold@httpunit.org">Russell Gold</a>
26   */
27  public class URLEncodedString implements ParameterProcessor {
28  
29      public static final int DEFAULT_BUFFER_SIZE = 128;
30  
31      private StringBuilder _buffer = new StringBuilder(DEFAULT_BUFFER_SIZE);
32  
33      private boolean _haveParameters = false;
34  
35      public String getString() {
36          return _buffer.toString();
37      }
38  
39      @Override
40      public void addParameter(String name, String value, String characterSet) {
41          if (_haveParameters) {
42              _buffer.append('&');
43          }
44          _buffer.append(encode(name, characterSet));
45          if (value != null) {
46              _buffer.append('=').append(encode(value, characterSet));
47          }
48          _haveParameters = true;
49      }
50  
51      @Override
52      public void addFile(String parameterName, UploadFileSpec fileSpec) {
53          throw new RuntimeException("May not URL-encode a file upload request");
54      }
55  
56      /**
57       * Returns a URL-encoded version of the string.
58       **/
59      private String encode(String source, String characterSet) {
60          try {
61              return URLEncoder.encode(source, characterSet);
62          } catch (java.io.UnsupportedEncodingException e) {
63              return "???"; // XXX should pass the exception through as IOException ultimately
64          }
65      }
66  
67  }
