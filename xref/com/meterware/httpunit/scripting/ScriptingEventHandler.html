8   package com.meterware.httpunit.scripting;
9   
10  /**
11   * interface for every object that may have excutable events and their scripts attached.
12   */
13  public interface ScriptingEventHandler {
14  
15      /**
16       * run the Script for the given Event.
17       *
18       * @param eventScript
19       *            the event script
20       *
21       * @return true if the script is empty or the result of the script
22       *
23       * @deprecated since 1.7
24       */
25      @Deprecated
26      boolean doEvent(String eventScript);
27  
28      /**
29       * run the Script for the given Event.
30       *
31       * @param eventScript
32       *            the event script
33       *
34       * @return true if the script is empty or the result of the script
35       */
36      boolean doEventScript(String eventScript);
37  
38      /**
39       * handle the event with the given name by getting the attribute and then executing the eventScript for it.
40       *
41       * @param eventName
42       *            the event name
43       *
44       * @return the result of doEventScript
45       */
46      boolean handleEvent(String eventName);
47  
48  }
