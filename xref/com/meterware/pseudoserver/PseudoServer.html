115      * create a PseudoServer with the given socketTimeout
116      *
117      * @param socketTimeout
118      *            - the time out to use
119      */
120     public PseudoServer(int socketTimeout) {
121         _socketTimeout = socketTimeout;
122         _serverNum = ++_numServers;
123 
124         try {
125             _serverSocket = new ServerSocket(0);
126             _serverSocket.setSoTimeout(1000);
127         } catch (IOException e) {
128             System.out.println("Error while creating socket: " + e);
129             throw new RuntimeException(e);
130         }
131         Thread t = new Thread("PseudoServer " + _serverNum) {
132             @Override
133             public void run() {
134                 while (_active) {
135                     try {
136                         handleNewConnection(_serverSocket.accept());
137                         Thread.sleep(20);
138                     } catch (InterruptedIOException e) {
139                     } catch (IOException e) {
140                         System.out.println("Error in pseudo server: " + e);
141                         HttpUnitUtils.handleException(e);
142                     } catch (InterruptedException e) {
143                         System.out.println("Interrupted. Shutting down");
144                         _active = false;
145                     }
146                 }
147                 try {
148                     _serverSocket.close();
149                 } catch (IOException e) {
150                     System.out.println("Error while closing socket: " + e);
151                 }
152                 debug("Pseudoserver shutting down");
153             }
154         };
155         debug("Starting pseudoserver");
156         t.start();
157     }

164     private void debug(String message) {
165         if (!_debug) {
166             return;
167         }
168         message = replaceDebugToken(message, "thread", "thread (" + Thread.currentThread().getName() + ")");
169         message = replaceDebugToken(message, "server", "server " + _serverNum);
170         System.out.println("** " + message);
171     }

173     private static String replaceDebugToken(String message, String token, String replacement) {
174         return !message.contains(token) ? message : message.replaceFirst(token, replacement);
175     }

282     private String asResourceName(String rawName) {
283         if (rawName.startsWith("http:") || rawName.startsWith("/")) {
284             return escape(rawName);
285         }
286         return escape("/" + rawName);
287     }

289     private static String escape(String urlString) {
290         if (urlString.indexOf(' ') < 0) {
291             return urlString;
292         }
293         StringBuilder sb = new StringBuilder();
294 
295         int start = 0;
296         do {
297             int index = urlString.indexOf(' ', start);
298             if (index < 0) {
299                 sb.append(urlString.substring(start));
300                 break;
301             }
302             sb.append(urlString.substring(start, index)).append("%20");
303             start = index + 1;
304         } while (true);
305         return sb.toString();
306     }

308     private void handleNewConnection(final Socket socket) {
309         Thread t = new Thread("PseudoServer " + _serverNum + " connection " + (++_connectionNum)) {
310             @Override
311             public void run() {
312                 try {
313                     serveRequests(socket);
314                 } catch (IOException e) {
315                     e.printStackTrace(); // To change body of catch statement use Options | File Templates.
316                 }
317             }
318         };
319         t.start();
320     }

322     private void serveRequests(Socket socket) throws IOException {
323         socket.setSoTimeout(_socketTimeout);
324         socket.setTcpNoDelay(true);
325 
326         debug("Created server thread " + socket.getInetAddress() + ':' + socket.getPort());
327         final BufferedInputStream inputStream = new BufferedInputStream(socket.getInputStream());
328         final HttpResponseStream outputStream = new HttpResponseStream(socket.getOutputStream());
329 
330         try {
331             while (_active) {
332                 HttpRequest request = new HttpRequest(inputStream);
333                 boolean keepAlive = respondToRequest(request, outputStream);
334                 if (!keepAlive) {
335                     break;
336                 }
337                 while (_active && 0 == inputStream.available()) {
338                     try {
339                         Thread.sleep(INPUT_POLL_INTERVAL);
340                     } catch (InterruptedException e) {
341                     }
342                 }
343             }
344         } catch (IOException e) {
345             outputStream.restart();
346             outputStream.setProtocol("HTTP/1.0");
347             outputStream.setResponse(HttpURLConnection.HTTP_BAD_REQUEST, e.toString());
348         }
349         debug("Closing server thread");
350         outputStream.close();
351         socket.close();
352         debug("Server thread closed");
353     }

355     /**
356      * respond to the given request
357      *
358      * @param request
359      *            - the request
360      * @param response
361      *            - the response stream
362      *
363      * @return
364      */
365     private boolean respondToRequest(HttpRequest request, HttpResponseStream response) {
366         debug("Server thread handling request: " + request);
367         boolean keepAlive = isKeepAlive(request);
368         WebResource resource = null;
369         try {
370             response.restart();
371             response.setProtocol(getResponseProtocol(request));
372             resource = getResource(request);
373             if (resource == null) {
374                 // what resource could not be find?
375                 String uri = request.getURI();
376                 // 404 - Not Found error code
377                 int errorCode = HttpURLConnection.HTTP_NOT_FOUND;
378                 // typical 404 error Message
379                 String errorMessage = "unable to find " + uri;
380                 // make sure there is a resource and
381                 // next time we'll take it from the resource Cache
382                 resource = setErrorResource(uri, errorCode, errorMessage);
383                 // set the errorCode for this response
384                 response.setResponse(errorCode, errorMessage);
385             } else if (resource.getResponseCode() != HttpURLConnection.HTTP_OK) {
386                 response.setResponse(resource.getResponseCode(), "");
387             }
388             if (resource.closesConnection()) {
389                 keepAlive = false;
390             }
391             String[] headers = resource.getHeaders();
392             for (String header : headers) {
393                 debug("Server thread sending header: " + header);
394                 response.addHeader(header);
395             }
396         } catch (UnknownMethodException e) {
397             response.setResponse(HttpURLConnection.HTTP_BAD_METHOD, "unsupported method: " + e.getMethod());
398         } catch (Throwable t) {
399             t.printStackTrace();
400             response.setResponse(HttpURLConnection.HTTP_INTERNAL_ERROR, t.toString());
401         }
402         try {
403             response.write(resource);
404         } catch (IOException e) {
405             System.out.println("*** Failed to send reply: " + e);
406         }
407         return keepAlive;
408     }

410     private boolean isKeepAlive(HttpRequest request) {
411         return request.wantsKeepAlive() && _maxProtocolLevel.equals("1.1");
412     }

414     private String getResponseProtocol(HttpRequest request) {
415         return _maxProtocolLevel.equalsIgnoreCase("1.1") ? request.getProtocol() : "HTTP/1.0";
416     }

418     /**
419      * get the resource for the given request by first trying to look it up in the cache then depending on the type of
420      * request PseudoServlet and the method / command e.g. GET/HEAD finally the extension of the uri ".zip" ".class" and
421      * ".jar" are handled
422      *
423      * @param request
424      *
425      * @return the WebResource or null if non of the recipes above will lead to a valid resource
426      *
427      * @throws IOException
428      */
429     private WebResource getResource(HttpRequest request) throws IOException {
430         Object resource = _resources.get(request.getURI());
431         if (resource == null) {
432             resource = _resources.get(withoutParameters(request.getURI()));
433         }
434 
435         // check the method of the request
436         String command = request.getCommand();
437         if ((command.equals("GET") || command.equals("HEAD")) && resource instanceof WebResource) {
438             return (WebResource) resource;
439         }
440         if (resource instanceof PseudoServlet) {
441             return getResource((PseudoServlet) resource, request);
442         }
443         if (request.getURI().endsWith(".class")) {
444             for (Iterator iterator = _classpathDirs.iterator(); iterator.hasNext();) {
445                 String directory = (String) iterator.next();
446                 if (request.getURI().startsWith(directory)) {
447                     String resourceName = request.getURI().substring(directory.length() + 1);
448                     return new WebResource(getClass().getClassLoader().getResourceAsStream(resourceName),
449                             "application/class", 200);
450                 }
451             }
452         } else if (request.getURI().endsWith(".zip") || request.getURI().endsWith(".jar")) {
453             for (Iterator iterator = _classpathDirs.iterator(); iterator.hasNext();) {
454                 String directory = (String) iterator.next();
455                 if (request.getURI().startsWith(directory)) {
456                     String resourceName = request.getURI().substring(directory.length() + 1);
457                     String classPath = System.getProperty("java.class.path");
458                     StringTokenizer st = new StringTokenizer(classPath, ":;,");
459                     while (st.hasMoreTokens()) {
460                         String file = st.nextToken();
461                         if (file.endsWith(resourceName)) {
462                             File f = new File(file);
463                             return new WebResource(new FileInputStream(f), "application/zip", 200);
464                         }
465                     }
466                 }
467             }
468         }
469         return null;
470     }

472     private String withoutParameters(String uri) {
473         return uri.indexOf('?') < 0 ? uri : uri.substring(0, uri.indexOf('?'));
474     }

476     private WebResource getResource(PseudoServlet servlet, HttpRequest request) throws IOException {
477         servlet.init(request);
478         return servlet.getResponse(request.getCommand());
479     }

485 class HttpResponseStream {
486 
487     final private static String CRLF = "\r\n";
488 
489     void restart() {
490         _headersWritten = false;
491         _headers.clear();
492         _responseCode = HttpURLConnection.HTTP_OK;
493         _responseText = "OK";
494     }
495 
496     void close() throws IOException {
497         flushHeaders();
498         _pw.close();
499     }
500 
501     HttpResponseStream(OutputStream stream) {
502         _stream = stream;
503         try {
504             setCharacterSet("us-ascii");
505         } catch (UnsupportedEncodingException e) {
506             _pw = new PrintWriter(new OutputStreamWriter(_stream));
507         }
508     }
509 
510     void setProtocol(String protocol) {
511         _protocol = protocol;
512     }
513 
514     /**
515      * set the response to the given response Code
516      *
517      * @param responseCode
518      * @param responseText
519      */
520     void setResponse(int responseCode, String responseText) {
521         _responseCode = responseCode;
522         _responseText = responseText;
523     }
524 
525     void addHeader(String header) {
526         _headers.addElement(header);
527     }
528 
529     void write(String contents, String charset) throws IOException {
530         flushHeaders();
531         setCharacterSet(charset);
532         sendText(contents);
533     }
534 
535     void write(WebResource resource) throws IOException {
536         flushHeaders();
537         if (resource != null) {
538             resource.writeTo(_stream);
539         }
540         _stream.flush();
541     }
542 
543     private void setCharacterSet(String characterSet) throws UnsupportedEncodingException {
544         if (_pw != null) {
545             _pw.flush();
546         }
547         _pw = new PrintWriter(new OutputStreamWriter(_stream, characterSet));
548     }
549 
550     private void flushHeaders() {
551         if (!_headersWritten) {
552             sendResponse(_responseCode, _responseText);
553             for (Enumeration e = _headers.elements(); e.hasMoreElements();) {
554                 sendLine((String) e.nextElement());
555             }
556             sendText(CRLF);
557             _headersWritten = true;
558             _pw.flush();
559         }
560     }
561 
562     private void sendResponse(int responseCode, String responseText) {
563         sendLine(_protocol + ' ' + responseCode + ' ' + responseText);
564     }
565 
566     private void sendLine(String text) {
567         sendText(text);
568         sendText(CRLF);
569     }
570 
571     private void sendText(String text) {
572         _pw.write(text);
573     }
574 
575     private OutputStream _stream;
576     private PrintWriter _pw;
577 
578     private Vector _headers = new Vector();
579     private String _protocol = "HTTP/1.0";
580     private int _responseCode = HttpURLConnection.HTTP_OK;
581     private String _responseText = "OK";
582 
583     private boolean _headersWritten;
584 
585 }

587 class RecordingOutputStream extends OutputStream {
588 
589     private OutputStream _nestedStream;
590     private PrintStream _log;
591 
592     public RecordingOutputStream(OutputStream nestedStream, PrintStream log) {
593         _nestedStream = nestedStream;
594         _log = log;
595     }
596 
597     @Override
598     public void write(int b) throws IOException {
599         _nestedStream.write(b);
600         _log.println("sending " + Integer.toHexString(b));
601     }
602 
603     @Override
604     public void write(byte b[], int offset, int len) throws IOException {
605         _nestedStream.write(b, offset, len);
606         _log.print("sending");
607         for (int i = offset; i < offset + len; i++) {
608             _log.print(' ' + Integer.toHexString(b[i]));
609         }
610         _log.println();
611     }
612 }
