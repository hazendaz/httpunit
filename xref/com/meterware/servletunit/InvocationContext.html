20  package com.meterware.servletunit;
21  
22  import com.meterware.httpunit.FrameSelector;
23  import com.meterware.httpunit.WebResponse;
24  
25  import jakarta.servlet.*;
26  import jakarta.servlet.http.HttpServletRequest;
27  import jakarta.servlet.http.HttpServletResponse;
28  
29  import java.io.IOException;
30  
31  /**
32   * An interface which represents the invocation of a servlet.
33   *
34   * @author <a href="mailto:russgold@httpunit.org">Russell Gold</a>
35   **/
36  public interface InvocationContext {
37  
38      /**
39       * Returns the request to be processed by the servlet or filter.
40       **/
41      HttpServletRequest getRequest();
42  
43      /**
44       * Returns the response which the servlet or filter should modify during its operation.
45       **/
46      HttpServletResponse getResponse();
47  
48      /**
49       * Invokes the current servlet or filter.
50       *
51       * @since 1.6
52       */
53      void service() throws ServletException, IOException;
54  
55      /**
56       * Returns the selected servlet, initialized to provide access to sessions and servlet context information. Only
57       * valid to call if {@link #isFilterActive} returns false.
58       **/
59      Servlet getServlet() throws ServletException;
60  
61      /**
62       * Returns the final response from the servlet. Note that this method should only be invoked after all processing
63       * has been done to the servlet response.
64       **/
65      WebResponse getServletResponse() throws IOException;
66  
67      /**
68       * Returns the target frame for the original request.
69       *
70       * @since 1.6
71       */
72      FrameSelector getFrame();
73  
74      /**
75       * Adds a request dispatcher to this context to simulate an include request.
76       */
77      void pushIncludeRequest(RequestDispatcher rd, HttpServletRequest request, HttpServletResponse response)
78              throws ServletException;
79  
80      /**
81       * Adds a request dispatcher to this context to simulate a forward request.
82       */
83      void pushForwardRequest(RequestDispatcher rd, HttpServletRequest request, HttpServletResponse response)
84              throws ServletException;
85  
86      /**
87       * Removes the top request dispatcher or filter from this context.
88       */
89      void popRequest();
90  
91      /**
92       * Returns true if the current context is a filter, rather than a servlet.
93       *
94       * @since 1.6
95       */
96      boolean isFilterActive();
97  
98      /**
99       * Returns the current active filter object. Only valid to call if {@link #isFilterActive} returns true.
100      *
101      * @since 1.6
102      */
103     Filter getFilter() throws ServletException;
104 
105     /**
106      * Returns the current filter chain. Only valid to call if {@link #isFilterActive} returns true.
107      *
108      * @since 1.6
109      */
110     FilterChain getFilterChain();
111 
112     /**
113      * Pushes the current filter onto the execution stack and switches to the next filter or the selected servlet. This
114      * can be used to simulate the effect of the {@link jakarta.servlet.FilterChain#doFilter doFilter} call. <br>
115      * <b>Note:</b> this method specifies {@link ServletRequest} and {@link ServletResponse} because those are the types
116      * passed to {@link Filter#doFilter}; however, HttpUnit requires the objects to implement {@link HttpServletRequest}
117      * and {@link HttpServletResponse} because they will eventually be passed to an
118      * {@link jakarta.servlet.http.HttpServlet}.
119      *
120      * @param request
121      *            the request to pass to the next filter. May be a wrapper.
122      * @param response
123      *            the response object to pass to the next filter. May be a wrapper.
124      *
125      * @since 1.6
126      */
127     void pushFilter(ServletRequest request, ServletResponse response);
128 
129 }
