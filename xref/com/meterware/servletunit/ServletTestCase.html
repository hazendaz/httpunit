20  package com.meterware.servletunit;
21  
22  import junit.framework.TestCase;
23  
24  /**
25   * A base class for test cases to be run via {@link JUnitServlet JUnitServlet}.
26   *
27   * @author <a href="mailto:russgold@acm.org">Russell Gold</a>
28   **/
29  public abstract class ServletTestCase extends TestCase {
30  
31      /**
32       * construct a ServletTestCase with the given name
33       *
34       * @param name
35       */
36      protected ServletTestCase(String name) {
37          super(name);
38      }
39  
40      /**
41       * Returns a client object which can access the servlet context in which this test is running.
42       */
43      final protected ServletUnitClient newClient() {
44          if (_invocationContextFactory == null) {
45              throw new RuntimeException(
46                      "ServletTestCase.newClient called before setInvocationContextFactory was called");
47          }
48          return ServletUnitClient.newClient(_invocationContextFactory);
49      }
50  
51      /**
52       * set the invocation context factory to be used
53       *
54       * @param invocationContextFactory
55       */
56      static void setInvocationContextFactory(InvocationContextFactory invocationContextFactory) {
57          if (invocationContextFactory == null) {
58              throw new RuntimeException(
59                      "setInvocationContextFactory called with null invocationContextFactory parameter");
60          }
61          _invocationContextFactory = invocationContextFactory;
62      }
63  
64      private static InvocationContextFactory _invocationContextFactory;
65  }
