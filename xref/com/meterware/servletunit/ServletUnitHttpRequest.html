78      /**
79       * Constructs a ServletUnitHttpRequest from a WebRequest object.
80       **/
81      ServletUnitHttpRequest(ServletMetaData servletRequest, WebRequest request, ServletUnitContext context,
82              Dictionary clientHeaders, byte[] messageBody) throws MalformedURLException {
83          if (context == null) {
84              throw new IllegalArgumentException("Context must not be null");
85          }
86  
87          _servletRequest = servletRequest;
88          _request = request;
89          _context = context;
90          _headers = new WebClient.HeaderDictionary();
91          _headers.addEntries(clientHeaders);
92          _headers.addEntries(request.getHeaders());
93          setCookiesFromHeader(_headers);
94          _messageBody = messageBody;
95          _protocol = request.getURL().getProtocol().toLowerCase();
96          _secure = _protocol.endsWith("s");
97          _serverName = request.getURL().getHost();
98          _serverPort = request.getURL().getPort();
99          if (_serverPort == -1) {
100             _serverPort = request.getURL().getDefaultPort();
101         }
102 
103         _requestContext = new RequestContext(request.getURL());
104         String contentTypeHeader = (String) _headers.get("Content-Type");
105         if (contentTypeHeader != null) {
106             String[] res = HttpUnitUtils.parseContentTypeHeader(contentTypeHeader);
107             _charset = res[1];
108             _requestContext.setMessageEncoding(_charset);
109         }
110         if (_headers.get("Content-Length") == null) {
111             _headers.put("Content-Length", Integer.toString(messageBody.length));
112         }
113 
114         boolean setBody =
115                 // pre [ 1509117 ] getContentType()
116                 // _messageBody != null && (_contentType == null || _contentType.indexOf( "x-www-form-urlencoded" ) >= 0
117                 // );
118                 // patch version:
119                 _messageBody != null
120                 && (contentTypeHeader == null || contentTypeHeader.indexOf("x-www-form-urlencoded") >= 0);
121         if (setBody) {
122             _requestContext.setMessageBody(_messageBody);
123         }
124     }

543     /**
544      * Parses the accept-language header to obtain a vector of preferred locales
545      *
546      * @return the preferred locales, sorted by qvalue
547      */
548     private Vector getPreferredLocales() {
549         if (_locales == null) {
550             _locales = new Vector();
551             String languages = getHeader("accept-language");
552             if (languages == null) {
553                 _locales.add(Locale.getDefault());
554             } else {
555                 StringTokenizer st = new StringTokenizer(languages, ",");
556                 ArrayList al = new ArrayList();
557                 while (st.hasMoreTokens()) {
558                     String token = st.nextToken();
559                     al.add(new PrioritizedLocale(token));
560                 }
561                 Collections.sort(al);
562                 for (Iterator iterator = al.iterator(); iterator.hasNext();) {
563                     _locales.add(((PrioritizedLocale) iterator.next()).getLocale());
564                 }
565             }
566         }
567         return _locales;
568     }

579     /**
580      * Returns a RequestDispatcher object that acts as a wrapper for the resource located at the given path. A
581      * RequestDispatcher object can be used to forward a request to the resource or to include the resource in a
582      * response. The resource can be dynamic or static. The pathname specified may be relative, although it cannot
583      * extend outside the current servlet context. If the path begins with a "/" it is interpreted as relative to the
584      * current context root. This method returns null if the servlet container cannot return a RequestDispatcher. The
585      * difference between this method and ServletContext.getRequestDispatcher(java.lang.String) is that this method can
586      * take a relative path.
587      **/
588     @Override
589     public RequestDispatcher getRequestDispatcher(String path) {
590         try {
591             if (!path.startsWith("/")) {
592                 path = combinedPath(getServletPath(), path);
593             }
594             return _servletRequest.getServlet().getServletConfig().getServletContext().getRequestDispatcher(path);
595         } catch (ServletException e) {
596             return null;
597         }
598     }

600     private String combinedPath(String basePath, String relativePath) {
601         if (basePath.indexOf('/') < 0) {
602             return relativePath;
603         }
604         return basePath.substring(0, basePath.lastIndexOf('/')) + '/' + relativePath;
605     }

688     /**
689      * Reconstructs the URL the client used to make the request. The returned URL contains a protocol, server name, port
690      * number, and server path, but it does not include query string parameters. Because this method returns a
691      * StringBuffer, not a string, you can modify the URL easily, for example, to append query parameters. This method
692      * is useful for creating redirect messages and for reporting errors.
693      *
694      * @since 1.3
695      */
696     @Override
697     public StringBuffer getRequestURL() {
698         StringBuilder url = new StringBuilder();
699         try {
700             url.append(_request.getURL().getProtocol()).append("://");
701             url.append(_request.getURL().getHost());
702             String portPortion = _request.getURL().getPort() == -1 ? "" : ":" + _request.getURL().getPort();
703             url.append(portPortion);
704             url.append(_request.getURL().getPath());
705         } catch (MalformedURLException e) {
706             throw new RuntimeException("unable to read URL from request: " + _request);
707         }
708         return new StringBuffer(url);
709     }

799     private void setCookiesFromHeader(Dictionary clientHeaders) {
800         String cookieHeader = (String) clientHeaders.get("Cookie");
801         if (cookieHeader == null) {
802             return;
803         }
804 
805         StringTokenizer st = new StringTokenizer(cookieHeader, ",;=", true);
806         String lastToken = st.nextToken();
807         while (st.hasMoreTokens()) {
808             String token = st.nextToken();
809             if (token.equals("=") && st.hasMoreTokens()) {
810                 addCookie(new Cookie(lastToken.trim(), st.nextToken().trim()));
811             }
812             lastToken = token;
813         }

816     static class PrioritizedLocale implements Comparable {
817 
818         private Locale _locale;
819         private float _priority;
820 
821         PrioritizedLocale(String languageSpec) {
822             int semiIndex = languageSpec.indexOf(';');
823             if (semiIndex < 0) {
824                 _priority = 1;
825                 _locale = parseLocale(languageSpec);
826             } else {
827                 _priority = Float.parseFloat(languageSpec.substring(languageSpec.indexOf('=', semiIndex) + 1));
828                 _locale = parseLocale(languageSpec.substring(0, semiIndex));
829             }
830         }
831 
832         private Locale parseLocale(String range) {
833             range = range.trim();
834             int dashIndex = range.indexOf('-');
835             if (dashIndex < 0) {
836                 return new Locale(range, "");
837             }
838             return new Locale(range.substring(0, dashIndex), range.substring(dashIndex + 1));
839         }
840 
841         public Locale getLocale() {
842             return _locale;
843         }
844 
845         @Override
846         public int compareTo(Object o) {
847             if (!(o instanceof PrioritizedLocale)) {
848                 throw new IllegalArgumentException("may only combine with other prioritized locales");
849             }
850             PrioritizedLocale other = (PrioritizedLocale) o;
851             return _priority == other._priority ? _locale.getLanguage().compareTo(other._locale.getLanguage())
852                     : _priority < other._priority ? +1 : -1;
853         }
