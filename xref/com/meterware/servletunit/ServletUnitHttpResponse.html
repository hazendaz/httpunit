195     private String asDateHeaderValue(long date) {
196         Date value = new Date(date);
197         SimpleDateFormat formatter = new SimpleDateFormat(RFC1123_DATE_SPEC, Locale.US);
198         formatter.setTimeZone(TimeZone.getTimeZone("Greenwich Mean Time"));
199         return formatter.format(value);

227     /**
228      * Returns a {@link ServletOutputStream} suitable for writing binary data in the response. The servlet engine does
229      * not encode the binary data.
230      *
231      * @exception IllegalStateException
232      *                if you have already called the <code>getWriter</code> method
233      **/
234     @Override
235     public ServletOutputStream getOutputStream() throws IOException {
236         if (_writer != null) {
237             throw new IllegalStateException("Tried to create output stream; writer already exists");
238         }
239         if (_servletStream == null) {
240             _outputStream = new ByteArrayOutputStream();
241             _servletStream = new ServletUnitOutputStream(_outputStream);
242         }
243         return _servletStream;
244     }

246     /**
247      * Returns a <code>PrintWriter</code> object that you can use to send character text to the client. The character
248      * encoding used is the one specified in the <code>charset=</code> property of the {@link #setContentType} method,
249      * which you must call <i>before</i> you call this method.
250      * <p>
251      * If necessary, the MIME type of the response is modified to reflect the character encoding used.
252      * <p>
253      * You cannot use this method if you have already called {@link #getOutputStream} for this
254      * <code>ServletResponse</code> object.
255      *
256      * @exception UnsupportedEncodingException
257      *                if the character encoding specified in <code>setContentType</code> cannot be used
258      * @exception IllegalStateException
259      *                if the <code>getOutputStream</code> method has already been called for this response object; in
260      *                that case, you can't use this method
261      **/
262     @Override
263     public PrintWriter getWriter() throws UnsupportedEncodingException {
264         if (_servletStream != null) {
265             throw new IllegalStateException("Tried to create writer; output stream already exists");
266         }
267         if (_writer == null) {
268             _outputStream = new ByteArrayOutputStream();
269             _writer = new PrintWriter(new OutputStreamWriter(_outputStream, getCharacterEncoding()));
270         }
271         return _writer;
272     }

369     /**
370      * Sets the locale of the response, setting the headers (including the Content-Type's charset) as appropriate. This
371      * method should be called before a call to getWriter(). By default, the response locale is the default locale for
372      * the server.
373      **/
374     @Override
375     public void setLocale(Locale locale) {
376         _locale = locale;
377         if (_encoding == null) {
378             for (Iterator it = ENCODING_MAP.entrySet().iterator(); it.hasNext();) {
379                 Map.Entry entry = (Map.Entry) it.next();
380                 String locales = (String) entry.getValue();
381                 if (locales.indexOf(locale.getLanguage()) >= 0 || locales.indexOf(locale.toString()) >= 0) {
382                     _encoding = (String) entry.getKey();
383                     return;
384                 }
385             }
386         }
387     }

445     public String[] getHeaderFieldNames() {
446         if (!_headersComplete) {

546     private void completeHeaders() {
547         if (_headersComplete) {
548             return;
549         }
550         addCookieHeader();
551         // BR 3301056 ServletUnit handling Content-Type incorrectly
552         if (getHeaderFieldDirect("Content-Type") == null) {
553             setHeader("Content-Type", _contentType + "; charset=" + getCharacterEncoding());
554         }
555         _headersComplete = true;
556     }
557 
558     private void addCookieHeader() {
559         if (_cookies.isEmpty()) {
560             return;
561         }
562 
563         StringBuilder sb = new StringBuilder();
564         for (Enumeration e = _cookies.elements(); e.hasMoreElements();) {
565             Cookie cookie = (Cookie) e.nextElement();
566             sb.append(cookie.getName()).append('=').append(cookie.getValue());
567             if (cookie.getPath() != null) {
568                 sb.append(";path=").append(cookie.getPath());
569             }
570             if (cookie.getDomain() != null) {
571                 sb.append(";domain=").append(cookie.getDomain());
572             }
573             if (e.hasMoreElements()) {
574                 sb.append(',');
575             }
576         }
577         setHeader("Set-Cookie", sb.toString());
578     }
